76       """Check if an instance of a dataclass contains the default repr. 
77    
78       Args: 
79           obj (object): A dataclass instance. 
80    
81       Returns: 
82           bool: True if the default repr is used, False if there is a custom repr. 
83       """
84       # Digging in to a lot of internals here
85       # Catching all exceptions in case something is missing on a non CPython implementation
86       try:
87           return obj.__repr__.__code__.co_filename == dataclasses.__file__
88       except Exception:  # pragma: no coverage
89           return False

95   def _has_default_namedtuple_repr(obj: object) -> bool:
96       """Check if an instance of namedtuple contains the default repr 
97    
98       Args: 
99           obj (object): A namedtuple 
100   
101      Returns: 
102          bool: True if the default repr is used, False if there's a custom repr. 
103      """
104      obj_file = None
105      try:
106          obj_file = inspect.getfile(obj.__repr__)
107      except (OSError, TypeError):
108          # OSError handles case where object is defined in __main__ scope, e.g. REPL - no filename available.
109          # TypeError trapped defensively, in case of object without filename slips through.
110          pass
111      default_repr_file = inspect.getfile(_dummy_namedtuple.__repr__)
112      return obj_file == default_repr_file

115  def _ipy_display_hook(
116      value: Any,
117      console: Optional["Console"] = None,
118      overflow: "OverflowMethod" = "ignore",
119      crop: bool = False,
120      indent_guides: bool = False,
121      max_length: Optional[int] = None,
122      max_string: Optional[int] = None,
123      expand_all: bool = False,
124  ) -> None:
125      # needed here to prevent circular import:
126      from ._inspect import is_object_one_of_types
127      from .console import ConsoleRenderable
128  
129      # always skip rich generated jupyter renderables or None values
130      if _safe_isinstance(value, JupyterRenderable) or value is None:
131          return
132  
133      console = console or get_console()
134      if console.is_jupyter:
135          # Delegate rendering to IPython if the object (and IPython) supports it
136          #  https://ipython.readthedocs.io/en/stable/config/integrating.html#rich-display
137          ipython_repr_methods = [
138              "_repr_html_",
139              "_repr_markdown_",
140              "_repr_json_",
141              "_repr_latex_",
142              "_repr_jpeg_",
143              "_repr_png_",
144              "_repr_svg_",
145              "_repr_mimebundle_",
146          ]
147          for repr_method in ipython_repr_methods:
148              method = getattr(value, repr_method, None)
149              if inspect.ismethod(method):
150                  # Calling the method ourselves isn't ideal. The interface for the `_repr_*_` methods
151                  #  specifies that if they return None, then they should not be rendered
152                  #  by the notebook.
153                  try:
154                      repr_result = method()
155                  except Exception:
156                      continue  # If the method raises, treat it as if it doesn't exist, try any others
157                  if repr_result is not None:
158                      return  # Delegate rendering to IPython
159  
160          # When in a Jupyter notebook let's avoid the display of some specific classes,
161          # as they result in the rendering of useless and noisy lines such as `<Figure size 432x288 with 1 Axes>`.
162          # What does this do?
163          # --> if the class has "matplotlib.artist.Artist" in its hierarchy for example, we don't render it.
164          if is_object_one_of_types(value, JUPYTER_CLASSES_TO_NOT_RENDER):
165              return
166  
167      # certain renderables should start on a new line
168      if _safe_isinstance(value, ConsoleRenderable):
169          console.line()
170  
171      console.print(
172          value
173          if _safe_isinstance(value, RichRenderable)
174          else Pretty(
175              value,
176              overflow=overflow,
177              indent_guides=indent_guides,
178              max_length=max_length,
179              max_string=max_string,
180              expand_all=expand_all,
181              margin=12,
182          ),
183          crop=crop,
184          new_line_start=True,
185      )

198  def install(
199      console: Optional["Console"] = None,
200      overflow: "OverflowMethod" = "ignore",
201      crop: bool = False,
202      indent_guides: bool = False,
203      max_length: Optional[int] = None,
204      max_string: Optional[int] = None,
205      expand_all: bool = False,
206  ) -> None:
207      """Install automatic pretty printing in the Python REPL. 
208   
209      Args: 
210          console (Console, optional): Console instance or ``None`` to use global console. Defaults to None. 
211          overflow (Optional[OverflowMethod], optional): Overflow method. Defaults to "ignore". 
212          crop (Optional[bool], optional): Enable cropping of long lines. Defaults to False. 
213          indent_guides (bool, optional): Enable indentation guides. Defaults to False. 
214          max_length (int, optional): Maximum length of containers before abbreviating, or None for no abbreviation. 
215              Defaults to None. 
216          max_string (int, optional): Maximum length of string before truncating, or None to disable. Defaults to None. 
217          expand_all (bool, optional): Expand all containers. Defaults to False. 
218          max_frames (int): Maximum number of frames to show in a traceback, 0 for no maximum. Defaults to 100. 
219      """
220      from pip._vendor.rich import get_console
221  
222      console = console or get_console()
223      assert console is not None
224  
225      def display_hook(value: Any) -> None:
226          """Replacement sys.displayhook which prettifies objects with Rich."""
227          if value is not None:
228              assert console is not None
229              builtins._ = None  # type: ignore[attr-defined]
230              console.print(
231                  value
232                  if _safe_isinstance(value, RichRenderable)
233                  else Pretty(
234                      value,
235                      overflow=overflow,
236                      indent_guides=indent_guides,
237                      max_length=max_length,
238                      max_string=max_string,
239                      expand_all=expand_all,
240                  ),
241                  crop=crop,
242              )
243              builtins._ = value  # type: ignore[attr-defined]
244  
245      try:  # pragma: no cover
246          ip = get_ipython()  # type: ignore[name-defined]
247          from IPython.core.formatters import BaseFormatter
248  
249          class RichFormatter(BaseFormatter):  # type: ignore[misc]
250              pprint: bool = True
251  
252              def __call__(self, value: Any) -> Any:
253                  if self.pprint:
254                      return _ipy_display_hook(
255                          value,
256                          console=get_console(),
257                          overflow=overflow,
258                          indent_guides=indent_guides,
259                          max_length=max_length,
260                          max_string=max_string,
261                          expand_all=expand_all,
262                      )
263                  else:
264                      return repr(value)
265  
266          # replace plain text formatter with rich formatter
267          rich_formatter = RichFormatter()
268          ip.display_formatter.formatters["text/plain"] = rich_formatter
269      except Exception:
270          sys.displayhook = display_hook

273  class Pretty(JupyterMixin):
274      """A rich renderable that pretty prints an object. 
275   
276      Args: 
277          _object (Any): An object to pretty print. 
278          highlighter (HighlighterType, optional): Highlighter object to apply to result, or None for ReprHighlighter. Defaults to None. 
279          indent_size (int, optional): Number of spaces in indent. Defaults to 4. 
280          justify (JustifyMethod, optional): Justify method, or None for default. Defaults to None. 
281          overflow (OverflowMethod, optional): Overflow method, or None for default. Defaults to None. 
282          no_wrap (Optional[bool], optional): Disable word wrapping. Defaults to False. 
283          indent_guides (bool, optional): Enable indentation guides. Defaults to False. 
284          max_length (int, optional): Maximum length of containers before abbreviating, or None for no abbreviation. 
285              Defaults to None. 
286          max_string (int, optional): Maximum length of string before truncating, or None to disable. Defaults to None. 
287          max_depth (int, optional): Maximum depth of nested data structures, or None for no maximum. Defaults to None. 
288          expand_all (bool, optional): Expand all containers. Defaults to False. 
289          margin (int, optional): Subtrace a margin from width to force containers to expand earlier. Defaults to 0. 
290          insert_line (bool, optional): Insert a new line if the output has multiple new lines. Defaults to False. 
291      """
292  
293      def __init__(
294          self,
295          _object: Any,
296          highlighter: Optional["HighlighterType"] = None,
297          *,
298          indent_size: int = 4,
299          justify: Optional["JustifyMethod"] = None,
300          overflow: Optional["OverflowMethod"] = None,
301          no_wrap: Optional[bool] = False,
302          indent_guides: bool = False,
303          max_length: Optional[int] = None,
304          max_string: Optional[int] = None,
305          max_depth: Optional[int] = None,
306          expand_all: bool = False,
307          margin: int = 0,
308          insert_line: bool = False,
309      ) -> None:
310          self._object = _object
311          self.highlighter = highlighter or ReprHighlighter()
312          self.indent_size = indent_size
313          self.justify: Optional["JustifyMethod"] = justify
314          self.overflow: Optional["OverflowMethod"] = overflow
315          self.no_wrap = no_wrap
316          self.indent_guides = indent_guides
317          self.max_length = max_length
318          self.max_string = max_string
319          self.max_depth = max_depth
320          self.expand_all = expand_all
321          self.margin = margin
322          self.insert_line = insert_line
323  
324      def __rich_console__(
325          self, console: "Console", options: "ConsoleOptions"
326      ) -> "RenderResult":
327          pretty_str = pretty_repr(
328              self._object,
329              max_width=options.max_width - self.margin,
330              indent_size=self.indent_size,
331              max_length=self.max_length,
332              max_string=self.max_string,
333              max_depth=self.max_depth,
334              expand_all=self.expand_all,
335          )
336          pretty_text = Text(
337              pretty_str,
338              justify=self.justify or options.justify,
339              overflow=self.overflow or options.overflow,
340              no_wrap=pick_bool(self.no_wrap, options.no_wrap),
341              style="pretty",
342          )
343          pretty_text = (
344              self.highlighter(pretty_text)
345              if pretty_text
346              else Text(
347                  f"{type(self._object)}.__repr__ returned empty string",
348                  style="dim italic",
349              )
350          )
351          if self.indent_guides and not options.ascii_only:
352              pretty_text = pretty_text.with_indent_guides(
353                  self.indent_size, style="repr.indent"
354              )
355          if self.insert_line and "\n" in pretty_text:
356              yield ""
357          yield pretty_text
358  
359      def __rich_measure__(
360          self, console: "Console", options: "ConsoleOptions"
361      ) -> "Measurement":
362          pretty_str = pretty_repr(
363              self._object,
364              max_width=options.max_width,
365              indent_size=self.indent_size,
366              max_length=self.max_length,
367              max_string=self.max_string,
368              expand_all=self.expand_all,
369          )
370          text_width = (
371              max(cell_len(line) for line in pretty_str.splitlines()) if pretty_str else 0
372          )
373          return Measurement(text_width, text_width)

417  @dataclass
418  class Node:
419      """A node in a repr tree. May be atomic or a container."""
420  
421      key_repr: str = ""
422      value_repr: str = ""
423      open_brace: str = ""
424      close_brace: str = ""
425      empty: str = ""
426      last: bool = False
427      is_tuple: bool = False
428      is_namedtuple: bool = False
429      children: Optional[List["Node"]] = None
430      key_separator = ": "
431      separator: str = ", "
432  
433      def iter_tokens(self) -> Iterable[str]:
434          """Generate tokens for this node."""
435          if self.key_repr:
436              yield self.key_repr
437              yield self.key_separator
438          if self.value_repr:
439              yield self.value_repr
440          elif self.children is not None:
441              if self.children:
442                  yield self.open_brace
443                  if self.is_tuple and not self.is_namedtuple and len(self.children) == 1:
444                      yield from self.children[0].iter_tokens()
445                      yield ","
446                  else:
447                      for child in self.children:
448                          yield from child.iter_tokens()
449                          if not child.last:
450                              yield self.separator
451                  yield self.close_brace
452              else:
453                  yield self.empty
454  
455      def check_length(self, start_length: int, max_length: int) -> bool:
456          """Check the length fits within a limit. 
457   
458          Args: 
459              start_length (int): Starting length of the line (indent, prefix, suffix). 
460              max_length (int): Maximum length. 
461   
462          Returns: 
463              bool: True if the node can be rendered within max length, otherwise False. 
464          """
465          total_length = start_length
466          for token in self.iter_tokens():
467              total_length += cell_len(token)
468              if total_length > max_length:
469                  return False
470          return True
471  
472      def __str__(self) -> str:
473          repr_text = "".join(self.iter_tokens())
474          return repr_text
475  
476      def render(
477          self, max_width: int = 80, indent_size: int = 4, expand_all: bool = False
478      ) -> str:
479          """Render the node to a pretty repr. 
480   
481          Args: 
482              max_width (int, optional): Maximum width of the repr. Defaults to 80. 
483              indent_size (int, optional): Size of indents. Defaults to 4. 
484              expand_all (bool, optional): Expand all levels. Defaults to False. 
485   
486          Returns: 
487              str: A repr string of the original object. 
488          """
489          lines = [_Line(node=self, is_root=True)]
490          line_no = 0
491          while line_no < len(lines):
492              line = lines[line_no]
493              if line.expandable and not line.expanded:
494                  if expand_all or not line.check_length(max_width):
495                      lines[line_no : line_no + 1] = line.expand(indent_size)
496              line_no += 1
497  
498          repr_str = "\n".join(str(line) for line in lines)
499          return repr_str

502  @dataclass
503  class _Line:
504      """A line in repr output."""
505  
506      parent: Optional["_Line"] = None
507      is_root: bool = False
508      node: Optional[Node] = None
509      text: str = ""
510      suffix: str = ""
511      whitespace: str = ""
512      expanded: bool = False
513      last: bool = False
514  
515      @property
516      def expandable(self) -> bool:
517          """Check if the line may be expanded."""
518          return bool(self.node is not None and self.node.children)
519  
520      def check_length(self, max_length: int) -> bool:
521          """Check this line fits within a given number of cells."""
522          start_length = (
523              len(self.whitespace) + cell_len(self.text) + cell_len(self.suffix)
524          )
525          assert self.node is not None
526          return self.node.check_length(start_length, max_length)
527  
528      def expand(self, indent_size: int) -> Iterable["_Line"]:
529          """Expand this line by adding children on their own line."""
530          node = self.node
531          assert node is not None
532          whitespace = self.whitespace
533          assert node.children
534          if node.key_repr:
535              new_line = yield _Line(
536                  text=f"{node.key_repr}{node.key_separator}{node.open_brace}",
537                  whitespace=whitespace,
538              )
539          else:
540              new_line = yield _Line(text=node.open_brace, whitespace=whitespace)
541          child_whitespace = self.whitespace + " " * indent_size
542          tuple_of_one = node.is_tuple and len(node.children) == 1
543          for last, child in loop_last(node.children):
544              separator = "," if tuple_of_one else node.separator
545              line = _Line(
546                  parent=new_line,
547                  node=child,
548                  whitespace=child_whitespace,
549                  suffix=separator,
550                  last=last and not tuple_of_one,
551              )
552              yield line
553  
554          yield _Line(
555              text=node.close_brace,
556              whitespace=whitespace,
557              suffix=self.suffix,
558              last=self.last,
559          )
560  
561      def __str__(self) -> str:
562          if self.last:
563              return f"{self.whitespace}{self.text}{self.node or ''}"
564          else:
565              return (
566                  f"{self.whitespace}{self.text}{self.node or ''}{self.suffix.rstrip()}"
567              )

570  def _is_namedtuple(obj: Any) -> bool:
571      """Checks if an object is most likely a namedtuple. It is possible 
572      to craft an object that passes this check and isn't a namedtuple, but 
573      there is only a minuscule chance of this happening unintentionally. 
574   
575      Args: 
576          obj (Any): The object to test 
577   
578      Returns: 
579          bool: True if the object is a namedtuple. False otherwise. 
580      """
581      try:
582          fields = getattr(obj, "_fields", None)
583      except Exception:
584          # Being very defensive - if we cannot get the attr then its not a namedtuple
585          return False
586      return isinstance(obj, tuple) and isinstance(fields, tuple)

589  def traverse(
590      _object: Any,
591      max_length: Optional[int] = None,
592      max_string: Optional[int] = None,
593      max_depth: Optional[int] = None,
594  ) -> Node:
595      """Traverse object and generate a tree. 
596   
597      Args: 
598          _object (Any): Object to be traversed. 
599          max_length (int, optional): Maximum length of containers before abbreviating, or None for no abbreviation. 
600              Defaults to None. 
601          max_string (int, optional): Maximum length of string before truncating, or None to disable truncating. 
602              Defaults to None. 
603          max_depth (int, optional): Maximum depth of data structures, or None for no maximum. 
604              Defaults to None. 
605   
606      Returns: 
607          Node: The root of a tree structure which can be used to render a pretty repr. 
608      """
609  
610      def to_repr(obj: Any) -> str:
611          """Get repr string for an object, but catch errors."""
612          if (
613              max_string is not None
614              and _safe_isinstance(obj, (bytes, str))
615              and len(obj) > max_string
616          ):
617              truncated = len(obj) - max_string
618              obj_repr = f"{obj[:max_string]!r}+{truncated}"
619          else:
620              try:
621                  obj_repr = repr(obj)
622              except Exception as error:
623                  obj_repr = f"<repr-error {str(error)!r}>"
624          return obj_repr
625  
626      visited_ids: Set[int] = set()
627      push_visited = visited_ids.add
628      pop_visited = visited_ids.remove
629  
630      def _traverse(obj: Any, root: bool = False, depth: int = 0) -> Node:
631          """Walk the object depth first."""
632  
633          obj_type = type(obj)
634          py_version = (sys.version_info.major, sys.version_info.minor)
635          children: List[Node]
636          reached_max_depth = max_depth is not None and depth >= max_depth
637  
638          def iter_rich_args(rich_args: Any) -> Iterable[Union[Any, Tuple[str, Any]]]:
639              for arg in rich_args:
640                  if _safe_isinstance(arg, tuple):
641                      if len(arg) == 3:
642                          key, child, default = arg
643                          if default == child:
644                              continue
645                          yield key, child
646                      elif len(arg) == 2:
647                          key, child = arg
648                          yield key, child
649                      elif len(arg) == 1:
650                          yield arg[0]
651                  else:
652                      yield arg
653  
654          try:
655              fake_attributes = hasattr(
656                  obj, "awehoi234_wdfjwljet234_234wdfoijsdfmmnxpi492"
657              )
658          except Exception:
659              fake_attributes = False
660  
661          rich_repr_result: Optional[RichReprResult] = None
662          if not fake_attributes:
663              try:
664                  if hasattr(obj, "__rich_repr__") and not isclass(obj):
665                      rich_repr_result = obj.__rich_repr__()
666              except Exception:
667                  pass
668  
669          if rich_repr_result is not None:
670              angular = getattr(obj.__rich_repr__, "angular", False)
671              args = list(iter_rich_args(rich_repr_result))
672              class_name = obj.__class__.__name__
673  
674              if args:
675                  children = []
676                  append = children.append
677  
678                  if reached_max_depth:
679                      node = Node(value_repr=f"...")
680                  else:
681                      if angular:
682                          node = Node(
683                              open_brace=f"<{class_name} ",
684                              close_brace=">",
685                              children=children,
686                              last=root,
687                              separator=" ",
688                          )
689                      else:
690                          node = Node(
691                              open_brace=f"{class_name}(",
692                              close_brace=")",
693                              children=children,
694                              last=root,
695                          )
696                      for last, arg in loop_last(args):
697                          if _safe_isinstance(arg, tuple):
698                              key, child = arg
699                              child_node = _traverse(child, depth=depth + 1)
700                              child_node.last = last
701                              child_node.key_repr = key
702                              child_node.key_separator = "="
703                              append(child_node)
704                          else:
705                              child_node = _traverse(arg, depth=depth + 1)
706                              child_node.last = last
707                              append(child_node)
708              else:
709                  node = Node(
710                      value_repr=f"<{class_name}>" if angular else f"{class_name}()",
711                      children=[],
712                      last=root,
713                  )
714          elif _is_attr_object(obj) and not fake_attributes:
715              children = []
716              append = children.append
717  
718              attr_fields = _get_attr_fields(obj)
719              if attr_fields:
720                  if reached_max_depth:
721                      node = Node(value_repr=f"...")
722                  else:
723                      node = Node(
724                          open_brace=f"{obj.__class__.__name__}(",
725                          close_brace=")",
726                          children=children,
727                          last=root,
728                      )
729  
730                      def iter_attrs() -> Iterable[
731                          Tuple[str, Any, Optional[Callable[[Any], str]]]
732                      ]:
733                          """Iterate over attr fields and values."""
734                          for attr in attr_fields:
735                              if attr.repr:
736                                  try:
737                                      value = getattr(obj, attr.name)
738                                  except Exception as error:
739                                      # Can happen, albeit rarely
740                                      yield (attr.name, error, None)
741                                  else:
742                                      yield (
743                                          attr.name,
744                                          value,
745                                          attr.repr if callable(attr.repr) else None,
746                                      )
747  
748                      for last, (name, value, repr_callable) in loop_last(iter_attrs()):
749                          if repr_callable:
750                              child_node = Node(value_repr=str(repr_callable(value)))
751                          else:
752                              child_node = _traverse(value, depth=depth + 1)
753                          child_node.last = last
754                          child_node.key_repr = name
755                          child_node.key_separator = "="
756                          append(child_node)
757              else:
758                  node = Node(
759                      value_repr=f"{obj.__class__.__name__}()", children=[], last=root
760                  )
761  
762          elif (
763              is_dataclass(obj)
764              and not _safe_isinstance(obj, type)
765              and not fake_attributes
766              and (_is_dataclass_repr(obj) or py_version == (3, 6))
767          ):
768              obj_id = id(obj)
769              if obj_id in visited_ids:
770                  # Recursion detected
771                  return Node(value_repr="...")
772              push_visited(obj_id)
773  
774              children = []
775              append = children.append
776              if reached_max_depth:
777                  node = Node(value_repr=f"...")
778              else:
779                  node = Node(
780                      open_brace=f"{obj.__class__.__name__}(",
781                      close_brace=")",
782                      children=children,
783                      last=root,
784                  )
785  
786                  for last, field in loop_last(
787                      field for field in fields(obj) if field.repr
788                  ):
789                      child_node = _traverse(getattr(obj, field.name), depth=depth + 1)
790                      child_node.key_repr = field.name
791                      child_node.last = last
792                      child_node.key_separator = "="
793                      append(child_node)
794  
795                  pop_visited(obj_id)
796          elif _is_namedtuple(obj) and _has_default_namedtuple_repr(obj):
797              if reached_max_depth:
798                  node = Node(value_repr="...")
799              else:
800                  children = []
801                  class_name = obj.__class__.__name__
802                  node = Node(
803                      open_brace=f"{class_name}(",
804                      close_brace=")",
805                      children=children,
806                      empty=f"{class_name}()",
807                  )
808                  append = children.append
809                  for last, (key, value) in loop_last(obj._asdict().items()):
810                      child_node = _traverse(value, depth=depth + 1)
811                      child_node.key_repr = key
812                      child_node.last = last
813                      child_node.key_separator = "="
814                      append(child_node)
815          elif _safe_isinstance(obj, _CONTAINERS):
816              for container_type in _CONTAINERS:
817                  if _safe_isinstance(obj, container_type):
818                      obj_type = container_type
819                      break
820  
821              obj_id = id(obj)
822              if obj_id in visited_ids:
823                  # Recursion detected
824                  return Node(value_repr="...")
825              push_visited(obj_id)
826  
827              open_brace, close_brace, empty = _BRACES[obj_type](obj)
828  
829              if reached_max_depth:
830                  node = Node(value_repr=f"...", last=root)
831              elif obj_type.__repr__ != type(obj).__repr__:
832                  node = Node(value_repr=to_repr(obj), last=root)
833              elif obj:
834                  children = []
835                  node = Node(
836                      open_brace=open_brace,
837                      close_brace=close_brace,
838                      children=children,
839                      last=root,
840                  )
841                  append = children.append
842                  num_items = len(obj)
843                  last_item_index = num_items - 1
844  
845                  if _safe_isinstance(obj, _MAPPING_CONTAINERS):
846                      iter_items = iter(obj.items())
847                      if max_length is not None:
848                          iter_items = islice(iter_items, max_length)
849                      for index, (key, child) in enumerate(iter_items):
850                          child_node = _traverse(child, depth=depth + 1)
851                          child_node.key_repr = to_repr(key)
852                          child_node.last = index == last_item_index
853                          append(child_node)
854                  else:
855                      iter_values = iter(obj)
856                      if max_length is not None:
857                          iter_values = islice(iter_values, max_length)
858                      for index, child in enumerate(iter_values):
859                          child_node = _traverse(child, depth=depth + 1)
860                          child_node.last = index == last_item_index
861                          append(child_node)
862                  if max_length is not None and num_items > max_length:
863                      append(Node(value_repr=f"... +{num_items - max_length}", last=True))
864              else:
865                  node = Node(empty=empty, children=[], last=root)
866  
867              pop_visited(obj_id)
868          else:
869              node = Node(value_repr=to_repr(obj), last=root)
870          node.is_tuple = _safe_isinstance(obj, tuple)
871          node.is_namedtuple = _is_namedtuple(obj)
872          return node
873  
874      node = _traverse(_object, root=True)
875      return node

878  def pretty_repr(
879      _object: Any,
880      *,
881      max_width: int = 80,
882      indent_size: int = 4,
883      max_length: Optional[int] = None,
884      max_string: Optional[int] = None,
885      max_depth: Optional[int] = None,
886      expand_all: bool = False,
887  ) -> str:
888      """Prettify repr string by expanding on to new lines to fit within a given width. 
889   
890      Args: 
891          _object (Any): Object to repr. 
892          max_width (int, optional): Desired maximum width of repr string. Defaults to 80. 
893          indent_size (int, optional): Number of spaces to indent. Defaults to 4. 
894          max_length (int, optional): Maximum length of containers before abbreviating, or None for no abbreviation. 
895              Defaults to None. 
896          max_string (int, optional): Maximum length of string before truncating, or None to disable truncating. 
897              Defaults to None. 
898          max_depth (int, optional): Maximum depth of nested data structure, or None for no depth. 
899              Defaults to None. 
900          expand_all (bool, optional): Expand all containers regardless of available width. Defaults to False. 
901   
902      Returns: 
903          str: A possibly multi-line representation of the object. 
904      """
905  
906      if _safe_isinstance(_object, Node):
907          node = _object
908      else:
909          node = traverse(
910              _object, max_length=max_length, max_string=max_string, max_depth=max_depth
911          )
912      repr_str: str = node.render(
913          max_width=max_width, indent_size=indent_size, expand_all=expand_all
914      )
915      return repr_str
