18   import functools
19   import sys
20   import typing
21   from asyncio import sleep
22   
23   from pip._vendor.tenacity import AttemptManager
24   from pip._vendor.tenacity import BaseRetrying
25   from pip._vendor.tenacity import DoAttempt
26   from pip._vendor.tenacity import DoSleep
27   from pip._vendor.tenacity import RetryCallState
28   
29   WrappedFn = typing.TypeVar("WrappedFn", bound=typing.Callable)
30   _RetValT = typing.TypeVar("_RetValT")
31   
32   
33   class AsyncRetrying(BaseRetrying):
34       def __init__(self, sleep: typing.Callable[[float], typing.Awaitable] = sleep, **kwargs: typing.Any) -> None:
35           super().__init__(**kwargs)
36           self.sleep = sleep
37   
38       async def __call__(  # type: ignore  # Change signature from supertype
39           self,
40           fn: typing.Callable[..., typing.Awaitable[_RetValT]],
41           *args: typing.Any,
42           **kwargs: typing.Any,
43       ) -> _RetValT:
44           self.begin()
45   
46           retry_state = RetryCallState(retry_object=self, fn=fn, args=args, kwargs=kwargs)
47           while True:
48               do = self.iter(retry_state=retry_state)
49               if isinstance(do, DoAttempt):
50                   try:
51                       result = await fn(*args, **kwargs)
52                   except BaseException:  # noqa: B902
53                       retry_state.set_exception(sys.exc_info())
54                   else:
55                       retry_state.set_result(result)
56               elif isinstance(do, DoSleep):
57                   retry_state.prepare_for_next_attempt()
58                   await self.sleep(do)
59               else:
60                   return do
61   
62       def __aiter__(self) -> "AsyncRetrying":
63           self.begin()
64           self._retry_state = RetryCallState(self, fn=None, args=(), kwargs={})
65           return self
66   
67       async def __anext__(self) -> typing.Union[AttemptManager, typing.Any]:
68           while True:
69               do = self.iter(retry_state=self._retry_state)
70               if do is None:
71                   raise StopAsyncIteration
72               elif isinstance(do, DoAttempt):
73                   return AttemptManager(retry_state=self._retry_state)
74               elif isinstance(do, DoSleep):
75                   self._retry_state.prepare_for_next_attempt()
76                   await self.sleep(do)
77               else:
78                   return do
79   
80       def wraps(self, fn: WrappedFn) -> WrappedFn:
81           fn = super().wraps(fn)
82           # Ensure wrapper is recognized as a coroutine function.
83   
84           @functools.wraps(fn)
85           async def async_wrapped(*args: typing.Any, **kwargs: typing.Any) -> typing.Any:
86               return await fn(*args, **kwargs)
87   
88           # Preserve attributes
89           async_wrapped.retry = fn.retry
90           async_wrapped.retry_with = fn.retry_with
91   
92           return async_wrapped
