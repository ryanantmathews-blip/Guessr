16   # Non-greedy matching of an optional build number may be too clever (more
17   # invalid wheel filenames will match). Separate regex for .dist-info?
18   WHEEL_INFO_RE = re.compile(
19       r"""^(?P<namever>(?P<name>[^\s-]+?)-(?P<ver>[^\s-]+?))(-(?P<build>\d[^\s-]*))? 
20        -(?P<pyver>[^\s-]+?)-(?P<abi>[^\s-]+?)-(?P<plat>\S+)\.whl$""",
21       re.VERBOSE,
22   )
23   MINIMUM_TIMESTAMP = 315532800  # 1980-01-01 00:00:00 UTC

26   def get_zipinfo_datetime(timestamp=None):
27       # Some applications need reproducible .whl files, but they can't do this without
28       # forcing the timestamp of the individual ZipInfo objects. See issue #143.
29       timestamp = int(os.environ.get("SOURCE_DATE_EPOCH", timestamp or time.time()))
30       timestamp = max(timestamp, MINIMUM_TIMESTAMP)
31       return time.gmtime(timestamp)[0:6]

34   class WheelFile(ZipFile):
35       """A ZipFile derivative class that also reads SHA-256 hashes from 
36       .dist-info/RECORD and checks any read files against those. 
37       """
38   
39       _default_algorithm = hashlib.sha256
40   
41       def __init__(self, file, mode="r", compression=ZIP_DEFLATED):
42           basename = os.path.basename(file)
43           self.parsed_filename = WHEEL_INFO_RE.match(basename)
44           if not basename.endswith(".whl") or self.parsed_filename is None:
45               raise WheelError(f"Bad wheel filename {basename!r}")
46   
47           ZipFile.__init__(self, file, mode, compression=compression, allowZip64=True)
48   
49           self.dist_info_path = "{}.dist-info".format(
50               self.parsed_filename.group("namever")
51           )
52           self.record_path = self.dist_info_path + "/RECORD"
53           self._file_hashes = OrderedDict()
54           self._file_sizes = {}
55           if mode == "r":
56               # Ignore RECORD and any embedded wheel signatures
57               self._file_hashes[self.record_path] = None, None
58               self._file_hashes[self.record_path + ".jws"] = None, None
59               self._file_hashes[self.record_path + ".p7s"] = None, None
60   
61               # Fill in the expected hashes by reading them from RECORD
62               try:
63                   record = self.open(self.record_path)
64               except KeyError:
65                   raise WheelError(f"Missing {self.record_path} file")
66   
67               with record:
68                   for line in csv.reader(
69                       TextIOWrapper(record, newline="", encoding="utf-8")
70                   ):
71                       path, hash_sum, size = line
72                       if not hash_sum:
73                           continue
74   
75                       algorithm, hash_sum = hash_sum.split("=")
76                       try:
77                           hashlib.new(algorithm)
78                       except ValueError:
79                           raise WheelError(f"Unsupported hash algorithm: {algorithm}")
80   
81                       if algorithm.lower() in {"md5", "sha1"}:
82                           raise WheelError(
83                               "Weak hash algorithm ({}) is not permitted by PEP "
84                               "427".format(algorithm)
85                           )
86   
87                       self._file_hashes[path] = (
88                           algorithm,
89                           urlsafe_b64decode(hash_sum.encode("ascii")),
90                       )
91   
92       def open(self, name_or_info, mode="r", pwd=None):
93           def _update_crc(newdata):
94               eof = ef._eof
95               update_crc_orig(newdata)
96               running_hash.update(newdata)
97               if eof and running_hash.digest() != expected_hash:
98                   raise WheelError(f"Hash mismatch for file '{ef_name}'")
99   
100          ef_name = (
101              name_or_info.filename if isinstance(name_or_info, ZipInfo) else name_or_info
102          )
103          if (
104              mode == "r"
105              and not ef_name.endswith("/")
106              and ef_name not in self._file_hashes
107          ):
108              raise WheelError(f"No hash found for file '{ef_name}'")
109  
110          ef = ZipFile.open(self, name_or_info, mode, pwd)
111          if mode == "r" and not ef_name.endswith("/"):
112              algorithm, expected_hash = self._file_hashes[ef_name]
113              if expected_hash is not None:
114                  # Monkey patch the _update_crc method to also check for the hash from
115                  # RECORD
116                  running_hash = hashlib.new(algorithm)
117                  update_crc_orig, ef._update_crc = ef._update_crc, _update_crc
118  
119          return ef

121      def write_files(self, base_dir):
122          log.info(f"creating '{self.filename}' and adding '{base_dir}' to it")
123          deferred = []
124          for root, dirnames, filenames in os.walk(base_dir):
125              # Sort the directory names so that `os.walk` will walk them in a
126              # defined order on the next iteration.
127              dirnames.sort()
128              for name in sorted(filenames):
129                  path = os.path.normpath(os.path.join(root, name))
130                  if os.path.isfile(path):
131                      arcname = os.path.relpath(path, base_dir).replace(os.path.sep, "/")
132                      if arcname == self.record_path:
133                          pass
134                      elif root.endswith(".dist-info"):
135                          deferred.append((path, arcname))
136                      else:
137                          self.write(path, arcname)
138  
139          deferred.sort()
140          for path, arcname in deferred:
141              self.write(path, arcname)

143      def write(self, filename, arcname=None, compress_type=None):
144          with open(filename, "rb") as f:
145              st = os.fstat(f.fileno())
146              data = f.read()
147  
148          zinfo = ZipInfo(
149              arcname or filename, date_time=get_zipinfo_datetime(st.st_mtime)
150          )
151          zinfo.external_attr = (stat.S_IMODE(st.st_mode) | stat.S_IFMT(st.st_mode)) << 16
152          zinfo.compress_type = compress_type or self.compression
153          self.writestr(zinfo, data, compress_type)
154  
155      def writestr(self, zinfo_or_arcname, data, compress_type=None):
156          if isinstance(data, str):
157              data = data.encode("utf-8")
158  
159          ZipFile.writestr(self, zinfo_or_arcname, data, compress_type)
160          fname = (
161              zinfo_or_arcname.filename
162              if isinstance(zinfo_or_arcname, ZipInfo)
163              else zinfo_or_arcname
164          )
165          log.info(f"adding '{fname}'")
166          if fname != self.record_path:
167              hash_ = self._default_algorithm(data)
168              self._file_hashes[fname] = (
169                  hash_.name,
170                  urlsafe_b64encode(hash_.digest()).decode("ascii"),
171              )
172              self._file_sizes[fname] = len(data)
173  
174      def close(self):
175          # Write RECORD
176          if self.fp is not None and self.mode == "w" and self._file_hashes:
177              data = StringIO()
178              writer = csv.writer(data, delimiter=",", quotechar='"', lineterminator="\n")
179              writer.writerows(
180                  (
181                      (fname, algorithm + "=" + hash_, self._file_sizes[fname])
182                      for fname, (algorithm, hash_) in self._file_hashes.items()
183                  )
184              )
185              writer.writerow((format(self.record_path), "", ""))
186              zinfo = ZipInfo(self.record_path, date_time=get_zipinfo_datetime())
187              zinfo.compress_type = self.compression
188              zinfo.external_attr = 0o664 << 16
189              self.writestr(zinfo, data.getvalue())
190  
191          ZipFile.close(self)
