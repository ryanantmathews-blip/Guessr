18   def _load_pyproject(source_dir):
19       with open(
20               os.path.join(source_dir, 'pyproject.toml'),
21               'rb',
22               ) as f:
23           pyproject_data = tomllib.load(f)
24       buildsys = pyproject_data['build-system']
25       return (
26           buildsys['requires'],
27           buildsys['build-backend'],
28           buildsys.get('backend-path'),
29       )

58       path = None
59   
60       def __init__(self, cleanup=True):
61           self._cleanup = cleanup
62   
63       def __enter__(self):
64           self.path = mkdtemp(prefix='pep517-build-env-')
65           log.info('Temporary build environment: %s', self.path)
66   
67           self.save_path = os.environ.get('PATH', None)
68           self.save_pythonpath = os.environ.get('PYTHONPATH', None)
69   
70           install_scheme = 'nt' if (os.name == 'nt') else 'posix_prefix'
71           install_dirs = get_paths(install_scheme, vars={
72               'base': self.path,
73               'platbase': self.path,
74           })
75   
76           scripts = install_dirs['scripts']
77           if self.save_path:
78               os.environ['PATH'] = scripts + os.pathsep + self.save_path
79           else:
80               os.environ['PATH'] = scripts + os.pathsep + os.defpath
81   
82           if install_dirs['purelib'] == install_dirs['platlib']:
83               lib_dirs = install_dirs['purelib']
84           else:
85               lib_dirs = install_dirs['purelib'] + os.pathsep + \
86                   install_dirs['platlib']
87           if self.save_pythonpath:
88               os.environ['PYTHONPATH'] = lib_dirs + os.pathsep + \
89                   self.save_pythonpath
90           else:
91               os.environ['PYTHONPATH'] = lib_dirs
92   
93           return self
94   
95       def pip_install(self, reqs):
96           """Install dependencies into this env by calling pip in a subprocess"""
97           if not reqs:
98               return
99           log.info('Calling pip to install %s', reqs)
100          cmd = [
101              sys.executable, '-m', 'pip', 'install', '--ignore-installed',
102              '--prefix', self.path] + list(reqs)
103          check_call(
104              cmd,
105              stdout=LoggerWrapper(log, logging.INFO),
106              stderr=LoggerWrapper(log, logging.ERROR),
107          )
108  
109      def __exit__(self, exc_type, exc_val, exc_tb):
110          needs_cleanup = (
111              self._cleanup and
112              self.path is not None and
113              os.path.isdir(self.path)
114          )
115          if needs_cleanup:
116              shutil.rmtree(self.path)
117  
118          if self.save_path is None:
119              os.environ.pop('PATH', None)
120          else:
121              os.environ['PATH'] = self.save_path
122  
123          if self.save_pythonpath is None:
124              os.environ.pop('PYTHONPATH', None)
125          else:
126              os.environ['PYTHONPATH'] = self.save_pythonpath

129  def build_wheel(source_dir, wheel_dir, config_settings=None):
130      """Build a wheel from a source directory using PEP 517 hooks. 
131   
132      :param str source_dir: Source directory containing pyproject.toml 
133      :param str wheel_dir: Target directory to create wheel in 
134      :param dict config_settings: Options to pass to build backend 
135   
136      This is a blocking function which will run pip in a subprocess to install 
137      build requirements. 
138      """
139      if config_settings is None:
140          config_settings = {}
141      requires, backend, backend_path = _load_pyproject(source_dir)
142      hooks = Pep517HookCaller(source_dir, backend, backend_path)
143  
144      with BuildEnvironment() as env:
145          env.pip_install(requires)
146          reqs = hooks.get_requires_for_build_wheel(config_settings)
147          env.pip_install(reqs)
148          return hooks.build_wheel(wheel_dir, config_settings)
