319      def __rich_measure__(
320          self, console: "Console", options: "ConsoleOptions"
321      ) -> Measurement:
322          max_width = options.max_width
323          if self.width is not None:
324              max_width = self.width
325          if max_width < 0:
326              return Measurement(0, 0)
327  
328          extra_width = self._extra_width
329          max_width = sum(
330              self._calculate_column_widths(
331                  console, options.update_width(max_width - extra_width)
332              )
333          )
334          _measure_column = self._measure_column
335  
336          measurements = [
337              _measure_column(console, options.update_width(max_width), column)
338              for column in self.columns
339          ]
340          minimum_width = (
341              sum(measurement.minimum for measurement in measurements) + extra_width
342          )
343          maximum_width = (
344              sum(measurement.maximum for measurement in measurements) + extra_width
345              if (self.width is None)
346              else self.width
347          )
348          measurement = Measurement(minimum_width, maximum_width)
349          measurement = measurement.clamp(self.min_width)
350          return measurement

418      def add_row(
419          self,
420          *renderables: Optional["RenderableType"],
421          style: Optional[StyleType] = None,
422          end_section: bool = False,
423      ) -> None:
424          """Add a row of renderables. 
425   
426          Args: 
427              *renderables (None or renderable): Each cell in a row must be a renderable object (including str), 
428                  or ``None`` for a blank cell. 
429              style (StyleType, optional): An optional style to apply to the entire row. Defaults to None. 
430              end_section (bool, optional): End a section and draw a line. Defaults to False. 
431   
432          Raises: 
433              errors.NotRenderableError: If you add something that can't be rendered. 
434          """
435  
436          def add_cell(column: Column, renderable: "RenderableType") -> None:
437              column._cells.append(renderable)
438  
439          cell_renderables: List[Optional["RenderableType"]] = list(renderables)
440  
441          columns = self.columns
442          if len(cell_renderables) < len(columns):
443              cell_renderables = [
444                  *cell_renderables,
445                  *[None] * (len(columns) - len(cell_renderables)),
446              ]
447          for index, renderable in enumerate(cell_renderables):
448              if index == len(columns):
449                  column = Column(_index=index)
450                  for _ in self.rows:
451                      add_cell(column, Text(""))
452                  self.columns.append(column)
453              else:
454                  column = columns[index]
455              if renderable is None:
456                  add_cell(column, "")
457              elif is_renderable(renderable):
458                  add_cell(column, renderable)
459              else:
460                  raise errors.NotRenderableError(
461                      f"unable to render {type(renderable).__name__}; a string or other renderable object is required"
462                  )
463          self.rows.append(Row(style=style, end_section=end_section))

465      def __rich_console__(
466          self, console: "Console", options: "ConsoleOptions"
467      ) -> "RenderResult":
468  
469          if not self.columns:
470              yield Segment("\n")
471              return
472  
473          max_width = options.max_width
474          if self.width is not None:
475              max_width = self.width
476  
477          extra_width = self._extra_width
478          widths = self._calculate_column_widths(
479              console, options.update_width(max_width - extra_width)
480          )
481          table_width = sum(widths) + extra_width
482  
483          render_options = options.update(
484              width=table_width, highlight=self.highlight, height=None
485          )
486  
487          def render_annotation(
488              text: TextType, style: StyleType, justify: "JustifyMethod" = "center"
489          ) -> "RenderResult":
490              render_text = (
491                  console.render_str(text, style=style, highlight=False)
492                  if isinstance(text, str)
493                  else text
494              )
495              return console.render(
496                  render_text, options=render_options.update(justify=justify)
497              )
498  
499          if self.title:
500              yield from render_annotation(
501                  self.title,
502                  style=Style.pick_first(self.title_style, "table.title"),
503                  justify=self.title_justify,
504              )
505          yield from self._render(console, render_options, widths)
506          if self.caption:
507              yield from render_annotation(
508                  self.caption,
509                  style=Style.pick_first(self.caption_style, "table.caption"),
510                  justify=self.caption_justify,
511              )

513      def _calculate_column_widths(
514          self, console: "Console", options: "ConsoleOptions"
515      ) -> List[int]:
516          """Calculate the widths of each column, including padding, not including borders."""
517          max_width = options.max_width
518          columns = self.columns
519          width_ranges = [
520              self._measure_column(console, options, column) for column in columns
521          ]
522          widths = [_range.maximum or 1 for _range in width_ranges]
523          get_padding_width = self._get_padding_width
524          extra_width = self._extra_width
525          if self.expand:
526              ratios = [col.ratio or 0 for col in columns if col.flexible]
527              if any(ratios):
528                  fixed_widths = [
529                      0 if column.flexible else _range.maximum
530                      for _range, column in zip(width_ranges, columns)
531                  ]
532                  flex_minimum = [
533                      (column.width or 1) + get_padding_width(column._index)
534                      for column in columns
535                      if column.flexible
536                  ]
537                  flexible_width = max_width - sum(fixed_widths)
538                  flex_widths = ratio_distribute(flexible_width, ratios, flex_minimum)
539                  iter_flex_widths = iter(flex_widths)
540                  for index, column in enumerate(columns):
541                      if column.flexible:
542                          widths[index] = fixed_widths[index] + next(iter_flex_widths)
543          table_width = sum(widths)
544  
545          if table_width > max_width:
546              widths = self._collapse_widths(
547                  widths,
548                  [(column.width is None and not column.no_wrap) for column in columns],
549                  max_width,
550              )
551              table_width = sum(widths)
552              # last resort, reduce columns evenly
553              if table_width > max_width:
554                  excess_width = table_width - max_width
555                  widths = ratio_reduce(excess_width, [1] * len(widths), widths, widths)
556                  table_width = sum(widths)
557  
558              width_ranges = [
559                  self._measure_column(console, options.update_width(width), column)
560                  for width, column in zip(widths, columns)
561              ]
562              widths = [_range.maximum or 0 for _range in width_ranges]
563  
564          if (table_width < max_width and self.expand) or (
565              self.min_width is not None and table_width < (self.min_width - extra_width)
566          ):
567              _max_width = (
568                  max_width
569                  if self.min_width is None
570                  else min(self.min_width - extra_width, max_width)
571              )
572              pad_widths = ratio_distribute(_max_width - table_width, widths)
573              widths = [_width + pad for _width, pad in zip(widths, pad_widths)]
574  
575          return widths
576  
577      @classmethod
578      def _collapse_widths(
579          cls, widths: List[int], wrapable: List[bool], max_width: int
580      ) -> List[int]:
581          """Reduce widths so that the total is under max_width. 
582   
583          Args: 
584              widths (List[int]): List of widths. 
585              wrapable (List[bool]): List of booleans that indicate if a column may shrink. 
586              max_width (int): Maximum width to reduce to. 
587   
588          Returns: 
589              List[int]: A new list of widths. 
590          """
591          total_width = sum(widths)
592          excess_width = total_width - max_width
593          if any(wrapable):
594              while total_width and excess_width > 0:
595                  max_column = max(
596                      width for width, allow_wrap in zip(widths, wrapable) if allow_wrap
597                  )
598                  second_max_column = max(
599                      width if allow_wrap and width != max_column else 0
600                      for width, allow_wrap in zip(widths, wrapable)
601                  )
602                  column_difference = max_column - second_max_column
603                  ratios = [
604                      (1 if (width == max_column and allow_wrap) else 0)
605                      for width, allow_wrap in zip(widths, wrapable)
606                  ]
607                  if not any(ratios) or not column_difference:
608                      break
609                  max_reduce = [min(excess_width, column_difference)] * len(widths)
610                  widths = ratio_reduce(excess_width, ratios, max_reduce, widths)
611  
612                  total_width = sum(widths)
613                  excess_width = total_width - max_width
614          return widths
615  
616      def _get_cells(
617          self, console: "Console", column_index: int, column: Column
618      ) -> Iterable[_Cell]:
619          """Get all the cells with padding and optional header."""
620  
621          collapse_padding = self.collapse_padding
622          pad_edge = self.pad_edge
623          padding = self.padding
624          any_padding = any(padding)
625  
626          first_column = column_index == 0
627          last_column = column_index == len(self.columns) - 1
628  
629          _padding_cache: Dict[Tuple[bool, bool], Tuple[int, int, int, int]] = {}
630  
631          def get_padding(first_row: bool, last_row: bool) -> Tuple[int, int, int, int]:
632              cached = _padding_cache.get((first_row, last_row))
633              if cached:
634                  return cached
635              top, right, bottom, left = padding
636  
637              if collapse_padding:
638                  if not first_column:
639                      left = max(0, left - right)
640                  if not last_row:
641                      bottom = max(0, top - bottom)
642  
643              if not pad_edge:
644                  if first_column:
645                      left = 0
646                  if last_column:
647                      right = 0
648                  if first_row:
649                      top = 0
650                  if last_row:
651                      bottom = 0
652              _padding = (top, right, bottom, left)
653              _padding_cache[(first_row, last_row)] = _padding
654              return _padding
655  
656          raw_cells: List[Tuple[StyleType, "RenderableType"]] = []
657          _append = raw_cells.append
658          get_style = console.get_style
659          if self.show_header:
660              header_style = get_style(self.header_style or "") + get_style(
661                  column.header_style
662              )
663              _append((header_style, column.header))
664          cell_style = get_style(column.style or "")
665          for cell in column.cells:
666              _append((cell_style, cell))
667          if self.show_footer:
668              footer_style = get_style(self.footer_style or "") + get_style(
669                  column.footer_style
670              )
671              _append((footer_style, column.footer))
672  
673          if any_padding:
674              _Padding = Padding
675              for first, last, (style, renderable) in loop_first_last(raw_cells):
676                  yield _Cell(
677                      style,
678                      _Padding(renderable, get_padding(first, last)),
679                      getattr(renderable, "vertical", None) or column.vertical,
680                  )
681          else:
682              for (style, renderable) in raw_cells:
683                  yield _Cell(
684                      style,
685                      renderable,
686                      getattr(renderable, "vertical", None) or column.vertical,
687                  )
688  
689      def _get_padding_width(self, column_index: int) -> int:
690          """Get extra width from padding."""
691          _, pad_right, _, pad_left = self.padding
692          if self.collapse_padding:
693              if column_index > 0:
694                  pad_left = max(0, pad_left - pad_right)
695          return pad_left + pad_right
696  
697      def _measure_column(
698          self,
699          console: "Console",
700          options: "ConsoleOptions",
701          column: Column,
702      ) -> Measurement:
703          """Get the minimum and maximum width of the column."""
704  
705          max_width = options.max_width
706          if max_width < 1:
707              return Measurement(0, 0)
708  
709          padding_width = self._get_padding_width(column._index)
710  
711          if column.width is not None:
712              # Fixed width column
713              return Measurement(
714                  column.width + padding_width, column.width + padding_width
715              ).with_maximum(max_width)
716          # Flexible column, we need to measure contents
717          min_widths: List[int] = []
718          max_widths: List[int] = []
719          append_min = min_widths.append
720          append_max = max_widths.append
721          get_render_width = Measurement.get
722          for cell in self._get_cells(console, column._index, column):
723              _min, _max = get_render_width(console, options, cell.renderable)
724              append_min(_min)
725              append_max(_max)
726  
727          measurement = Measurement(
728              max(min_widths) if min_widths else 1,
729              max(max_widths) if max_widths else max_width,
730          ).with_maximum(max_width)
731          measurement = measurement.clamp(
732              None if column.min_width is None else column.min_width + padding_width,
733              None if column.max_width is None else column.max_width + padding_width,
734          )
735          return measurement

737      def _render(
738          self, console: "Console", options: "ConsoleOptions", widths: List[int]
739      ) -> "RenderResult":
740          table_style = console.get_style(self.style or "")
741  
742          border_style = table_style + console.get_style(self.border_style or "")
743          _column_cells = (
744              self._get_cells(console, column_index, column)
745              for column_index, column in enumerate(self.columns)
746          )
747          row_cells: List[Tuple[_Cell, ...]] = list(zip(*_column_cells))
748          _box = (
749              self.box.substitute(
750                  options, safe=pick_bool(self.safe_box, console.safe_box)
751              )
752              if self.box
753              else None
754          )
755          _box = _box.get_plain_headed_box() if _box and not self.show_header else _box
756  
757          new_line = Segment.line()
758  
759          columns = self.columns
760          show_header = self.show_header
761          show_footer = self.show_footer
762          show_edge = self.show_edge
763          show_lines = self.show_lines
764          leading = self.leading
765  
766          _Segment = Segment
767          if _box:
768              box_segments = [
769                  (
770                      _Segment(_box.head_left, border_style),
771                      _Segment(_box.head_right, border_style),
772                      _Segment(_box.head_vertical, border_style),
773                  ),
774                  (
775                      _Segment(_box.foot_left, border_style),
776                      _Segment(_box.foot_right, border_style),
777                      _Segment(_box.foot_vertical, border_style),
778                  ),
779                  (
780                      _Segment(_box.mid_left, border_style),
781                      _Segment(_box.mid_right, border_style),
782                      _Segment(_box.mid_vertical, border_style),
783                  ),
784              ]
785              if show_edge:
786                  yield _Segment(_box.get_top(widths), border_style)
787                  yield new_line
788          else:
789              box_segments = []
790  
791          get_row_style = self.get_row_style
792          get_style = console.get_style
793  
794          for index, (first, last, row_cell) in enumerate(loop_first_last(row_cells)):
795              header_row = first and show_header
796              footer_row = last and show_footer
797              row = (
798                  self.rows[index - show_header]
799                  if (not header_row and not footer_row)
800                  else None
801              )
802              max_height = 1
803              cells: List[List[List[Segment]]] = []
804              if header_row or footer_row:
805                  row_style = Style.null()
806              else:
807                  row_style = get_style(
808                      get_row_style(console, index - 1 if show_header else index)
809                  )
810              for width, cell, column in zip(widths, row_cell, columns):
811                  render_options = options.update(
812                      width=width,
813                      justify=column.justify,
814                      no_wrap=column.no_wrap,
815                      overflow=column.overflow,
816                      height=None,
817                  )
818                  lines = console.render_lines(
819                      cell.renderable,
820                      render_options,
821                      style=get_style(cell.style) + row_style,
822                  )
823                  max_height = max(max_height, len(lines))
824                  cells.append(lines)
825  
826              row_height = max(len(cell) for cell in cells)
827  
828              def align_cell(
829                  cell: List[List[Segment]],
830                  vertical: "VerticalAlignMethod",
831                  width: int,
832                  style: Style,
833              ) -> List[List[Segment]]:
834                  if header_row:
835                      vertical = "bottom"
836                  elif footer_row:
837                      vertical = "top"
838  
839                  if vertical == "top":
840                      return _Segment.align_top(cell, width, row_height, style)
841                  elif vertical == "middle":
842                      return _Segment.align_middle(cell, width, row_height, style)
843                  return _Segment.align_bottom(cell, width, row_height, style)
844  
845              cells[:] = [
846                  _Segment.set_shape(
847                      align_cell(
848                          cell,
849                          _cell.vertical,
850                          width,
851                          get_style(_cell.style) + row_style,
852                      ),
853                      width,
854                      max_height,
855                  )
856                  for width, _cell, cell, column in zip(widths, row_cell, cells, columns)
857              ]
858  
859              if _box:
860                  if last and show_footer:
861                      yield _Segment(
862                          _box.get_row(widths, "foot", edge=show_edge), border_style
863                      )
864                      yield new_line
865                  left, right, _divider = box_segments[0 if first else (2 if last else 1)]
866  
867                  # If the column divider is whitespace also style it with the row background
868                  divider = (
869                      _divider
870                      if _divider.text.strip()
871                      else _Segment(
872                          _divider.text, row_style.background_style + _divider.style
873                      )
874                  )
875                  for line_no in range(max_height):
876                      if show_edge:
877                          yield left
878                      for last_cell, rendered_cell in loop_last(cells):
879                          yield from rendered_cell[line_no]
880                          if not last_cell:
881                              yield divider
882                      if show_edge:
883                          yield right
884                      yield new_line
885              else:
886                  for line_no in range(max_height):
887                      for rendered_cell in cells:
888                          yield from rendered_cell[line_no]
889                      yield new_line
890              if _box and first and show_header:
891                  yield _Segment(
892                      _box.get_row(widths, "head", edge=show_edge), border_style
893                  )
894                  yield new_line
895              end_section = row and row.end_section
896              if _box and (show_lines or leading or end_section):
897                  if (
898                      not last
899                      and not (show_footer and index >= len(row_cells) - 2)
900                      and not (show_header and header_row)
901                  ):
902                      if leading:
903                          yield _Segment(
904                              _box.get_row(widths, "mid", edge=show_edge) * leading,
905                              border_style,
906                          )
907                      else:
908                          yield _Segment(
909                              _box.get_row(widths, "row", edge=show_edge), border_style
910                          )
911                      yield new_line
912  
913          if _box and show_edge:
914              yield _Segment(_box.get_bottom(widths), border_style)
915              yield new_line
