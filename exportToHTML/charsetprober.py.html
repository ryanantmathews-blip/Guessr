65       @staticmethod
66       def filter_high_byte_only(buf):
67           buf = re.sub(b"([\x00-\x7F])+", b" ", buf)
68           return buf
69   
70       @staticmethod
71       def filter_international_words(buf):
72           """ 
73           We define three types of bytes: 
74           alphabet: english alphabets [a-zA-Z] 
75           international: international characters [\x80-\xFF] 
76           marker: everything else [^a-zA-Z\x80-\xFF] 
77           The input buffer can be thought to contain a series of words delimited 
78           by markers. This function works to filter all words that contain at 
79           least one international character. All contiguous sequences of markers 
80           are replaced by a single space ascii character. 
81           This filter applies to all scripts which do not use English characters. 
82           """
83           filtered = bytearray()
84   
85           # This regex expression filters out only words that have at-least one
86           # international character. The word may include one marker character at
87           # the end.
88           words = INTERNATIONAL_WORDS_PATTERN.findall(buf)
89   
90           for word in words:
91               filtered.extend(word[:-1])
92   
93               # If the last character in the word is a marker, replace it with a
94               # space as markers shouldn't affect our analysis (they are used
95               # similarly across all languages and may thus have similar
96               # frequencies).
97               last_char = word[-1:]
98               if not last_char.isalpha() and last_char < b"\x80":
99                   last_char = b" "
100              filtered.extend(last_char)
101  
102          return filtered

104      @staticmethod
105      def remove_xml_tags(buf):
106          """ 
107          Returns a copy of ``buf`` that retains only the sequences of English 
108          alphabet and high byte characters that are not between <> characters. 
109          This filter can be applied to all scripts which contain both English 
110          characters and extended ASCII characters, but is currently only used by 
111          ``Latin1Prober``. 
112          """
113          filtered = bytearray()
114          in_tag = False
115          prev = 0
116          buf = memoryview(buf).cast("c")
117  
118          for curr, buf_char in enumerate(buf):
119              # Check if we're coming out of or entering an XML tag
120              if buf_char == b">":
121                  prev = curr + 1
122                  in_tag = False
123              elif buf_char == b"<":
124                  if curr > prev and not in_tag:
125                      # Keep everything after last non-extended-ASCII,
126                      # non-alphabetic character
127                      filtered.extend(buf[prev:curr])
128                      # Output a space to delimit stretch we kept
129                      filtered.extend(b" ")
130                  in_tag = True
131  
132          # If we're not in a tag...
133          if not in_tag:
134              # Keep everything after last non-extended-ASCII, non-alphabetic
135              # character
136              filtered.extend(buf[prev:])
137  
138          return filtered
