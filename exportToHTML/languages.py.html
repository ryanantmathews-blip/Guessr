5    This code is based on the language metadata from the uchardet project. 
6    """

8    from string import ascii_letters

13   class Language:
14       """Metadata about a language useful for training models 
15    
16       :ivar name: The human name for the language, in English. 
17       :type name: str 
18       :ivar iso_code: 2-letter ISO 639-1 if possible, 3-letter ISO code otherwise, 
19                       or use another catalog as a last resort. 
20       :type iso_code: str 
21       :ivar use_ascii: Whether or not ASCII letters should be included in trained 
22                        models. 
23       :type use_ascii: bool 
24       :ivar charsets: The charsets we want to support and create data for. 
25       :type charsets: list of str 
26       :ivar alphabet: The characters in the language's alphabet. If `use_ascii` is 
27                       `True`, you only need to add those not in the ASCII set. 
28       :type alphabet: str 
29       :ivar wiki_start_pages: The Wikipedia pages to start from if we're crawling 
30                               Wikipedia for training data. 
31       :type wiki_start_pages: list of str 
32       """
33   
34       def __init__(
35           self,
36           name=None,
37           iso_code=None,
38           use_ascii=True,
39           charsets=None,
40           alphabet=None,
41           wiki_start_pages=None,
42       ):
43           super().__init__()
44           self.name = name
45           self.iso_code = iso_code
46           self.use_ascii = use_ascii
47           self.charsets = charsets
48           if self.use_ascii:
49               if alphabet:
50                   alphabet += ascii_letters
51               else:
52                   alphabet = ascii_letters
53           elif not alphabet:
54               raise ValueError("Must supply alphabet if use_ascii is False")
55           self.alphabet = "".join(sorted(set(alphabet))) if alphabet else None
56           self.wiki_start_pages = wiki_start_pages
57   
58       def __repr__(self):
59           param_str = ", ".join(
60               f"{k}={v!r}" for k, v in self.__dict__.items() if not k.startswith("_")
61           )
62           return f"{self.__class__.__name__}({param_str})"
