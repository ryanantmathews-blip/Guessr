21   class DownloadCommand(RequirementCommand):
22       """ 
23       Download packages from: 
24    
25       - PyPI (and other indexes) using requirement specifiers. 
26       - VCS project urls. 
27       - Local project directories. 
28       - Local or remote source archives. 
29    
30       pip also supports downloading from "requirements files", which provide 
31       an easy way to specify a whole environment to be downloaded. 
32       """
33   
34       usage = """ 
35         %prog [options] <requirement specifier> [package-index-options] ... 
36         %prog [options] -r <requirements file> [package-index-options] ... 
37         %prog [options] <vcs project url> ... 
38         %prog [options] <local project path> ... 
39         %prog [options] <archive url/path> ..."""
40   
41       def add_options(self) -> None:
42           self.cmd_opts.add_option(cmdoptions.constraints())
43           self.cmd_opts.add_option(cmdoptions.requirements())
44           self.cmd_opts.add_option(cmdoptions.no_deps())
45           self.cmd_opts.add_option(cmdoptions.global_options())
46           self.cmd_opts.add_option(cmdoptions.no_binary())
47           self.cmd_opts.add_option(cmdoptions.only_binary())
48           self.cmd_opts.add_option(cmdoptions.prefer_binary())
49           self.cmd_opts.add_option(cmdoptions.src())
50           self.cmd_opts.add_option(cmdoptions.pre())
51           self.cmd_opts.add_option(cmdoptions.require_hashes())
52           self.cmd_opts.add_option(cmdoptions.progress_bar())
53           self.cmd_opts.add_option(cmdoptions.no_build_isolation())
54           self.cmd_opts.add_option(cmdoptions.use_pep517())
55           self.cmd_opts.add_option(cmdoptions.no_use_pep517())
56           self.cmd_opts.add_option(cmdoptions.check_build_deps())
57           self.cmd_opts.add_option(cmdoptions.ignore_requires_python())
58   
59           self.cmd_opts.add_option(
60               "-d",
61               "--dest",
62               "--destination-dir",
63               "--destination-directory",
64               dest="download_dir",
65               metavar="dir",
66               default=os.curdir,
67               help="Download packages into <dir>.",
68           )
69   
70           cmdoptions.add_target_python_options(self.cmd_opts)
71   
72           index_opts = cmdoptions.make_option_group(
73               cmdoptions.index_group,
74               self.parser,
75           )
76   
77           self.parser.insert_option_group(0, index_opts)
78           self.parser.insert_option_group(0, self.cmd_opts)
79   
80       @with_cleanup
81       def run(self, options: Values, args: List[str]) -> int:
82   
83           options.ignore_installed = True
84           # editable doesn't really make sense for `pip download`, but the bowels
85           # of the RequirementSet code require that property.
86           options.editables = []
87   
88           cmdoptions.check_dist_restriction(options)
89   
90           options.download_dir = normalize_path(options.download_dir)
91           ensure_dir(options.download_dir)
92   
93           session = self.get_default_session(options)
94   
95           target_python = make_target_python(options)
96           finder = self._build_package_finder(
97               options=options,
98               session=session,
99               target_python=target_python,
100              ignore_requires_python=options.ignore_requires_python,
101          )
102  
103          build_tracker = self.enter_context(get_build_tracker())
104  
105          directory = TempDirectory(
106              delete=not options.no_clean,
107              kind="download",
108              globally_managed=True,
109          )
110  
111          reqs = self.get_requirements(args, options, finder, session)
112          check_legacy_setup_py_options(
113              options, reqs, LegacySetupPyOptionsCheckMode.DOWNLOAD
114          )
115  
116          preparer = self.make_requirement_preparer(
117              temp_build_dir=directory,
118              options=options,
119              build_tracker=build_tracker,
120              session=session,
121              finder=finder,
122              download_dir=options.download_dir,
123              use_user_site=False,
124              verbosity=self.verbosity,
125          )
126  
127          resolver = self.make_resolver(
128              preparer=preparer,
129              finder=finder,
130              options=options,
131              ignore_requires_python=options.ignore_requires_python,
132              use_pep517=options.use_pep517,
133              py_version_info=options.python_version,
134          )
135  
136          self.trace_basic_info(finder)
137  
138          requirement_set = resolver.resolve(reqs, check_supported_wheels=True)
139  
140          downloaded: List[str] = []
141          for req in requirement_set.requirements.values():
142              if req.satisfied_by is None:
143                  assert req.name is not None
144                  preparer.save_linked_requirement(req)
145                  downloaded.append(req.name)
146          if downloaded:
147              write_output("Successfully downloaded %s", " ".join(downloaded))
148  
149          return SUCCESS
