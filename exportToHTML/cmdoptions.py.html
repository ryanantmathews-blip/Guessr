36   def raise_option_error(parser: OptionParser, option: Option, msg: str) -> None:
37       """ 
38       Raise an option parsing error using parser.error(). 
39    
40       Args: 
41         parser: an OptionParser instance. 
42         option: an Option instance. 
43         msg: the error text. 
44       """
45       msg = f"{option} error: {msg}"
46       msg = textwrap.fill(" ".join(msg.split()))
47       parser.error(msg)

50   def make_option_group(group: Dict[str, Any], parser: ConfigOptionParser) -> OptionGroup:
51       """ 
52       Return an OptionGroup object 
53       group  -- assumed to be dict with 'name' and 'options' keys 
54       parser -- an optparse Parser 
55       """
56       option_group = OptionGroup(parser, group["name"])
57       for option in group["options"]:
58           option_group.add_option(option())
59       return option_group

62   def check_dist_restriction(options: Values, check_target: bool = False) -> None:
63       """Function for determining if custom platform options are allowed. 
64    
65       :param options: The OptionParser options. 
66       :param check_target: Whether or not to check if --target is being used. 
67       """
68       dist_restriction_set = any(
69           [
70               options.python_version,
71               options.platforms,
72               options.abis,
73               options.implementation,
74           ]
75       )
76   
77       binary_only = FormatControl(set(), {":all:"})
78       sdist_dependencies_allowed = (
79           options.format_control != binary_only and not options.ignore_dependencies
80       )
81   
82       # Installations or downloads using dist restrictions must not combine
83       # source distributions and dist-specific wheels, as they are not
84       # guaranteed to be locally compatible.
85       if dist_restriction_set and sdist_dependencies_allowed:
86           raise CommandError(
87               "When restricting platform and interpreter constraints using "
88               "--python-version, --platform, --abi, or --implementation, "
89               "either --no-deps must be set, or --only-binary=:all: must be "
90               "set and --no-binary must not be set (or must be set to "
91               ":none:)."
92           )
93   
94       if check_target:
95           if dist_restriction_set and not options.target_dir:
96               raise CommandError(
97                   "Can not use any platform or abi specific options unless "
98                   "installing via '--target'"
99               )

526  # This was made a separate function for unit-testing purposes.
527  def _convert_python_version(value: str) -> Tuple[Tuple[int, ...], Optional[str]]:
528      """ 
529      Convert a version string like "3", "37", or "3.7.3" into a tuple of ints. 
530   
531      :return: A 2-tuple (version_info, error_msg), where `error_msg` is 
532          non-None if and only if there was a parsing error. 
533      """
534      if not value:
535          # The empty string is the same as not providing a value.
536          return (None, None)
537  
538      parts = value.split(".")
539      if len(parts) > 3:
540          return ((), "at most three version parts are allowed")
541  
542      if len(parts) == 1:
543          # Then we are in the case of "3" or "37".
544          value = parts[0]
545          if len(value) > 1:
546              parts = [value[0], value[1:]]
547  
548      try:
549          version_info = tuple(int(part) for part in parts)
550      except ValueError:
551          return ((), "each version part must be an integer")
552  
553      return (version_info, None)

556  def _handle_python_version(
557      option: Option, opt_str: str, value: str, parser: OptionParser
558  ) -> None:
559      """ 
560      Handle a provided --python-version value. 
561      """
562      version_info, error_msg = _convert_python_version(value)
563      if error_msg is not None:
564          msg = "invalid --python-version value: {!r}: {}".format(
565              value,
566              error_msg,
567          )
568          raise_option_error(parser, option=option, msg=msg)
569  
570      parser.values.python_version = version_info

667  def _handle_no_cache_dir(
668      option: Option, opt: str, value: str, parser: OptionParser
669  ) -> None:
670      """ 
671      Process a value provided for the --no-cache-dir option. 
672   
673      This is an optparse.Option callback for the --no-cache-dir option. 
674      """
675      # The value argument will be None if --no-cache-dir is passed via the
676      # command-line, since the option doesn't accept arguments.  However,
677      # the value can be non-None if the option is triggered e.g. by an
678      # environment variable, like PIP_NO_CACHE_DIR=true.
679      if value is not None:
680          # Then parse the string value to get argument error-checking.
681          try:
682              strtobool(value)
683          except ValueError as exc:
684              raise_option_error(parser, option=option, msg=str(exc))
685  
686      # Originally, setting PIP_NO_CACHE_DIR to a value that strtobool()
687      # converted to 0 (like "false" or "no") caused cache_dir to be disabled
688      # rather than enabled (logic would say the latter).  Thus, we disable
689      # the cache directory not just on values that parse to True, but (for
690      # backwards compatibility reasons) also on values that parse to False.
691      # In other words, always set it to False if the option is provided in
692      # some (valid) form.
693      parser.values.cache_dir = False
694

744  def _handle_no_use_pep517(
745      option: Option, opt: str, value: str, parser: OptionParser
746  ) -> None:
747      """ 
748      Process a value provided for the --no-use-pep517 option. 
749   
750      This is an optparse.Option callback for the no_use_pep517 option. 
751      """
752      # Since --no-use-pep517 doesn't accept arguments, the value argument
753      # will be None if --no-use-pep517 is passed via the command-line.
754      # However, the value can be non-None if the option is triggered e.g.
755      # by an environment variable, for example "PIP_NO_USE_PEP517=true".
756      if value is not None:
757          msg = """A value was passed for --no-use-pep517, 
758          probably using either the PIP_NO_USE_PEP517 environment variable 
759          or the "no-use-pep517" config file option. Use an appropriate value 
760          of the PIP_USE_PEP517 environment variable or the "use-pep517" 
761          config file option instead. 
762          """
763          raise_option_error(parser, option=option, msg=msg)
764  
765      # If user doesn't wish to use pep517, we check if setuptools is installed
766      # and raise error if it is not.
767      if not importlib.util.find_spec("setuptools"):
768          msg = "It is not possible to use --no-use-pep517 without setuptools installed."
769          raise_option_error(parser, option=option, msg=msg)
770  
771      # Otherwise, --no-use-pep517 was passed via the command-line.
772      parser.values.use_pep517 = False

796  def _handle_config_settings(
797      option: Option, opt_str: str, value: str, parser: OptionParser
798  ) -> None:
799      key, sep, val = value.partition("=")
800      if sep != "=":
801          parser.error(f"Arguments to {opt_str} must be of the form KEY=VAL")  # noqa
802      dest = getattr(parser.values, option.dest)
803      if dest is None:
804          dest = {}
805          setattr(parser.values, option.dest, dest)
806      dest[key] = val

891  def _handle_merge_hash(
892      option: Option, opt_str: str, value: str, parser: OptionParser
893  ) -> None:
894      """Given a value spelled "algo:digest", append the digest to a list 
895      pointed to in a dict by the algo name."""
896      if not parser.values.hashes:
897          parser.values.hashes = {}
898      try:
899          algo, digest = value.split(":", 1)
900      except ValueError:
901          parser.error(
902              "Arguments to {} must be a hash name "  # noqa
903              "followed by a value, like --hash=sha256:"
904              "abcde...".format(opt_str)
905          )
906      if algo not in STRONG_HASHES:
907          parser.error(
908              "Allowed hash algorithms for {} are {}.".format(  # noqa
909                  opt_str, ", ".join(STRONG_HASHES)
910              )
911          )
912      parser.values.hashes.setdefault(algo, []).append(digest)
