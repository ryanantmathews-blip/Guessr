10   from typing import Iterator

12   from pkg_resources import Requirement, safe_extra, split_sections

15   def requires_to_requires_dist(requirement: Requirement) -> str:
16       """Return the version specifier for a requirement in PEP 345/566 fashion."""
17       if getattr(requirement, "url", None):
18           return " @ " + requirement.url
19   
20       requires_dist = []
21       for op, ver in requirement.specs:
22           requires_dist.append(op + ver)
23   
24       if requires_dist:
25           return " (" + ",".join(sorted(requires_dist)) + ")"
26       else:
27           return ""
28

30   def convert_requirements(requirements: list[str]) -> Iterator[str]:
31       """Yield Requires-Dist: strings for parsed requirements strings."""
32       for req in requirements:
33           parsed_requirement = Requirement.parse(req)
34           spec = requires_to_requires_dist(parsed_requirement)
35           extras = ",".join(sorted(parsed_requirement.extras))
36           if extras:
37               extras = f"[{extras}]"
38   
39           yield parsed_requirement.project_name + extras + spec

42   def generate_requirements(
43       extras_require: dict[str, list[str]]
44   ) -> Iterator[tuple[str, str]]:
45       """ 
46       Convert requirements from a setup()-style dictionary to 
47       ('Requires-Dist', 'requirement') and ('Provides-Extra', 'extra') tuples. 
48    
49       extras_require is a dictionary of {extra: [requirements]} as passed to setup(), 
50       using the empty extra {'': [requirements]} to hold install_requires. 
51       """
52       for extra, depends in extras_require.items():
53           condition = ""
54           extra = extra or ""
55           if ":" in extra:  # setuptools extra:condition syntax
56               extra, condition = extra.split(":", 1)
57   
58           extra = safe_extra(extra)
59           if extra:
60               yield "Provides-Extra", extra
61               if condition:
62                   condition = "(" + condition + ") and "
63               condition += "extra == '%s'" % extra
64   
65           if condition:
66               condition = " ; " + condition
67   
68           for new_req in convert_requirements(depends):
69               yield "Requires-Dist", new_req + condition
70

72   def pkginfo_to_metadata(egg_info_path: str, pkginfo_path: str) -> Message:
73       """ 
74       Convert .egg-info directory with PKG-INFO to the Metadata 2.1 format 
75       """
76       with open(pkginfo_path, encoding="utf-8") as headers:
77           pkg_info = Parser().parse(headers)
78   
79       pkg_info.replace_header("Metadata-Version", "2.1")
80       # Those will be regenerated from `requires.txt`.
81       del pkg_info["Provides-Extra"]
82       del pkg_info["Requires-Dist"]
83       requires_path = os.path.join(egg_info_path, "requires.txt")
84       if os.path.exists(requires_path):
85           with open(requires_path) as requires_file:
86               requires = requires_file.read()
87   
88           parsed_requirements = sorted(split_sections(requires), key=lambda x: x[0] or "")
89           for extra, reqs in parsed_requirements:
90               for key, value in generate_requirements({extra: reqs}):
91                   if (key, value) not in pkg_info.items():
92                       pkg_info[key] = value
93   
94       description = pkg_info["Description"]
95       if description:
96           description_lines = pkg_info["Description"].splitlines()
97           dedented_description = "\n".join(
98               # if the first line of long_description is blank,
99               # the first line here will be indented.
100              (
101                  description_lines[0].lstrip(),
102                  textwrap.dedent("\n".join(description_lines[1:])),
103                  "\n",
104              )
105          )
106          pkg_info.set_payload(dedented_description)
107          del pkg_info["Description"]
108  
109      return pkg_info
