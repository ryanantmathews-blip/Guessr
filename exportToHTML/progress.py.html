63   class _TrackThread(Thread):
64       """A thread to periodically update progress."""
65   
66       def __init__(self, progress: "Progress", task_id: "TaskID", update_period: float):
67           self.progress = progress
68           self.task_id = task_id
69           self.update_period = update_period
70           self.done = Event()
71   
72           self.completed = 0
73           super().__init__()
74   
75       def run(self) -> None:
76           task_id = self.task_id
77           advance = self.progress.advance
78           update_period = self.update_period
79           last_completed = 0
80           wait = self.done.wait
81           while not wait(update_period):
82               completed = self.completed
83               if last_completed != completed:
84                   advance(task_id, completed - last_completed)
85                   last_completed = completed
86   
87           self.progress.update(self.task_id, completed=self.completed, refresh=True)
88   
89       def __enter__(self) -> "_TrackThread":
90           self.start()
91           return self
92   
93       def __exit__(
94           self,
95           exc_type: Optional[Type[BaseException]],
96           exc_val: Optional[BaseException],
97           exc_tb: Optional[TracebackType],
98       ) -> None:
99           self.done.set()
100          self.join()

103  def track(
104      sequence: Union[Sequence[ProgressType], Iterable[ProgressType]],
105      description: str = "Working...",
106      total: Optional[float] = None,
107      auto_refresh: bool = True,
108      console: Optional[Console] = None,
109      transient: bool = False,
110      get_time: Optional[Callable[[], float]] = None,
111      refresh_per_second: float = 10,
112      style: StyleType = "bar.back",
113      complete_style: StyleType = "bar.complete",
114      finished_style: StyleType = "bar.finished",
115      pulse_style: StyleType = "bar.pulse",
116      update_period: float = 0.1,
117      disable: bool = False,
118      show_speed: bool = True,
119  ) -> Iterable[ProgressType]:
120      """Track progress by iterating over a sequence. 
121   
122      Args: 
123          sequence (Iterable[ProgressType]): A sequence (must support "len") you wish to iterate over. 
124          description (str, optional): Description of task show next to progress bar. Defaults to "Working". 
125          total: (float, optional): Total number of steps. Default is len(sequence). 
126          auto_refresh (bool, optional): Automatic refresh, disable to force a refresh after each iteration. Default is True. 
127          transient: (bool, optional): Clear the progress on exit. Defaults to False. 
128          console (Console, optional): Console to write to. Default creates internal Console instance. 
129          refresh_per_second (float): Number of times per second to refresh the progress information. Defaults to 10. 
130          style (StyleType, optional): Style for the bar background. Defaults to "bar.back". 
131          complete_style (StyleType, optional): Style for the completed bar. Defaults to "bar.complete". 
132          finished_style (StyleType, optional): Style for a finished bar. Defaults to "bar.done". 
133          pulse_style (StyleType, optional): Style for pulsing bars. Defaults to "bar.pulse". 
134          update_period (float, optional): Minimum time (in seconds) between calls to update(). Defaults to 0.1. 
135          disable (bool, optional): Disable display of progress. 
136          show_speed (bool, optional): Show speed if total isn't known. Defaults to True. 
137      Returns: 
138          Iterable[ProgressType]: An iterable of the values in the sequence. 
139   
140      """
141  
142      columns: List["ProgressColumn"] = (
143          [TextColumn("[progress.description]{task.description}")] if description else []
144      )
145      columns.extend(
146          (
147              BarColumn(
148                  style=style,
149                  complete_style=complete_style,
150                  finished_style=finished_style,
151                  pulse_style=pulse_style,
152              ),
153              TaskProgressColumn(show_speed=show_speed),
154              TimeRemainingColumn(),
155          )
156      )
157      progress = Progress(
158          *columns,
159          auto_refresh=auto_refresh,
160          console=console,
161          transient=transient,
162          get_time=get_time,
163          refresh_per_second=refresh_per_second or 10,
164          disable=disable,
165      )
166  
167      with progress:
168          yield from progress.track(
169              sequence, total=total, description=description, update_period=update_period
170          )

173  class _Reader(RawIOBase, BinaryIO):
174      """A reader that tracks progress while it's being read from."""
175  
176      def __init__(
177          self,
178          handle: BinaryIO,
179          progress: "Progress",
180          task: TaskID,
181          close_handle: bool = True,
182      ) -> None:
183          self.handle = handle
184          self.progress = progress
185          self.task = task
186          self.close_handle = close_handle
187          self._closed = False
188  
189      def __enter__(self) -> "_Reader":
190          self.handle.__enter__()
191          return self
192  
193      def __exit__(
194          self,
195          exc_type: Optional[Type[BaseException]],
196          exc_val: Optional[BaseException],
197          exc_tb: Optional[TracebackType],
198      ) -> None:
199          self.close()
200  
201      def __iter__(self) -> BinaryIO:
202          return self
203  
204      def __next__(self) -> bytes:
205          line = next(self.handle)
206          self.progress.advance(self.task, advance=len(line))
207          return line
208  
209      @property
210      def closed(self) -> bool:
211          return self._closed
212  
213      def fileno(self) -> int:
214          return self.handle.fileno()
215  
216      def isatty(self) -> bool:
217          return self.handle.isatty()
218  
219      @property
220      def name(self) -> str:
221          return self.handle.name
222  
223      def readable(self) -> bool:
224          return self.handle.readable()
225  
226      def seekable(self) -> bool:
227          return self.handle.seekable()
228  
229      def writable(self) -> bool:
230          return False
231  
232      def read(self, size: int = -1) -> bytes:
233          block = self.handle.read(size)
234          self.progress.advance(self.task, advance=len(block))
235          return block
236  
237      def readinto(self, b: Union[bytearray, memoryview, mmap]):  # type: ignore[no-untyped-def, override]
238          n = self.handle.readinto(b)  # type: ignore[attr-defined]
239          self.progress.advance(self.task, advance=n)
240          return n
241  
242      def readline(self, size: int = -1) -> bytes:  # type: ignore[override]
243          line = self.handle.readline(size)
244          self.progress.advance(self.task, advance=len(line))
245          return line
246  
247      def readlines(self, hint: int = -1) -> List[bytes]:
248          lines = self.handle.readlines(hint)
249          self.progress.advance(self.task, advance=sum(map(len, lines)))
250          return lines
251  
252      def close(self) -> None:
253          if self.close_handle:
254              self.handle.close()
255          self._closed = True
256  
257      def seek(self, offset: int, whence: int = 0) -> int:
258          pos = self.handle.seek(offset, whence)
259          self.progress.update(self.task, completed=pos)
260          return pos
261  
262      def tell(self) -> int:
263          return self.handle.tell()
264  
265      def write(self, s: Any) -> int:
266          raise UnsupportedOperation("write")

491  class ProgressColumn(ABC):
492      """Base class for a widget to use in progress display."""
493  
494      max_refresh: Optional[float] = None
495  
496      def __init__(self, table_column: Optional[Column] = None) -> None:
497          self._table_column = table_column
498          self._renderable_cache: Dict[TaskID, Tuple[float, RenderableType]] = {}
499          self._update_time: Optional[float] = None
500  
501      def get_table_column(self) -> Column:
502          """Get a table column, used to build tasks table."""
503          return self._table_column or Column()
504  
505      def __call__(self, task: "Task") -> RenderableType:
506          """Called by the Progress object to return a renderable for the given task. 
507   
508          Args: 
509              task (Task): An object containing information regarding the task. 
510   
511          Returns: 
512              RenderableType: Anything renderable (including str). 
513          """
514          current_time = task.get_time()
515          if self.max_refresh is not None and not task.completed:
516              try:
517                  timestamp, renderable = self._renderable_cache[task.id]
518              except KeyError:
519                  pass
520              else:
521                  if timestamp + self.max_refresh > current_time:
522                      return renderable
523  
524          renderable = self.render(task)
525          self._renderable_cache[task.id] = (current_time, renderable)
526          return renderable
527  
528      @abstractmethod
529      def render(self, task: "Task") -> RenderableType:
530          """Should return a renderable object."""

573          )
574          super().__init__(table_column=table_column)

577          self,
578          spinner_name: str,
579          spinner_style: Optional[StyleType] = "progress.spinner",

626              self.highlighter.highlight(text)

688          text_format (str, optional): Format for percentage display. Defaults to "[progress.percentage]{task.percentage:>3.0f}%".

690          style (StyleType, optional): Style of output. Defaults to "none". 
691          justify (JustifyMethod, optional): Text justification. Defaults to "left". 
692          markup (bool, optional): Enable markup. Defaults to True. 
693          highlighter (Optional[Highlighter], optional): Highlighter to apply to output. Defaults to None. 
694          table_column (Optional[Column], optional): Table Column to use. Defaults to None. 
695          show_speed (bool, optional): Show speed if total is unknown. Defaults to False.

699          self,
700          text_format: str = "[progress.percentage]{task.percentage:>3.0f}%",
701          text_format_no_percentage: str = "",
702          style: StyleType = "none",
703          justify: JustifyMethod = "left",
704          markup: bool = True,
705          highlighter: Optional[Highlighter] = None,
706          table_column: Optional[Column] = None,
707          show_speed: bool = False,
708      ) -> None:
709  
710          self.text_format_no_percentage = text_format_no_percentage
711          self.show_speed = show_speed
712          super().__init__(
713              text_format=text_format,
714              style=style,
715              justify=justify,
716              markup=markup,
717              highlighter=highlighter,
718              table_column=table_column,
719          )

721      @classmethod
722      def render_speed(cls, speed: Optional[float]) -> Text:
723          """Render the speed in iterations per second. 
724   
725          Args: 
726              task (Task): A Task object. 
727   
728          Returns: 
729              Text: Text object containing the task speed. 
730          """
731          if speed is None:
732              return Text("", style="progress.percentage")
733          unit, suffix = filesize.pick_unit_and_suffix(
734              int(speed),
735              ["", "10", "10", "10", "10"],
736              1000,
737          )
738          data_speed = speed / unit
739          return Text(f"{data_speed:.1f}{suffix} it/s", style="progress.percentage")

741      def render(self, task: "Task") -> Text:
742          if task.total is None and self.show_speed:
743              return self.render_speed(task.finished_speed or task.speed)
744          text_format = (
745              self.text_format_no_percentage if task.total is None else self.text_format
746          )
747          _text = text_format.format(task=task)
748          if self.markup:
749              text = Text.from_markup(_text, style=self.style, justify=self.justify)
750          else:
751              text = Text(_text, style=self.style, justify=self.justify)
752          if self.highlighter:
753              self.highlighter.highlight(text)
754          return text

920  @dataclass
921  class Task:
922      """Information regarding a progress task. 
923   
924      This object should be considered read-only outside of the :class:`~Progress` class. 
925   
926      """
927  
928      id: TaskID
929      """Task ID associated with this task (used in Progress methods)."""
930  
931      description: str
932      """str: Description of the task."""
933  
934      total: Optional[float]
935      """Optional[float]: Total number of steps in this task."""
936  
937      completed: float
938      """float: Number of steps completed"""
939  
940      _get_time: GetTimeCallable
941      """Callable to get the current time."""
942  
943      finished_time: Optional[float] = None
944      """float: Time task was finished."""
945  
946      visible: bool = True
947      """bool: Indicates if this task is visible in the progress display."""
948  
949      fields: Dict[str, Any] = field(default_factory=dict)
950      """dict: Arbitrary fields passed in via Progress.update."""
951  
952      start_time: Optional[float] = field(default=None, init=False, repr=False)
953      """Optional[float]: Time this task was started, or None if not started."""
954  
955      stop_time: Optional[float] = field(default=None, init=False, repr=False)
956      """Optional[float]: Time this task was stopped, or None if not stopped."""
957  
958      finished_speed: Optional[float] = None
959      """Optional[float]: The last speed for a finished task."""
960  
961      _progress: Deque[ProgressSample] = field(
962          default_factory=lambda: deque(maxlen=1000), init=False, repr=False
963      )
964  
965      _lock: RLock = field(repr=False, default_factory=RLock)
966      """Thread lock."""
967  
968      def get_time(self) -> float:
969          """float: Get the current time, in seconds."""
970          return self._get_time()
971  
972      @property
973      def started(self) -> bool:
974          """bool: Check if the task as started."""
975          return self.start_time is not None
976  
977      @property
978      def remaining(self) -> Optional[float]:
979          """Optional[float]: Get the number of steps remaining, if a non-None total was set."""
980          if self.total is None:
981              return None
982          return self.total - self.completed
983  
984      @property
985      def elapsed(self) -> Optional[float]:
986          """Optional[float]: Time elapsed since task was started, or ``None`` if the task hasn't started."""
987          if self.start_time is None:
988              return None
989          if self.stop_time is not None:
990              return self.stop_time - self.start_time
991          return self.get_time() - self.start_time
992  
993      @property
994      def finished(self) -> bool:
995          """Check if the task has finished."""
996          return self.finished_time is not None
997  
998      @property
999      def percentage(self) -> float:
1000         """float: Get progress of task as a percentage. If a None total was set, returns 0"""
1001         if not self.total:
1002             return 0.0
1003         completed = (self.completed / self.total) * 100.0
1004         completed = min(100.0, max(0.0, completed))
1005         return completed
1006 
1007     @property
1008     def speed(self) -> Optional[float]:
1009         """Optional[float]: Get the estimated speed in steps per second."""
1010         if self.start_time is None:
1011             return None
1012         with self._lock:
1013             progress = self._progress
1014             if not progress:
1015                 return None
1016             total_time = progress[-1].timestamp - progress[0].timestamp
1017             if total_time == 0:
1018                 return None
1019             iter_progress = iter(progress)
1020             next(iter_progress)
1021             total_completed = sum(sample.completed for sample in iter_progress)
1022             speed = total_completed / total_time
1023             return speed
1024 
1025     @property
1026     def time_remaining(self) -> Optional[float]:
1027         """Optional[float]: Get estimated time to completion, or ``None`` if no data."""
1028         if self.finished:
1029             return 0.0
1030         speed = self.speed
1031         if not speed:
1032             return None
1033         remaining = self.remaining
1034         if remaining is None:
1035             return None
1036         estimate = ceil(remaining / speed)
1037         return estimate
1038 
1039     def _reset(self) -> None:
1040         """Reset progress."""
1041         self._progress.clear()
1042         self.finished_time = None
1043         self.finished_speed = None

1177         self,
1178         sequence: Union[Iterable[ProgressType], Sequence[ProgressType]],
1179         total: Optional[float] = None,
1180         task_id: Optional[TaskID] = None,
1181         description: str = "Working...",
1182         update_period: float = 0.1,
1183     ) -> Iterable[ProgressType]:
1184         """Track progress by iterating over a sequence. 
1185  
1186         Args: 
1187             sequence (Sequence[ProgressType]): A sequence of values you want to iterate over and track progress. 
1188             total: (float, optional): Total number of steps. Default is len(sequence). 
1189             task_id: (TaskID): Task to track. Default is new task. 
1190             description: (str, optional): Description of task, if new task is created. 
1191             update_period (float, optional): Minimum time (in seconds) between calls to update(). Defaults to 0.1. 
1192  
1193         Returns: 
1194             Iterable[ProgressType]: An iterable of values taken from the provided sequence. 
1195         """
1196 
1197         task_total: Optional[float] = None
1198         if total is None:
1199             if isinstance(sequence, Sized):
1200                 task_total = float(len(sequence))
1201         else:
1202             task_total = total
1203 
1204         if task_id is None:
1205             task_id = self.add_task(description, total=task_total)
1206         else:
1207             self.update(task_id, total=task_total)
1208 
1209         if self.live.auto_refresh:
1210             with _TrackThread(self, task_id, update_period) as track_thread:
1211                 for value in sequence:
1212                     yield value
1213                     track_thread.completed += 1
1214         else:
1215             advance = self.advance
1216             refresh = self.refresh
1217             for value in sequence:
1218                 yield value
1219                 advance(task_id, 1)
1220                 refresh()

1297         self,

1300         buffering: int = -1,
1301         encoding: Optional[str] = None,
1302         errors: Optional[str] = None,
1303         newline: Optional[str] = None,
1304         *,
1305         total: Optional[int] = None,
1306         task_id: Optional[TaskID] = None,
1307         description: str = "Reading...",
1308     ) -> Union[BinaryIO, TextIO]:
1309         """Track progress while reading from a binary file. 
1310  
1311         Args: 
1312             path (Union[str, PathLike[str]]): The path to the file to read. 
1313             mode (str): The mode to use to open the file. Only supports "r", "rb" or "rt". 
1314             buffering (int): The buffering strategy to use, see :func:`io.open`. 
1315             encoding (str, optional): The encoding to use when reading in text mode, see :func:`io.open`. 
1316             errors (str, optional): The error handling strategy for decoding errors, see :func:`io.open`. 
1317             newline (str, optional): The strategy for handling newlines in text mode, see :func:`io.open`. 
1318             total (int, optional): Total number of bytes to read. If none given, os.stat(path).st_size is used. 
1319             task_id (TaskID): Task to track. Default is new task. 
1320             description (str, optional): Description of task, if new task is created. 
1321  
1322         Returns: 
1323             BinaryIO: A readable file-like object in binary mode. 
1324  
1325         Raises: 
1326             ValueError: When an invalid mode is given. 
1327         """
1328         # normalize the mode (always rb, rt)
1329         _mode = "".join(sorted(mode, reverse=False))
1330         if _mode not in ("br", "rt", "r"):
1331             raise ValueError("invalid mode {!r}".format(mode))
1332 
1333         # patch buffering to provide the same behaviour as the builtin `open`
1334         line_buffering = buffering == 1
1335         if _mode == "br" and buffering == 1:
1336             warnings.warn(
1337                 "line buffering (buffering=1) isn't supported in binary mode, the default buffer size will be used",
1338                 RuntimeWarning,
1339             )
1340             buffering = -1
1341         elif _mode == "rt" or _mode == "r":
1342             if buffering == 0:
1343                 raise ValueError("can't have unbuffered text I/O")
1344             elif buffering == 1:
1345                 buffering = -1
1346 
1347         # attempt to get the total with `os.stat`
1348         if total is None:
1349             total = stat(file).st_size
1350 
1351         # update total of task or create new task
1352         if task_id is None:
1353             task_id = self.add_task(description, total=total)
1354         else:
1355             self.update(task_id, total=total)
1356 
1357         # open the file in binary mode,
1358         handle = io.open(file, "rb", buffering=buffering)
1359         reader = _Reader(handle, self, task_id, close_handle=True)
1360 
1361         # wrap the reader in a `TextIOWrapper` if text mode
1362         if mode == "r" or mode == "rt":
1363             return io.TextIOWrapper(
1364                 reader,
1365                 encoding=encoding,
1366                 errors=errors,
1367                 newline=newline,
1368                 line_buffering=line_buffering,
1369             )
1370 
1371         return reader

1403         self,
1404         task_id: TaskID,
1405         *,
1406         total: Optional[float] = None,
1407         completed: Optional[float] = None,
1408         advance: Optional[float] = None,
1409         description: Optional[str] = None,
1410         visible: Optional[bool] = None,
1411         refresh: bool = False,
1412         **fields: Any,
1413     ) -> None:
1414         """Update information associated with a task. 
1415  
1416         Args: 
1417             task_id (TaskID): Task id (returned by add_task). 
1418             total (float, optional): Updates task.total if not None. 
1419             completed (float, optional): Updates task.completed if not None. 
1420             advance (float, optional): Add a value to task.completed if not None. 
1421             description (str, optional): Change task description if not None. 
1422             visible (bool, optional): Set visible flag if not None. 
1423             refresh (bool): Force a refresh of progress information. Default is False. 
1424             **fields (Any): Additional data fields required for rendering. 
1425         """
1426         with self._lock:
1427             task = self._tasks[task_id]
1428             completed_start = task.completed
1429 
1430             if total is not None and total != task.total:
1431                 task.total = total
1432                 task._reset()
1433             if advance is not None:
1434                 task.completed += advance
1435             if completed is not None:
1436                 task.completed = completed
1437             if description is not None:
1438                 task.description = description
1439             if visible is not None:
1440                 task.visible = visible
1441             task.fields.update(fields)
1442             update_completed = task.completed - completed_start
1443 
1444             current_time = self.get_time()
1445             old_sample_time = current_time - self.speed_estimate_period
1446             _progress = task._progress
1447 
1448             popleft = _progress.popleft
1449             while _progress and _progress[0].timestamp < old_sample_time:
1450                 popleft()
1451             if update_completed > 0:
1452                 _progress.append(ProgressSample(current_time, update_completed))
1453             if (
1454                 task.total is not None
1455                 and task.completed >= task.total
1456                 and task.finished_time is None
1457             ):
1458                 task.finished_time = task.elapsed
1459 
1460         if refresh:
1461             self.refresh()

1464         self,
1465         task_id: TaskID,
1466         *,
1467         start: bool = True,
1468         total: Optional[float] = None,
1469         completed: int = 0,
1470         visible: Optional[bool] = None,
1471         description: Optional[str] = None,
1472         **fields: Any,
1473     ) -> None:
1474         """Reset a task so completed is 0 and the clock is reset. 
1475  
1476         Args: 
1477             task_id (TaskID): ID of task. 
1478             start (bool, optional): Start the task after reset. Defaults to True. 
1479             total (float, optional): New total steps in task, or None to use current total. Defaults to None. 
1480             completed (int, optional): Number of steps completed. Defaults to 0. 
1481             visible (bool, optional): Enable display of the task. Defaults to True. 
1482             description (str, optional): Change task description if not None. Defaults to None. 
1483             **fields (str): Additional data fields required for rendering. 
1484         """
1485         current_time = self.get_time()
1486         with self._lock:
1487             task = self._tasks[task_id]
1488             task._reset()
1489             task.start_time = current_time if start else None
1490             if total is not None:
1491                 task.total = total
1492             task.completed = completed
1493             if visible is not None:
1494                 task.visible = visible
1495             if fields:
1496                 task.fields = fields
1497             if description is not None:
1498                 task.description = description
1499             task.finished_time = None
1500         self.refresh()

1503         """Advance task by a number of steps. 
1504  
1505         Args: 
1506             task_id (TaskID): ID of task. 
1507             advance (float): Number of steps to advance. Default is 1. 
1508         """
1509         current_time = self.get_time()

1511             task = self._tasks[task_id]
1512             completed_start = task.completed
1513             task.completed += advance
1514             update_completed = task.completed - completed_start
1515             old_sample_time = current_time - self.speed_estimate_period
1516             _progress = task._progress
1517 
1518             popleft = _progress.popleft
1519             while _progress and _progress[0].timestamp < old_sample_time:
1520                 popleft()

1522                 popleft()

1524             if (
1525                 task.total is not None
1526                 and task.completed >= task.total
1527                 and task.finished_time is None
1528             ):
1529                 task.finished_time = task.elapsed
1530                 task.finished_speed = task.speed
