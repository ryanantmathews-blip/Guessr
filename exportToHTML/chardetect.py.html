chardetect.py 
1    """ 
2    Script which takes one or more file paths and reports on their detected 
3    encodings 
4     
5    Example:: 
6     
7        % chardetect somefile someotherfile 
8        somefile: windows-1252 with confidence 0.5 
9        someotherfile: ascii with confidence 1.0 
10    
11   If no paths are provided, it takes its input from stdin. 
12    
13   """
14   
15   
16   import argparse
17   import sys
18   
19   from .. import __version__
20   from ..universaldetector import UniversalDetector
21   
22   
23   def description_of(lines, name="stdin"):
24       """ 
25       Return a string describing the probable encoding of a file or 
26       list of strings. 
27    
28       :param lines: The lines to get the encoding of. 
29       :type lines: Iterable of bytes 
30       :param name: Name of file or collection of lines 
31       :type name: str 
32       """
33       u = UniversalDetector()
34       for line in lines:
35           line = bytearray(line)
36           u.feed(line)
37           # shortcut out of the loop to save reading further - particularly useful if we read a BOM.
38           if u.done:
39               break
40       u.close()
41       result = u.result
42       if result["encoding"]:
43           return f'{name}: {result["encoding"]} with confidence {result["confidence"]}'
44       return f"{name}: no result"
45   
46   
47   def main(argv=None):
48       """ 
49       Handles command line arguments and gets things started. 
50    
51       :param argv: List of arguments, as if specified on the command-line. 
52                    If None, ``sys.argv[1:]`` is used instead. 
53       :type argv: list of str 
54       """
55       # Get command line arguments
56       parser = argparse.ArgumentParser(
57           description="Takes one or more file paths and reports their detected \ 
58                        encodings"
59       )
60       parser.add_argument(
61           "input",
62           help="File whose encoding we would like to determine. \ 
63                                 (default: stdin)",
64           type=argparse.FileType("rb"),
65           nargs="*",
66           default=[sys.stdin.buffer],
67       )
68       parser.add_argument(
69           "--version", action="version", version=f"%(prog)s {__version__}"
70       )
71       args = parser.parse_args(argv)
72   
73       for f in args.input:
74           if f.isatty():
75               print(
76                   "You are running chardetect interactively. Press "
77                   "CTRL-D twice at the start of a blank line to signal the "
78                   "end of your input. If you want help, run chardetect "
79                   "--help\n",
80                   file=sys.stderr,
81               )
82           print(description_of(f, f.name))
83   
84   
85   if __name__ == "__main__":
86       main()
