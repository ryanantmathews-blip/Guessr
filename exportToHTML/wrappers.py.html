61   def default_subprocess_runner(cmd, cwd=None, extra_environ=None):
62       """The default method of calling the wrapper subprocess."""
63       env = os.environ.copy()
64       if extra_environ:
65           env.update(extra_environ)
66   
67       check_call(cmd, cwd=cwd, env=env)
68   
69   
70   def quiet_subprocess_runner(cmd, cwd=None, extra_environ=None):
71       """A method of calling the wrapper subprocess while suppressing output."""
72       env = os.environ.copy()
73       if extra_environ:
74           env.update(extra_environ)
75   
76       check_output(cmd, cwd=cwd, env=env, stderr=STDOUT)

79   def norm_and_check(source_tree, requested):
80       """Normalise and check a backend path. 
81    
82       Ensure that the requested backend path is specified as a relative path, 
83       and resolves to a location under the given source tree. 
84    
85       Return an absolute version of the requested path. 
86       """
87       if os.path.isabs(requested):
88           raise ValueError("paths must be relative")
89   
90       abs_source = os.path.abspath(source_tree)
91       abs_requested = os.path.normpath(os.path.join(abs_source, requested))
92       # We have to use commonprefix for Python 2.7 compatibility. So we
93       # normalise case to avoid problems because commonprefix is a character
94       # based comparison :-(
95       norm_source = os.path.normcase(abs_source)
96       norm_requested = os.path.normcase(abs_requested)
97       if os.path.commonprefix([norm_source, norm_requested]) != norm_source:
98           raise ValueError("paths must be inside source tree")
99   
100      return abs_requested

103  class Pep517HookCaller:
104      """A wrapper around a source directory to be built with a PEP 517 backend. 
105   
106      :param source_dir: The path to the source directory, containing 
107          pyproject.toml. 
108      :param build_backend: The build backend spec, as per PEP 517, from 
109          pyproject.toml. 
110      :param backend_path: The backend path, as per PEP 517, from pyproject.toml. 
111      :param runner: A callable that invokes the wrapper subprocess. 
112      :param python_executable: The Python executable used to invoke the backend 
113   
114      The 'runner', if provided, must expect the following: 
115   
116      - cmd: a list of strings representing the command and arguments to 
117        execute, as would be passed to e.g. 'subprocess.check_call'. 
118      - cwd: a string representing the working directory that must be 
119        used for the subprocess. Corresponds to the provided source_dir. 
120      - extra_environ: a dict mapping environment variable names to values 
121        which must be set for the subprocess execution. 
122      """
123      def __init__(
124              self,
125              source_dir,
126              build_backend,
127              backend_path=None,
128              runner=None,
129              python_executable=None,
130      ):
131          if runner is None:
132              runner = default_subprocess_runner
133  
134          self.source_dir = abspath(source_dir)
135          self.build_backend = build_backend
136          if backend_path:
137              backend_path = [
138                  norm_and_check(self.source_dir, p) for p in backend_path
139              ]
140          self.backend_path = backend_path
141          self._subprocess_runner = runner
142          if not python_executable:
143              python_executable = sys.executable
144          self.python_executable = python_executable
145  
146      @contextmanager
147      def subprocess_runner(self, runner):
148          """A context manager for temporarily overriding the default subprocess 
149          runner. 
150          """
151          prev = self._subprocess_runner
152          self._subprocess_runner = runner
153          try:
154              yield
155          finally:
156              self._subprocess_runner = prev

178              self, metadata_directory, config_settings=None,
179              _allow_fallback=True):
180          """Prepare a ``*.dist-info`` folder with metadata for this project. 
181

190              'metadata_directory': abspath(metadata_directory),
191              'config_settings': config_settings,
192              '_allow_fallback': _allow_fallback,

197              metadata_directory=None):
198          """Build a wheel from this project.

208              metadata_directory = abspath(metadata_directory)
209          return self._call_hook('build_wheel', {
210              'wheel_directory': abspath(wheel_directory),
211              'config_settings': config_settings,
212              'metadata_directory': metadata_directory,
213          })

295      def _call_hook(self, hook_name, kwargs):
296          extra_environ = {'PEP517_BUILD_BACKEND': self.build_backend}
297  
298          if self.backend_path:
299              backend_path = os.pathsep.join(self.backend_path)
300              extra_environ['PEP517_BACKEND_PATH'] = backend_path
301  
302          with tempfile.TemporaryDirectory() as td:
303              hook_input = {'kwargs': kwargs}
304              write_json(hook_input, pjoin(td, 'input.json'), indent=2)
305  
306              # Run the hook in a subprocess
307              with _in_proc_script_path() as script:
308                  python = self.python_executable
309                  self._subprocess_runner(
310                      [python, abspath(str(script)), hook_name, td],
311                      cwd=self.source_dir,
312                      extra_environ=extra_environ
313                  )
314  
315              data = read_json(pjoin(td, 'output.json'))
316              if data.get('unsupported'):
317                  raise UnsupportedOperation(data.get('traceback', ''))
318              if data.get('no_backend'):
319                  raise BackendUnavailable(data.get('traceback', ''))
320              if data.get('backend_invalid'):
321                  raise BackendInvalid(
322                      backend_name=self.build_backend,
323                      backend_path=self.backend_path,
324                      message=data.get('backend_error', '')
325                  )
326              if data.get('hook_missing'):
327                  raise HookMissing(data.get('missing_hook_name') or hook_name)
328              return data['return_val']

331  class LoggerWrapper(threading.Thread):
332      """ 
333      Read messages from a pipe and redirect them 
334      to a logger (see python's logging module). 
335      """
336  
337      def __init__(self, logger, level):
338          threading.Thread.__init__(self)
339          self.daemon = True
340  
341          self.logger = logger
342          self.level = level
343  
344          # create the pipe and reader
345          self.fd_read, self.fd_write = os.pipe()
346          self.reader = os.fdopen(self.fd_read)
347  
348          self.start()
349  
350      def fileno(self):
351          return self.fd_write
352  
353      @staticmethod
354      def remove_newline(msg):
355          return msg[:-1] if msg.endswith(os.linesep) else msg
356  
357      def run(self):
358          for line in self.reader:
359              self._write(self.remove_newline(line))
360  
361      def _write(self, message):
362          self.logger.log(self.level, message)
