29   from collections import namedtuple
30   
31   from .charsetprober import CharSetProber
32   from .enums import CharacterCategory, ProbingState, SequenceLikelihood
33   
34   SingleByteCharSetModel = namedtuple(
35       "SingleByteCharSetModel",
36       [
37           "charset_name",
38           "language",
39           "char_to_order_map",
40           "language_model",
41           "typical_positive_ratio",
42           "keep_ascii_letters",
43           "alphabet",
44       ],
45   )
46   
47   
48   class SingleByteCharSetProber(CharSetProber):
49       SAMPLE_SIZE = 64
50       SB_ENOUGH_REL_THRESHOLD = 1024  # 0.25 * SAMPLE_SIZE^2
51       POSITIVE_SHORTCUT_THRESHOLD = 0.95
52       NEGATIVE_SHORTCUT_THRESHOLD = 0.05
53   
54       def __init__(self, model, is_reversed=False, name_prober=None):
55           super().__init__()
56           self._model = model
57           # TRUE if we need to reverse every pair in the model lookup
58           self._reversed = is_reversed
59           # Optional auxiliary prober for name decision
60           self._name_prober = name_prober
61           self._last_order = None
62           self._seq_counters = None
63           self._total_seqs = None
64           self._total_char = None
65           self._control_char = None
66           self._freq_char = None
67           self.reset()
68   
69       def reset(self):
70           super().reset()
71           # char order of last character
72           self._last_order = 255
73           self._seq_counters = [0] * SequenceLikelihood.get_num_categories()
74           self._total_seqs = 0
75           self._total_char = 0
76           self._control_char = 0
77           # characters that fall in our sampling range
78           self._freq_char = 0
79   
80       @property
81       def charset_name(self):
82           if self._name_prober:
83               return self._name_prober.charset_name
84           return self._model.charset_name
85   
86       @property
87       def language(self):
88           if self._name_prober:
89               return self._name_prober.language
90           return self._model.language
91   
92       def feed(self, byte_str):
93           # TODO: Make filter_international_words keep things in self.alphabet
94           if not self._model.keep_ascii_letters:
95               byte_str = self.filter_international_words(byte_str)
96           else:
97               byte_str = self.remove_xml_tags(byte_str)
98           if not byte_str:
99               return self.state
100          char_to_order_map = self._model.char_to_order_map
101          language_model = self._model.language_model
102          for char in byte_str:
103              order = char_to_order_map.get(char, CharacterCategory.UNDEFINED)
104              # XXX: This was SYMBOL_CAT_ORDER before, with a value of 250, but
105              #      CharacterCategory.SYMBOL is actually 253, so we use CONTROL
106              #      to make it closer to the original intent. The only difference
107              #      is whether or not we count digits and control characters for
108              #      _total_char purposes.
109              if order < CharacterCategory.CONTROL:
110                  self._total_char += 1
111              if order < self.SAMPLE_SIZE:
112                  self._freq_char += 1
113                  if self._last_order < self.SAMPLE_SIZE:
114                      self._total_seqs += 1
115                      if not self._reversed:
116                          lm_cat = language_model[self._last_order][order]
117                      else:
118                          lm_cat = language_model[order][self._last_order]
119                      self._seq_counters[lm_cat] += 1
120              self._last_order = order
121  
122          charset_name = self._model.charset_name
123          if self.state == ProbingState.DETECTING:
124              if self._total_seqs > self.SB_ENOUGH_REL_THRESHOLD:
125                  confidence = self.get_confidence()
126                  if confidence > self.POSITIVE_SHORTCUT_THRESHOLD:
127                      self.logger.debug(
128                          "%s confidence = %s, we have a winner", charset_name, confidence
129                      )
130                      self._state = ProbingState.FOUND_IT
131                  elif confidence < self.NEGATIVE_SHORTCUT_THRESHOLD:
132                      self.logger.debug(
133                          "%s confidence = %s, below negative shortcut threshold %s",
134                          charset_name,
135                          confidence,
136                          self.NEGATIVE_SHORTCUT_THRESHOLD,
137                      )
138                      self._state = ProbingState.NOT_ME
139  
140          return self.state
141  
142      def get_confidence(self):
143          r = 0.01
144          if self._total_seqs > 0:
145              r = (
146                  (
147                      self._seq_counters[SequenceLikelihood.POSITIVE]
148                      + 0.25 * self._seq_counters[SequenceLikelihood.LIKELY]
149                  )
150                  / self._total_seqs
151                  / self._model.typical_positive_ratio
152              )
153              # The more control characters (proportionnaly to the size
154              # of the text), the less confident we become in the current
155              # charset.
156              r = r * (self._total_char - self._control_char) / self._total_char
157              r = r * self._freq_char / self._total_char
158              if r >= 1.0:
159                  r = 0.99
160          return r
