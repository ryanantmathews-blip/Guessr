40   _SUPPORTED_HASHES = ("sha512", "sha384", "sha256", "sha224", "sha1", "md5")

43   @dataclass(frozen=True)
44   class LinkHash:
45       """Links to content may have embedded hash values. This class parses those. 
46    
47       `name` must be any member of `_SUPPORTED_HASHES`. 
48    
49       This class can be converted to and from `ArchiveInfo`. While ArchiveInfo intends to 
50       be JSON-serializable to conform to PEP 610, this class contains the logic for 
51       parsing a hash name and value for correctness, and then checking whether that hash 
52       conforms to a schema with `.is_hash_allowed()`."""
53   
54       name: str
55       value: str
56   
57       _hash_re = re.compile(
58           # NB: we do not validate that the second group (.*) is a valid hex
59           # digest. Instead, we simply keep that string in this class, and then check it
60           # against Hashes when hash-checking is needed. This is easier to debug than
61           # proactively discarding an invalid hex digest, as we handle incorrect hashes
62           # and malformed hashes in the same place.
63           r"({choices})=(.*)".format(
64               choices="|".join(re.escape(hash_name) for hash_name in _SUPPORTED_HASHES)
65           ),
66       )
67   
68       def __post_init__(self) -> None:
69           assert self._hash_re.match(f"{self.name}={self.value}")
70   
71       @classmethod
72       @functools.lru_cache(maxsize=None)
73       def split_hash_name_and_value(cls, url: str) -> Optional["LinkHash"]:
74           """Search a string for a checksum algorithm name and encoded output value."""
75           match = cls._hash_re.search(url)
76           if match is None:
77               return None
78           name, value = match.groups()
79           return cls(name=name, value=value)
80   
81       def as_hashes(self) -> Hashes:
82           """Return a Hashes instance which checks only for the current hash."""
83           return Hashes({self.name: [self.value]})
84   
85       def is_hash_allowed(self, hashes: Optional[Hashes]) -> bool:
86           """ 
87           Return True if the current hash is allowed by `hashes`. 
88           """
89           if hashes is None:
90               return False
91           return hashes.is_hash_allowed(self.name, hex_digest=self.value)

94   def _clean_url_path_part(part: str) -> str:
95       """ 
96       Clean a "part" of a URL path (i.e. after splitting on "@" characters). 
97       """
98       # We unquote prior to quoting to make sure nothing is double quoted.
99       return urllib.parse.quote(urllib.parse.unquote(part))

102  def _clean_file_url_path(part: str) -> str:
103      """ 
104      Clean the first part of a URL path that corresponds to a local 
105      filesystem path (i.e. the first part after splitting on "@" characters). 
106      """
107      # We unquote prior to quoting to make sure nothing is double quoted.
108      # Also, on Windows the path part might contain a drive letter which
109      # should not be quoted. On Linux where drive letters do not
110      # exist, the colon should be quoted. We rely on urllib.request
111      # to do the right thing here.
112      return urllib.request.pathname2url(urllib.request.url2pathname(part))
113  
114  
115  # percent-encoded:                   /
116  _reserved_chars_re = re.compile("(@|%2F)", re.IGNORECASE)
117  
118  
119  def _clean_url_path(path: str, is_local_path: bool) -> str:
120      """ 
121      Clean the path portion of a URL. 
122      """
123      if is_local_path:
124          clean_func = _clean_file_url_path
125      else:
126          clean_func = _clean_url_path_part
127  
128      # Split on the reserved characters prior to cleaning so that
129      # revision strings in VCS URLs are properly preserved.
130      parts = _reserved_chars_re.split(path)
131  
132      cleaned_parts = []
133      for to_clean, reserved in pairwise(itertools.chain(parts, [""])):
134          cleaned_parts.append(clean_func(to_clean))
135          # Normalize %xx escapes (e.g. %2f -> %2F)
136          cleaned_parts.append(reserved.upper())
137  
138      return "".join(cleaned_parts)

141  def _ensure_quoted_url(url: str) -> str:
142      """ 
143      Make sure a link is fully quoted. 
144      For example, if ' ' occurs in the URL, it will be replaced with "%20", 
145      and without double-quoting other characters. 
146      """
147      # Split the URL into parts according to the general structure
148      # `scheme://netloc/path;parameters?query#fragment`.
149      result = urllib.parse.urlparse(url)
150      # If the netloc is empty, then the URL refers to a local filesystem path.
151      is_local_path = not result.netloc
152      path = _clean_url_path(result.path, is_local_path=is_local_path)
153      return urllib.parse.urlunparse(result._replace(path=path))

156  class Link(KeyBasedCompareMixin):
157      """Represents a parsed link from a Package Index's simple URL"""
158  
159      __slots__ = [
160          "_parsed_url",
161          "_url",
162          "_hashes",
163          "comes_from",
164          "requires_python",
165          "yanked_reason",
166          "dist_info_metadata",
167          "link_hash",
168          "cache_link_parsing",
169      ]
170  
171      def __init__(
172          self,
173          url: str,
174          comes_from: Optional[Union[str, "IndexContent"]] = None,
175          requires_python: Optional[str] = None,
176          yanked_reason: Optional[str] = None,
177          dist_info_metadata: Optional[str] = None,
178          link_hash: Optional[LinkHash] = None,
179          cache_link_parsing: bool = True,
180          hashes: Optional[Mapping[str, str]] = None,
181      ) -> None:
182          """ 
183          :param url: url of the resource pointed to (href of the link) 
184          :param comes_from: instance of IndexContent where the link was found, 
185              or string. 
186          :param requires_python: String containing the `Requires-Python` 
187              metadata field, specified in PEP 345. This may be specified by 
188              a data-requires-python attribute in the HTML link tag, as 
189              described in PEP 503. 
190          :param yanked_reason: the reason the file has been yanked, if the 
191              file has been yanked, or None if the file hasn't been yanked. 
192              This is the value of the "data-yanked" attribute, if present, in 
193              a simple repository HTML link. If the file has been yanked but 
194              no reason was provided, this should be the empty string. See 
195              PEP 592 for more information and the specification. 
196          :param dist_info_metadata: the metadata attached to the file, or None if no such 
197              metadata is provided. This is the value of the "data-dist-info-metadata" 
198              attribute, if present, in a simple repository HTML link. This may be parsed 
199              into its own `Link` by `self.metadata_link()`. See PEP 658 for more 
200              information and the specification. 
201          :param link_hash: a checksum for the content the link points to. If not 
202              provided, this will be extracted from the link URL, if the URL has 
203              any checksum. 
204          :param cache_link_parsing: A flag that is used elsewhere to determine 
205                                     whether resources retrieved from this link 
206                                     should be cached. PyPI index urls should 
207                                     generally have this set to False, for 
208                                     example. 
209          :param hashes: A mapping of hash names to digests to allow us to 
210                         determine the validity of a download. 
211          """
212  
213          # url can be a UNC windows share
214          if url.startswith("\\\\"):
215              url = path_to_url(url)
216  
217          self._parsed_url = urllib.parse.urlsplit(url)
218          # Store the url as a private attribute to prevent accidentally
219          # trying to set a new value.
220          self._url = url
221          self._hashes = hashes if hashes is not None else {}
222  
223          self.comes_from = comes_from
224          self.requires_python = requires_python if requires_python else None
225          self.yanked_reason = yanked_reason
226          self.dist_info_metadata = dist_info_metadata
227          self.link_hash = link_hash or LinkHash.split_hash_name_and_value(self._url)
228  
229          super().__init__(key=url, defining_class=Link)
230  
231          self.cache_link_parsing = cache_link_parsing

233      @classmethod
234      def from_json(
235          cls,
236          file_data: Dict[str, Any],
237          page_url: str,
238      ) -> Optional["Link"]:
239          """ 
240          Convert an pypi json document from a simple repository page into a Link. 
241          """
242          file_url = file_data.get("url")
243          if file_url is None:
244              return None
245  
246          url = _ensure_quoted_url(urllib.parse.urljoin(page_url, file_url))
247          pyrequire = file_data.get("requires-python")
248          yanked_reason = file_data.get("yanked")
249          dist_info_metadata = file_data.get("dist-info-metadata")
250          hashes = file_data.get("hashes", {})
251  
252          # The Link.yanked_reason expects an empty string instead of a boolean.
253          if yanked_reason and not isinstance(yanked_reason, str):
254              yanked_reason = ""
255          # The Link.yanked_reason expects None instead of False.
256          elif not yanked_reason:
257              yanked_reason = None
258  
259          return cls(
260              url,
261              comes_from=page_url,
262              requires_python=pyrequire,
263              yanked_reason=yanked_reason,
264              hashes=hashes,
265              dist_info_metadata=dist_info_metadata,
266          )

268      @classmethod
269      def from_element(
270          cls,
271          anchor_attribs: Dict[str, Optional[str]],
272          page_url: str,
273          base_url: str,
274      ) -> Optional["Link"]:
275          """ 
276          Convert an anchor element's attributes in a simple repository page to a Link. 
277          """
278          href = anchor_attribs.get("href")
279          if not href:
280              return None
281  
282          url = _ensure_quoted_url(urllib.parse.urljoin(base_url, href))
283          pyrequire = anchor_attribs.get("data-requires-python")
284          yanked_reason = anchor_attribs.get("data-yanked")
285          dist_info_metadata = anchor_attribs.get("data-dist-info-metadata")
286  
287          return cls(
288              url,
289              comes_from=page_url,
290              requires_python=pyrequire,
291              yanked_reason=yanked_reason,
292              dist_info_metadata=dist_info_metadata,
293          )

295      def __str__(self) -> str:
296          if self.requires_python:
297              rp = f" (requires-python:{self.requires_python})"
298          else:
299              rp = ""
300          if self.comes_from:
301              return "{} (from {}){}".format(
302                  redact_auth_from_url(self._url), self.comes_from, rp
303              )
304          else:
305              return redact_auth_from_url(str(self._url))

377      def metadata_link(self) -> Optional["Link"]:
378          """Implementation of PEP 658 parsing."""
379          # Note that Link.from_element() parsing the "data-dist-info-metadata" attribute
380          # from an HTML anchor tag is typically how the Link.dist_info_metadata attribute
381          # gets set.
382          if self.dist_info_metadata is None:
383              return None
384          metadata_url = f"{self.url_without_fragment}.metadata"
385          link_hash: Optional[LinkHash] = None
386          # If data-dist-info-metadata="true" is set, then the metadata file exists,
387          # but there is no information about its checksum or anything else.
388          if self.dist_info_metadata != "true":
389              link_hash = LinkHash.split_hash_name_and_value(self.dist_info_metadata)
390          return Link(metadata_url, link_hash=link_hash)

447  class _CleanResult(NamedTuple):
448      """Convert link for equivalency check. 
449   
450      This is used in the resolver to check whether two URL-specified requirements 
451      likely point to the same distribution and can be considered equivalent. This 
452      equivalency logic avoids comparing URLs literally, which can be too strict 
453      (e.g. "a=1&b=2" vs "b=2&a=1") and produce conflicts unexpecting to users. 
454   
455      Currently this does three things: 
456   
457      1. Drop the basic auth part. This is technically wrong since a server can 
458         serve different content based on auth, but if it does that, it is even 
459         impossible to guarantee two URLs without auth are equivalent, since 
460         the user can input different auth information when prompted. So the 
461         practical solution is to assume the auth doesn't affect the response. 
462      2. Parse the query to avoid the ordering issue. Note that ordering under the 
463         same key in the query are NOT cleaned; i.e. "a=1&a=2" and "a=2&a=1" are 
464         still considered different. 
465      3. Explicitly drop most of the fragment part, except ``subdirectory=`` and 
466         hash values, since it should have no impact the downloaded content. Note 
467         that this drops the "egg=" part historically used to denote the requested 
468         project (and extras), which is wrong in the strictest sense, but too many 
469         people are supplying it inconsistently to cause superfluous resolution 
470         conflicts, so we choose to also ignore them. 
471      """
472  
473      parsed: urllib.parse.SplitResult
474      query: Dict[str, List[str]]
475      subdirectory: str
476      hashes: Dict[str, str]
477  
478  
479  def _clean_link(link: Link) -> _CleanResult:
480      parsed = link._parsed_url
481      netloc = parsed.netloc.rsplit("@", 1)[-1]
482      # According to RFC 8089, an empty host in file: means localhost.
483      if parsed.scheme == "file" and not netloc:
484          netloc = "localhost"
485      fragment = urllib.parse.parse_qs(parsed.fragment)
486      if "egg" in fragment:
487          logger.debug("Ignoring egg= fragment in %s", link)
488      try:
489          # If there are multiple subdirectory values, use the first one.
490          # This matches the behavior of Link.subdirectory_fragment.
491          subdirectory = fragment["subdirectory"][0]
492      except (IndexError, KeyError):
493          subdirectory = ""
494      # If there are multiple hash values under the same algorithm, use the
495      # first one. This matches the behavior of Link.hash_value.
496      hashes = {k: fragment[k][0] for k in _SUPPORTED_HASHES if k in fragment}
497      return _CleanResult(
498          parsed=parsed._replace(netloc=netloc, query="", fragment=""),
499          query=urllib.parse.parse_qs(parsed.query),
500          subdirectory=subdirectory,
501          hashes=hashes,
502      )

505  @functools.lru_cache(maxsize=None)
506  def links_equivalent(link1: Link, link2: Link) -> bool:
507      return _clean_link(link1) == _clean_link(link2)
