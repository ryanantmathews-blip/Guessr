47   def get_platform(archive_root):
48       """Return our platform name 'win32', 'linux_x86_64'"""
49       result = sysconfig.get_platform()
50       if result.startswith("macosx") and archive_root is not None:
51           result = calculate_macosx_platform_tag(archive_root, result)
52       elif result == "linux-x86_64" and sys.maxsize == 2147483647:
53           # pip pull request #3497
54           result = "linux-i686"
55   
56       return result.replace("-", "_")

59   def get_flag(var, fallback, expected=True, warn=True):
60       """Use a fallback value for determining SOABI flags if the needed config 
61       var is unset or unavailable."""
62       val = sysconfig.get_config_var(var)
63       if val is None:
64           if warn:
65               warnings.warn(
66                   "Config variable '{}' is unset, Python ABI tag may "
67                   "be incorrect".format(var),
68                   RuntimeWarning,
69                   2,
70               )
71           return fallback
72       return val == expected

75   def get_abi_tag():
76       """Return the ABI tag based on SOABI (if available) or emulate SOABI (PyPy2)."""
77       soabi = sysconfig.get_config_var("SOABI")
78       impl = tags.interpreter_name()
79       if not soabi and impl in ("cp", "pp") and hasattr(sys, "maxunicode"):
80           d = ""
81           m = ""
82           u = ""
83           if get_flag("Py_DEBUG", hasattr(sys, "gettotalrefcount"), warn=(impl == "cp")):
84               d = "d"
85   
86           if get_flag(
87               "WITH_PYMALLOC",
88               impl == "cp",
89               warn=(impl == "cp" and sys.version_info < (3, 8)),
90           ) and sys.version_info < (3, 8):
91               m = "m"
92   
93           abi = f"{impl}{tags.interpreter_version()}{d}{m}{u}"
94       elif soabi and impl == "cp":
95           abi = "cp" + soabi.split("-")[1]
96       elif soabi and impl == "pp":
97           # we want something like pypy36-pp73
98           abi = "-".join(soabi.split("-")[:2])
99           abi = abi.replace(".", "_").replace("-", "_")
100      elif soabi:
101          abi = soabi.replace(".", "_").replace("-", "_")
102      else:
103          abi = None
104  
105      return abi

190      def initialize_options(self):
191          self.bdist_dir = None
192          self.data_dir = None
193          self.plat_name = None
194          self.plat_tag = None
195          self.format = "zip"
196          self.keep_temp = False
197          self.dist_dir = None
198          self.egginfo_dir = None
199          self.root_is_pure = None
200          self.skip_build = None
201          self.relative = False
202          self.owner = None
203          self.group = None
204          self.universal = False
205          self.compression = "deflated"
206          self.python_tag = python_tag()
207          self.build_number = None
208          self.py_limited_api = False
209          self.plat_name_supplied = False

211      def finalize_options(self):
212          if self.bdist_dir is None:
213              bdist_base = self.get_finalized_command("bdist").bdist_base
214              self.bdist_dir = os.path.join(bdist_base, "wheel")
215  
216          self.data_dir = self.wheel_dist_name + ".data"
217          self.plat_name_supplied = self.plat_name is not None
218  
219          try:
220              self.compression = self.supported_compressions[self.compression]
221          except KeyError:
222              raise ValueError(f"Unsupported compression: {self.compression}")
223  
224          need_options = ("dist_dir", "plat_name", "skip_build")
225  
226          self.set_undefined_options("bdist", *zip(need_options, need_options))
227  
228          self.root_is_pure = not (
229              self.distribution.has_ext_modules() or self.distribution.has_c_libraries()
230          )
231  
232          if self.py_limited_api and not re.match(
233              PY_LIMITED_API_PATTERN, self.py_limited_api
234          ):
235              raise ValueError("py-limited-api must match '%s'" % PY_LIMITED_API_PATTERN)
236  
237          # Support legacy [wheel] section for setting universal
238          wheel = self.distribution.get_option_dict("wheel")
239          if "universal" in wheel:
240              # please don't define this in your global configs
241              log.warning(
242                  "The [wheel] section is deprecated. Use [bdist_wheel] instead.",
243              )
244              val = wheel["universal"][1].strip()
245              if val.lower() in ("1", "true", "yes"):
246                  self.universal = True
247  
248          if self.build_number is not None and not self.build_number[:1].isdigit():
249              raise ValueError("Build tag (build-number) must start with a digit.")

251      @property
252      def wheel_dist_name(self):
253          """Return distribution full name with - replaced with _"""
254          components = (
255              safer_name(self.distribution.get_name()),
256              safer_version(self.distribution.get_version()),
257          )
258          if self.build_number:
259              components += (self.build_number,)
260          return "-".join(components)

262      def get_tag(self):
263          # bdist sets self.plat_name if unset, we should only use it for purepy
264          # wheels if the user supplied it.
265          if self.plat_name_supplied:
266              plat_name = self.plat_name
267          elif self.root_is_pure:
268              plat_name = "any"
269          else:
270              # macosx contains system version in platform name so need special handle
271              if self.plat_name and not self.plat_name.startswith("macosx"):
272                  plat_name = self.plat_name
273              else:
274                  # on macosx always limit the platform name to comply with any
275                  # c-extension modules in bdist_dir, since the user can specify
276                  # a higher MACOSX_DEPLOYMENT_TARGET via tools like CMake
277  
278                  # on other platforms, and on macosx if there are no c-extension
279                  # modules, use the default platform name.
280                  plat_name = get_platform(self.bdist_dir)
281  
282              if (
283                  plat_name in ("linux-x86_64", "linux_x86_64")
284                  and sys.maxsize == 2147483647
285              ):
286                  plat_name = "linux_i686"
287  
288          plat_name = plat_name.lower().replace("-", "_").replace(".", "_")
289  
290          if self.root_is_pure:
291              if self.universal:
292                  impl = "py2.py3"
293              else:
294                  impl = self.python_tag
295              tag = (impl, "none", plat_name)
296          else:
297              impl_name = tags.interpreter_name()
298              impl_ver = tags.interpreter_version()
299              impl = impl_name + impl_ver
300              # We don't work on CPython 3.1, 3.0.
301              if self.py_limited_api and (impl_name + impl_ver).startswith("cp3"):
302                  impl = self.py_limited_api
303                  abi_tag = "abi3"
304              else:
305                  abi_tag = str(get_abi_tag()).lower()
306              tag = (impl, abi_tag, plat_name)
307              # issue gh-374: allow overriding plat_name
308              supported_tags = [
309                  (t.interpreter, t.abi, plat_name) for t in tags.sys_tags()
310              ]
311              assert (
312                  tag in supported_tags
313              ), f"would build wheel with unsupported tag {tag}"
314          return tag

316      def run(self):
317          build_scripts = self.reinitialize_command("build_scripts")
318          build_scripts.executable = "python"
319          build_scripts.force = True
320  
321          build_ext = self.reinitialize_command("build_ext")
322          build_ext.inplace = False
323  
324          if not self.skip_build:
325              self.run_command("build")
326  
327          install = self.reinitialize_command("install", reinit_subcommands=True)
328          install.root = self.bdist_dir
329          install.compile = False
330          install.skip_build = self.skip_build
331          install.warn_dir = False
332  
333          # A wheel without setuptools scripts is more cross-platform.
334          # Use the (undocumented) `no_ep` option to setuptools'
335          # install_scripts command to avoid creating entry point scripts.
336          install_scripts = self.reinitialize_command("install_scripts")
337          install_scripts.no_ep = True
338  
339          # Use a custom scheme for the archive, because we have to decide
340          # at installation time which scheme to use.
341          for key in ("headers", "scripts", "data", "purelib", "platlib"):
342              setattr(install, "install_" + key, os.path.join(self.data_dir, key))
343  
344          basedir_observed = ""
345  
346          if os.name == "nt":
347              # win32 barfs if any of these are ''; could be '.'?
348              # (distutils.command.install:change_roots bug)
349              basedir_observed = os.path.normpath(os.path.join(self.data_dir, ".."))
350              self.install_libbase = self.install_lib = basedir_observed
351  
352          setattr(
353              install,
354              "install_purelib" if self.root_is_pure else "install_platlib",
355              basedir_observed,
356          )
357  
358          log.info(f"installing to {self.bdist_dir}")
359  
360          self.run_command("install")
361  
362          impl_tag, abi_tag, plat_tag = self.get_tag()
363          archive_basename = f"{self.wheel_dist_name}-{impl_tag}-{abi_tag}-{plat_tag}"
364          if not self.relative:
365              archive_root = self.bdist_dir
366          else:
367              archive_root = os.path.join(
368                  self.bdist_dir, self._ensure_relative(install.install_base)
369              )
370  
371          self.set_undefined_options("install_egg_info", ("target", "egginfo_dir"))
372          distinfo_dirname = "{}-{}.dist-info".format(
373              safer_name(self.distribution.get_name()),
374              safer_version(self.distribution.get_version()),
375          )
376          distinfo_dir = os.path.join(self.bdist_dir, distinfo_dirname)
377          self.egg2dist(self.egginfo_dir, distinfo_dir)
378  
379          self.write_wheelfile(distinfo_dir)
380  
381          # Make the archive
382          if not os.path.exists(self.dist_dir):
383              os.makedirs(self.dist_dir)
384  
385          wheel_path = os.path.join(self.dist_dir, archive_basename + ".whl")
386          with WheelFile(wheel_path, "w", self.compression) as wf:
387              wf.write_files(archive_root)
388  
389          # Add to 'Distribution.dist_files' so that the "upload" command works
390          getattr(self.distribution, "dist_files", []).append(
391              (
392                  "bdist_wheel",
393                  "{}.{}".format(*sys.version_info[:2]),  # like 3.7
394                  wheel_path,
395              )
396          )
397  
398          if not self.keep_temp:
399              log.info(f"removing {self.bdist_dir}")
400              if not self.dry_run:
401                  rmtree(self.bdist_dir, onerror=remove_readonly)

403      def write_wheelfile(
404          self, wheelfile_base, generator="bdist_wheel (" + wheel_version + ")"
405      ):
406          from email.message import Message
407  
408          msg = Message()
409          msg["Wheel-Version"] = "1.0"  # of the spec
410          msg["Generator"] = generator
411          msg["Root-Is-Purelib"] = str(self.root_is_pure).lower()
412          if self.build_number is not None:
413              msg["Build"] = self.build_number
414  
415          # Doesn't work for bdist_wininst
416          impl_tag, abi_tag, plat_tag = self.get_tag()
417          for impl in impl_tag.split("."):
418              for abi in abi_tag.split("."):
419                  for plat in plat_tag.split("."):
420                      msg["Tag"] = "-".join((impl, abi, plat))
421  
422          wheelfile_path = os.path.join(wheelfile_base, "WHEEL")
423          log.info(f"creating {wheelfile_path}")
424          buffer = BytesIO()
425          BytesGenerator(buffer, maxheaderlen=0).flatten(msg)
426          with open(wheelfile_path, "wb") as f:
427              f.write(buffer.getvalue().replace(b"\r\n", b"\r"))

429      def _ensure_relative(self, path):
430          # copied from dir_util, deleted
431          drive, path = os.path.splitdrive(path)
432          if path[0:1] == os.sep:
433              path = drive + path[1:]
434          return path

436      @property
437      def license_paths(self):
438          if setuptools_major_version >= 57:
439              # Setuptools has resolved any patterns to actual file names
440              return self.distribution.metadata.license_files or ()
441  
442          files = set()
443          metadata = self.distribution.get_option_dict("metadata")
444          if setuptools_major_version >= 42:
445              # Setuptools recognizes the license_files option but does not do globbing
446              patterns = self.distribution.metadata.license_files
447          else:
448              # Prior to those, wheel is entirely responsible for handling license files
449              if "license_files" in metadata:
450                  patterns = metadata["license_files"][1].split()
451              else:
452                  patterns = ()
453  
454          if "license_file" in metadata:
455              warnings.warn(
456                  'The "license_file" option is deprecated. Use "license_files" instead.',
457                  DeprecationWarning,
458              )
459              files.add(metadata["license_file"][1])
460  
461          if not files and not patterns and not isinstance(patterns, list):
462              patterns = ("LICEN[CS]E*", "COPYING*", "NOTICE*", "AUTHORS*")
463  
464          for pattern in patterns:
465              for path in iglob(pattern):
466                  if path.endswith("~"):
467                      log.debug(
468                          f'ignoring license file "{path}" as it looks like a backup'
469                      )
470                      continue
471  
472                  if path not in files and os.path.isfile(path):
473                      log.info(
474                          f'adding license file "{path}" (matched pattern "{pattern}")'
475                      )
476                      files.add(path)
477  
478          return files

480      def egg2dist(self, egginfo_path, distinfo_path):
481          """Convert an .egg-info directory into a .dist-info directory"""
482  
483          def adios(p):
484              """Appropriately delete directory, file or link."""
485              if os.path.exists(p) and not os.path.islink(p) and os.path.isdir(p):
486                  shutil.rmtree(p)
487              elif os.path.exists(p):
488                  os.unlink(p)
489  
490          adios(distinfo_path)
491  
492          if not os.path.exists(egginfo_path):
493              # There is no egg-info. This is probably because the egg-info
494              # file/directory is not named matching the distribution name used
495              # to name the archive file. Check for this case and report
496              # accordingly.
497              import glob
498  
499              pat = os.path.join(os.path.dirname(egginfo_path), "*.egg-info")
500              possible = glob.glob(pat)
501              err = f"Egg metadata expected at {egginfo_path} but not found"
502              if possible:
503                  alt = os.path.basename(possible[0])
504                  err += f" ({alt} found - possible misnamed archive file?)"
505  
506              raise ValueError(err)
507  
508          if os.path.isfile(egginfo_path):
509              # .egg-info is a single file
510              pkginfo_path = egginfo_path
511              pkg_info = pkginfo_to_metadata(egginfo_path, egginfo_path)
512              os.mkdir(distinfo_path)
513          else:
514              # .egg-info is a directory
515              pkginfo_path = os.path.join(egginfo_path, "PKG-INFO")
516              pkg_info = pkginfo_to_metadata(egginfo_path, pkginfo_path)
517  
518              # ignore common egg metadata that is useless to wheel
519              shutil.copytree(
520                  egginfo_path,
521                  distinfo_path,
522                  ignore=lambda x, y: {
523                      "PKG-INFO",
524                      "requires.txt",
525                      "SOURCES.txt",
526                      "not-zip-safe",
527                  },
528              )
529  
530              # delete dependency_links if it is only whitespace
531              dependency_links_path = os.path.join(distinfo_path, "dependency_links.txt")
532              with open(dependency_links_path) as dependency_links_file:
533                  dependency_links = dependency_links_file.read().strip()
534              if not dependency_links:
535                  adios(dependency_links_path)
536  
537          pkg_info_path = os.path.join(distinfo_path, "METADATA")
538          serialization_policy = EmailPolicy(
539              utf8=True,
540              mangle_from_=False,
541              max_line_length=0,
542          )
543          with open(pkg_info_path, "w", encoding="utf-8") as out:
544              Generator(out, policy=serialization_policy).flatten(pkg_info)
545  
546          for license_path in self.license_paths:
547              filename = os.path.basename(license_path)
548              shutil.copy(license_path, os.path.join(distinfo_path, filename))
549  
550          adios(egginfo_path)
551
