98   class Latin1Prober(CharSetProber):
99       def __init__(self):
100          super().__init__()
101          self._last_char_class = None
102          self._freq_counter = None
103          self.reset()
104  
105      def reset(self):
106          self._last_char_class = OTH
107          self._freq_counter = [0] * FREQ_CAT_NUM
108          super().reset()
109  
110      @property
111      def charset_name(self):
112          return "ISO-8859-1"
113  
114      @property
115      def language(self):
116          return ""
117  
118      def feed(self, byte_str):
119          byte_str = self.remove_xml_tags(byte_str)
120          for c in byte_str:
121              char_class = Latin1_CharToClass[c]
122              freq = Latin1ClassModel[(self._last_char_class * CLASS_NUM) + char_class]
123              if freq == 0:
124                  self._state = ProbingState.NOT_ME
125                  break
126              self._freq_counter[freq] += 1
127              self._last_char_class = char_class
128  
129          return self.state
130  
131      def get_confidence(self):
132          if self.state == ProbingState.NOT_ME:
133              return 0.01
134  
135          total = sum(self._freq_counter)
136          confidence = (
137              0.0
138              if total < 0.01
139              else (self._freq_counter[3] - self._freq_counter[1] * 20.0) / total
140          )
141          confidence = max(confidence, 0.0)
142          # lower the confidence of latin1 so that other more accurate
143          # detector can take priority.
144          confidence *= 0.73
145          return confidence
