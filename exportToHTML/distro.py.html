16   """ 
17   The ``distro`` package (``distro`` stands for Linux Distribution) provides 
18   information about the Linux distribution it runs on, such as a reliable 
19   machine-readable distro ID, or version information. 
20    
21   It is the recommended replacement for Python's original 
22   :py:func:`platform.linux_distribution` function, but it provides much more 
23   functionality. An alternative implementation became necessary because Python 
24   3.5 deprecated this function, and Python 3.8 removed it altogether. Its 
25   predecessor function :py:func:`platform.dist` was already deprecated since 
26   Python 2.6 and removed in Python 3.8. Still, there are many cases in which 
27   access to OS distribution information is needed. See `Python issue 1322 
28   <https://bugs.python.org/issue1322>`_ for more information. 
29   """

137  def linux_distribution(full_distribution_name: bool = True) -> Tuple[str, str, str]:
138      """ 
139      .. deprecated:: 1.6.0 
140   
141          :func:`distro.linux_distribution()` is deprecated. It should only be 
142          used as a compatibility shim with Python's 
143          :py:func:`platform.linux_distribution()`. Please use :func:`distro.id`, 
144          :func:`distro.version` and :func:`distro.name` instead. 
145   
146      Return information about the current OS distribution as a tuple 
147      ``(id_name, version, codename)`` with items as follows: 
148   
149      * ``id_name``:  If *full_distribution_name* is false, the result of 
150        :func:`distro.id`. Otherwise, the result of :func:`distro.name`. 
151   
152      * ``version``:  The result of :func:`distro.version`. 
153   
154      * ``codename``:  The extra item (usually in parentheses) after the 
155        os-release version number, or the result of :func:`distro.codename`. 
156   
157      The interface of this function is compatible with the original 
158      :py:func:`platform.linux_distribution` function, supporting a subset of 
159      its parameters. 
160   
161      The data it returns may not exactly be the same, because it uses more data 
162      sources than the original function, and that may lead to different data if 
163      the OS distribution is not consistent across multiple data sources it 
164      provides (there are indeed such distributions ...). 
165   
166      Another reason for differences is the fact that the :func:`distro.id` 
167      method normalizes the distro ID string to a reliable machine-readable value 
168      for a number of popular OS distributions. 
169      """
170      warnings.warn(
171          "distro.linux_distribution() is deprecated. It should only be used as a "
172          "compatibility shim with Python's platform.linux_distribution(). Please use "
173          "distro.id(), distro.version() and distro.name() instead.",
174          DeprecationWarning,
175          stacklevel=2,
176      )
177      return _distro.linux_distribution(full_distribution_name)

180  def id() -> str:
181      """ 
182      Return the distro ID of the current distribution, as a 
183      machine-readable string. 
184   
185      For a number of OS distributions, the returned distro ID value is 
186      *reliable*, in the sense that it is documented and that it does not change 
187      across releases of the distribution. 
188   
189      This package maintains the following reliable distro ID values: 
190   
191      ==============  ========================================= 
192      Distro ID       Distribution 
193      ==============  ========================================= 
194      "ubuntu"        Ubuntu 
195      "debian"        Debian 
196      "rhel"          RedHat Enterprise Linux 
197      "centos"        CentOS 
198      "fedora"        Fedora 
199      "sles"          SUSE Linux Enterprise Server 
200      "opensuse"      openSUSE 
201      "amzn"          Amazon Linux 
202      "arch"          Arch Linux 
203      "cloudlinux"    CloudLinux OS 
204      "exherbo"       Exherbo Linux 
205      "gentoo"        GenToo Linux 
206      "ibm_powerkvm"  IBM PowerKVM 
207      "kvmibm"        KVM for IBM z Systems 
208      "linuxmint"     Linux Mint 
209      "mageia"        Mageia 
210      "mandriva"      Mandriva Linux 
211      "parallels"     Parallels 
212      "pidora"        Pidora 
213      "raspbian"      Raspbian 
214      "oracle"        Oracle Linux (and Oracle Enterprise Linux) 
215      "scientific"    Scientific Linux 
216      "slackware"     Slackware 
217      "xenserver"     XenServer 
218      "openbsd"       OpenBSD 
219      "netbsd"        NetBSD 
220      "freebsd"       FreeBSD 
221      "midnightbsd"   MidnightBSD 
222      "rocky"         Rocky Linux 
223      "aix"           AIX 
224      ==============  ========================================= 
225   
226      If you have a need to get distros for reliable IDs added into this set, 
227      or if you find that the :func:`distro.id` function returns a different 
228      distro ID for one of the listed distros, please create an issue in the 
229      `distro issue tracker`_. 
230   
231      **Lookup hierarchy and transformations:** 
232   
233      First, the ID is obtained from the following sources, in the specified 
234      order. The first available and non-empty value is used: 
235   
236      * the value of the "ID" attribute of the os-release file, 
237   
238      * the value of the "Distributor ID" attribute returned by the lsb_release 
239        command, 
240   
241      * the first part of the file name of the distro release file, 
242   
243      The so determined ID value then passes the following transformations, 
244      before it is returned by this method: 
245   
246      * it is translated to lower case, 
247   
248      * blanks (which should not be there anyway) are translated to underscores, 
249   
250      * a normalization of the ID is performed, based upon 
251        `normalization tables`_. The purpose of this normalization is to ensure 
252        that the ID is as reliable as possible, even across incompatible changes 
253        in the OS distributions. A common reason for an incompatible change is 
254        the addition of an os-release file, or the addition of the lsb_release 
255        command, with ID values that differ from what was previously determined 
256        from the distro release file name. 
257      """
258      return _distro.id()

261  def name(pretty: bool = False) -> str:
262      """ 
263      Return the name of the current OS distribution, as a human-readable 
264      string. 
265   
266      If *pretty* is false, the name is returned without version or codename. 
267      (e.g. "CentOS Linux") 
268   
269      If *pretty* is true, the version and codename are appended. 
270      (e.g. "CentOS Linux 7.1.1503 (Core)") 
271   
272      **Lookup hierarchy:** 
273   
274      The name is obtained from the following sources, in the specified order. 
275      The first available and non-empty value is used: 
276   
277      * If *pretty* is false: 
278   
279        - the value of the "NAME" attribute of the os-release file, 
280   
281        - the value of the "Distributor ID" attribute returned by the lsb_release 
282          command, 
283   
284        - the value of the "<name>" field of the distro release file. 
285   
286      * If *pretty* is true: 
287   
288        - the value of the "PRETTY_NAME" attribute of the os-release file, 
289   
290        - the value of the "Description" attribute returned by the lsb_release 
291          command, 
292   
293        - the value of the "<name>" field of the distro release file, appended 
294          with the value of the pretty version ("<version_id>" and "<codename>" 
295          fields) of the distro release file, if available. 
296      """
297      return _distro.name(pretty)

300  def version(pretty: bool = False, best: bool = False) -> str:
301      """ 
302      Return the version of the current OS distribution, as a human-readable 
303      string. 
304   
305      If *pretty* is false, the version is returned without codename (e.g. 
306      "7.0"). 
307   
308      If *pretty* is true, the codename in parenthesis is appended, if the 
309      codename is non-empty (e.g. "7.0 (Maipo)"). 
310   
311      Some distributions provide version numbers with different precisions in 
312      the different sources of distribution information. Examining the different 
313      sources in a fixed priority order does not always yield the most precise 
314      version (e.g. for Debian 8.2, or CentOS 7.1). 
315   
316      Some other distributions may not provide this kind of information. In these 
317      cases, an empty string would be returned. This behavior can be observed 
318      with rolling releases distributions (e.g. Arch Linux). 
319   
320      The *best* parameter can be used to control the approach for the returned 
321      version: 
322   
323      If *best* is false, the first non-empty version number in priority order of 
324      the examined sources is returned. 
325   
326      If *best* is true, the most precise version number out of all examined 
327      sources is returned. 
328   
329      **Lookup hierarchy:** 
330   
331      In all cases, the version number is obtained from the following sources. 
332      If *best* is false, this order represents the priority order: 
333   
334      * the value of the "VERSION_ID" attribute of the os-release file, 
335      * the value of the "Release" attribute returned by the lsb_release 
336        command, 
337      * the version number parsed from the "<version_id>" field of the first line 
338        of the distro release file, 
339      * the version number parsed from the "PRETTY_NAME" attribute of the 
340        os-release file, if it follows the format of the distro release files. 
341      * the version number parsed from the "Description" attribute returned by 
342        the lsb_release command, if it follows the format of the distro release 
343        files. 
344      """
345      return _distro.version(pretty, best)

421  def codename() -> str:
422      """ 
423      Return the codename for the release of the current OS distribution, 
424      as a string. 
425   
426      If the distribution does not have a codename, an empty string is returned. 
427   
428      Note that the returned codename is not always really a codename. For 
429      example, openSUSE returns "x86_64". This function does not handle such 
430      cases in any special way and just returns the string it finds, if any. 
431   
432      **Lookup hierarchy:** 
433   
434      * the codename within the "VERSION" attribute of the os-release file, if 
435        provided, 
436   
437      * the value of the "Codename" attribute returned by the lsb_release 
438        command, 
439   
440      * the value of the "<codename>" field of the distro release file. 
441      """
442      return _distro.codename()

603  try:
604      from functools import cached_property
605  except ImportError:
606      # Python < 3.8
607      class cached_property:  # type: ignore
608          """A version of @property which caches the value.  On access, it calls the 
609          underlying function and sets the value in `__dict__` so future accesses 
610          will not re-call the property. 
611          """
612  
613          def __init__(self, f: Callable[[Any], Any]) -> None:
614              self._fname = f.__name__
615              self._f = f
616  
617          def __get__(self, obj: Any, owner: Type[Any]) -> Any:
618              assert obj is not None, f"call {self._fname} on an instance"
619              ret = obj.__dict__[self._fname] = self._f(obj)
620              return ret

623  class LinuxDistribution:
624      """ 
625      Provides information about a OS distribution. 
626   
627      This package creates a private module-global instance of this class with 
628      default initialization arguments, that is used by the 
629      `consolidated accessor functions`_ and `single source accessor functions`_. 
630      By using default initialization arguments, that module-global instance 
631      returns data about the current OS distribution (i.e. the distro this 
632      package runs on). 
633   
634      Normally, it is not necessary to create additional instances of this class. 
635      However, in situations where control is needed over the exact data sources 
636      that are used, instances of this class can be created with a specific 
637      distro release file, or a specific os-release file, or without invoking the 
638      lsb_release command. 
639      """
640  
641      def __init__(
642          self,
643          include_lsb: Optional[bool] = None,
644          os_release_file: str = "",
645          distro_release_file: str = "",
646          include_uname: Optional[bool] = None,
647          root_dir: Optional[str] = None,
648          include_oslevel: Optional[bool] = None,
649      ) -> None:
650          """ 
651          The initialization method of this class gathers information from the 
652          available data sources, and stores that in private instance attributes. 
653          Subsequent access to the information items uses these private instance 
654          attributes, so that the data sources are read only once. 
655   
656          Parameters: 
657   
658          * ``include_lsb`` (bool): Controls whether the 
659            `lsb_release command output`_ is included as a data source. 
660   
661            If the lsb_release command is not available in the program execution 
662            path, the data source for the lsb_release command will be empty. 
663   
664          * ``os_release_file`` (string): The path name of the 
665            `os-release file`_ that is to be used as a data source. 
666   
667            An empty string (the default) will cause the default path name to 
668            be used (see `os-release file`_ for details). 
669   
670            If the specified or defaulted os-release file does not exist, the 
671            data source for the os-release file will be empty. 
672   
673          * ``distro_release_file`` (string): The path name of the 
674            `distro release file`_ that is to be used as a data source. 
675   
676            An empty string (the default) will cause a default search algorithm 
677            to be used (see `distro release file`_ for details). 
678   
679            If the specified distro release file does not exist, or if no default 
680            distro release file can be found, the data source for the distro 
681            release file will be empty. 
682   
683          * ``include_uname`` (bool): Controls whether uname command output is 
684            included as a data source. If the uname command is not available in 
685            the program execution path the data source for the uname command will 
686            be empty. 
687   
688          * ``root_dir`` (string): The absolute path to the root directory to use 
689            to find distro-related information files. Note that ``include_*`` 
690            parameters must not be enabled in combination with ``root_dir``. 
691   
692          * ``include_oslevel`` (bool): Controls whether (AIX) oslevel command 
693            output is included as a data source. If the oslevel command is not 
694            available in the program execution path the data source will be 
695            empty. 
696   
697          Public instance attributes: 
698   
699          * ``os_release_file`` (string): The path name of the 
700            `os-release file`_ that is actually used as a data source. The 
701            empty string if no distro release file is used as a data source. 
702   
703          * ``distro_release_file`` (string): The path name of the 
704            `distro release file`_ that is actually used as a data source. The 
705            empty string if no distro release file is used as a data source. 
706   
707          * ``include_lsb`` (bool): The result of the ``include_lsb`` parameter. 
708            This controls whether the lsb information will be loaded. 
709   
710          * ``include_uname`` (bool): The result of the ``include_uname`` 
711            parameter. This controls whether the uname information will 
712            be loaded. 
713   
714          * ``include_oslevel`` (bool): The result of the ``include_oslevel`` 
715            parameter. This controls whether (AIX) oslevel information will be 
716            loaded. 
717   
718          * ``root_dir`` (string): The result of the ``root_dir`` parameter. 
719            The absolute path to the root directory to use to find distro-related 
720            information files. 
721   
722          Raises: 
723   
724          * :py:exc:`ValueError`: Initialization parameters combination is not 
725             supported. 
726   
727          * :py:exc:`OSError`: Some I/O issue with an os-release file or distro 
728            release file. 
729   
730          * :py:exc:`UnicodeError`: A data source has unexpected characters or 
731            uses an unexpected encoding. 
732          """
733          self.root_dir = root_dir
734          self.etc_dir = os.path.join(root_dir, "etc") if root_dir else _UNIXCONFDIR
735          self.usr_lib_dir = (
736              os.path.join(root_dir, "usr/lib") if root_dir else _UNIXUSRLIBDIR
737          )
738  
739          if os_release_file:
740              self.os_release_file = os_release_file
741          else:
742              etc_dir_os_release_file = os.path.join(self.etc_dir, _OS_RELEASE_BASENAME)
743              usr_lib_os_release_file = os.path.join(
744                  self.usr_lib_dir, _OS_RELEASE_BASENAME
745              )
746  
747              # NOTE: The idea is to respect order **and** have it set
748              #       at all times for API backwards compatibility.
749              if os.path.isfile(etc_dir_os_release_file) or not os.path.isfile(
750                  usr_lib_os_release_file
751              ):
752                  self.os_release_file = etc_dir_os_release_file
753              else:
754                  self.os_release_file = usr_lib_os_release_file
755  
756          self.distro_release_file = distro_release_file or ""  # updated later
757  
758          is_root_dir_defined = root_dir is not None
759          if is_root_dir_defined and (include_lsb or include_uname or include_oslevel):
760              raise ValueError(
761                  "Including subprocess data sources from specific root_dir is disallowed"
762                  " to prevent false information"
763              )
764          self.include_lsb = (
765              include_lsb if include_lsb is not None else not is_root_dir_defined
766          )
767          self.include_uname = (
768              include_uname if include_uname is not None else not is_root_dir_defined
769          )
770          self.include_oslevel = (
771              include_oslevel if include_oslevel is not None else not is_root_dir_defined
772          )

807      def id(self) -> str:
808          """Return the distro ID of the OS distribution, as a string. 
809   
810          For details, see :func:`distro.id`. 
811          """
812  
813          def normalize(distro_id: str, table: Dict[str, str]) -> str:
814              distro_id = distro_id.lower().replace(" ", "_")
815              return table.get(distro_id, distro_id)
816  
817          distro_id = self.os_release_attr("id")
818          if distro_id:
819              return normalize(distro_id, NORMALIZED_OS_ID)
820  
821          distro_id = self.lsb_release_attr("distributor_id")
822          if distro_id:
823              return normalize(distro_id, NORMALIZED_LSB_ID)
824  
825          distro_id = self.distro_release_attr("id")
826          if distro_id:
827              return normalize(distro_id, NORMALIZED_DISTRO_ID)
828  
829          distro_id = self.uname_attr("id")
830          if distro_id:
831              return normalize(distro_id, NORMALIZED_DISTRO_ID)
832  
833          return ""

835      def name(self, pretty: bool = False) -> str:
836          """ 
837          Return the name of the OS distribution, as a string. 
838   
839          For details, see :func:`distro.name`. 
840          """
841          name = (
842              self.os_release_attr("name")
843              or self.lsb_release_attr("distributor_id")
844              or self.distro_release_attr("name")
845              or self.uname_attr("name")
846          )
847          if pretty:
848              name = self.os_release_attr("pretty_name") or self.lsb_release_attr(
849                  "description"
850              )
851              if not name:
852                  name = self.distro_release_attr("name") or self.uname_attr("name")
853                  version = self.version(pretty=True)
854                  if version:
855                      name = f"{name} {version}"
856          return name or ""

858      def version(self, pretty: bool = False, best: bool = False) -> str:
859          """ 
860          Return the version of the OS distribution, as a string. 
861   
862          For details, see :func:`distro.version`. 
863          """
864          versions = [
865              self.os_release_attr("version_id"),
866              self.lsb_release_attr("release"),
867              self.distro_release_attr("version_id"),
868              self._parse_distro_release_content(self.os_release_attr("pretty_name")).get(
869                  "version_id", ""
870              ),
871              self._parse_distro_release_content(
872                  self.lsb_release_attr("description")
873              ).get("version_id", ""),
874              self.uname_attr("release"),
875          ]
876          if self.uname_attr("id").startswith("aix"):
877              # On AIX platforms, prefer oslevel command output.
878              versions.insert(0, self.oslevel_info())
879          version = ""
880          if best:
881              # This algorithm uses the last version in priority order that has
882              # the best precision. If the versions are not in conflict, that
883              # does not matter; otherwise, using the last one instead of the
884              # first one might be considered a surprise.
885              for v in versions:
886                  if v.count(".") > version.count(".") or version == "":
887                      version = v
888          else:
889              for v in versions:
890                  if v != "":
891                      version = v
892                      break
893          if pretty and version and self.codename():
894              version = f"{version} ({self.codename()})"
895          return version

897      def version_parts(self, best: bool = False) -> Tuple[str, str, str]:
898          """ 
899          Return the version of the OS distribution, as a tuple of version 
900          numbers. 
901   
902          For details, see :func:`distro.version_parts`. 
903          """
904          version_str = self.version(best=best)
905          if version_str:
906              version_regex = re.compile(r"(\d+)\.?(\d+)?\.?(\d+)?")
907              matches = version_regex.match(version_str)
908              if matches:
909                  major, minor, build_number = matches.groups()
910                  return major, minor or "", build_number or ""
911          return "", "", ""

1061     @cached_property
1062     def _os_release_info(self) -> Dict[str, str]:
1063         """ 
1064         Get the information items from the specified os-release file. 
1065  
1066         Returns: 
1067             A dictionary containing all information items. 
1068         """

1071                 return self._parse_os_release_content(release_file)

1074     @staticmethod
1075     def _parse_os_release_content(lines: TextIO) -> Dict[str, str]:
1076         """ 
1077         Parse the lines of an os-release file. 
1078  
1079         Parameters: 
1080  
1081         * lines: Iterable through the lines in the os-release file. 
1082                  Each line must be a unicode string or a UTF-8 encoded byte 
1083                  string. 
1084  
1085         Returns: 
1086             A dictionary containing all information items. 
1087         """
1088         props = {}
1089         lexer = shlex.shlex(lines, posix=True)
1090         lexer.whitespace_split = True
1091 
1092         tokens = list(lexer)
1093         for token in tokens:
1094             # At this point, all shell-like parsing has been done (i.e.
1095             # comments processed, quotes and backslash escape sequences
1096             # processed, multi-line values assembled, trailing newlines
1097             # stripped, etc.), so the tokens are now either:
1098             # * variable assignments: var=value
1099             # * commands or their arguments (not allowed in os-release)
1100             # Ignore any tokens that are not variable assignments
1101             if "=" in token:
1102                 k, v = token.split("=", 1)
1103                 props[k.lower()] = v
1104 
1105         if "version" in props:
1106             # extract release codename (if any) from version attribute
1107             match = re.search(r"\((\D+)\)|,\s*(\D+)", props["version"])
1108             if match:
1109                 release_codename = match.group(1) or match.group(2)
1110                 props["codename"] = props["release_codename"] = release_codename
1111 
1112         if "version_codename" in props:
1113             # os-release added a version_codename field.  Use that in
1114             # preference to anything else Note that some distros purposefully
1115             # do not have code names.  They should be setting
1116             # version_codename=""
1117             props["codename"] = props["version_codename"]
1118         elif "ubuntu_codename" in props:
1119             # Same as above but a non-standard field name used on older Ubuntus
1120             props["codename"] = props["ubuntu_codename"]
1121 
1122         return props

1124     @cached_property
1125     def _lsb_release_info(self) -> Dict[str, str]:
1126         """ 
1127         Get the information items from the lsb_release command output. 
1128  
1129         Returns: 
1130             A dictionary containing all information items. 
1131         """
1132         if not self.include_lsb:
1133             return {}
1134         try:
1135             cmd = ("lsb_release", "-a")
1136             stdout = subprocess.check_output(cmd, stderr=subprocess.DEVNULL)
1137         # Command not found or lsb_release returned error
1138         except (OSError, subprocess.CalledProcessError):
1139             return {}
1140         content = self._to_str(stdout).splitlines()
1141         return self._parse_lsb_release_content(content)

1143     @staticmethod
1144     def _parse_lsb_release_content(lines: Iterable[str]) -> Dict[str, str]:
1145         """ 
1146         Parse the output of the lsb_release command. 
1147  
1148         Parameters: 
1149  
1150         * lines: Iterable through the lines of the lsb_release output. 
1151                  Each line must be a unicode string or a UTF-8 encoded byte 
1152                  string. 
1153  
1154         Returns: 
1155             A dictionary containing all information items. 
1156         """
1157         props = {}
1158         for line in lines:
1159             kv = line.strip("\n").split(":", 1)
1160             if len(kv) != 2:
1161                 # Ignore lines without colon.
1162                 continue
1163             k, v = kv
1164             props.update({k.replace(" ", "_").lower(): v.strip()})
1165         return props

1189     @staticmethod
1190     def _parse_uname_content(lines: Sequence[str]) -> Dict[str, str]:
1191         if not lines:
1192             return {}
1193         props = {}
1194         match = re.search(r"^([^\s]+)\s+([\d\.]+)", lines[0].strip())
1195         if match:
1196             name, version = match.groups()
1197 
1198             # This is to prevent the Linux kernel version from
1199             # appearing as the 'best' version on otherwise
1200             # identifiable distributions.
1201             if name == "Linux":
1202                 return {}
1203             props["id"] = name.lower()
1204             props["name"] = name
1205             props["release"] = version
1206         return props

1213     @cached_property
1214     def _distro_release_info(self) -> Dict[str, str]:
1215         """ 
1216         Get the information items from the specified distro release file. 
1217  
1218         Returns: 
1219             A dictionary containing all information items. 
1220         """
1221         if self.distro_release_file:
1222             # If it was specified, we use it and parse what we can, even if
1223             # its file name or content does not match the expected pattern.
1224             distro_info = self._parse_distro_release_file(self.distro_release_file)
1225             basename = os.path.basename(self.distro_release_file)
1226             # The file name pattern for user-specified distro release files
1227             # is somewhat more tolerant (compared to when searching for the
1228             # file), because we want to use what was specified as best as
1229             # possible.
1230             match = _DISTRO_RELEASE_BASENAME_PATTERN.match(basename)
1231             if "name" in distro_info and "cloudlinux" in distro_info["name"].lower():
1232                 distro_info["id"] = "cloudlinux"
1233             elif match:
1234                 distro_info["id"] = match.group(1)
1235             return distro_info
1236         else:
1237             try:
1238                 basenames = os.listdir(self.etc_dir)
1239                 # We sort for repeatability in cases where there are multiple
1240                 # distro specific files; e.g. CentOS, Oracle, Enterprise all
1241                 # containing `redhat-release` on top of their own.
1242                 basenames.sort()
1243             except OSError:
1244                 # This may occur when /etc is not readable but we can't be
1245                 # sure about the *-release files. Check common entries of
1246                 # /etc for information. If they turn out to not be there the
1247                 # error is handled in `_parse_distro_release_file()`.
1248                 basenames = [
1249                     "SuSE-release",
1250                     "arch-release",
1251                     "base-release",
1252                     "centos-release",
1253                     "fedora-release",
1254                     "gentoo-release",
1255                     "mageia-release",
1256                     "mandrake-release",
1257                     "mandriva-release",
1258                     "mandrivalinux-release",
1259                     "manjaro-release",
1260                     "oracle-release",
1261                     "redhat-release",
1262                     "rocky-release",
1263                     "sl-release",
1264                     "slackware-version",
1265                 ]
1266             for basename in basenames:
1267                 if basename in _DISTRO_RELEASE_IGNORE_BASENAMES:
1268                     continue
1269                 match = _DISTRO_RELEASE_BASENAME_PATTERN.match(basename)
1270                 if match:
1271                     filepath = os.path.join(self.etc_dir, basename)
1272                     distro_info = self._parse_distro_release_file(filepath)
1273                     if "name" in distro_info:
1274                         # The name is always present if the pattern matches
1275                         self.distro_release_file = filepath
1276                         distro_info["id"] = match.group(1)
1277                         if "cloudlinux" in distro_info["name"].lower():
1278                             distro_info["id"] = "cloudlinux"
1279                         return distro_info
1280             return {}
1281 
1282     def _parse_distro_release_file(self, filepath: str) -> Dict[str, str]:
1283         """ 
1284         Parse a distro release file. 
1285  
1286         Parameters: 
1287  
1288         * filepath: Path name of the distro release file. 
1289  
1290         Returns: 
1291             A dictionary containing all information items. 
1292         """
1293         try:
1294             with open(filepath, encoding="utf-8") as fp:
1295                 # Only parse the first line. For instance, on SLES there
1296                 # are multiple lines. We don't want them...
1297                 return self._parse_distro_release_content(fp.readline())
1298         except OSError:
1299             # Ignore not being able to read a specific, seemingly version
1300             # related file.
1301             # See https://github.com/python-distro/distro/issues/162
1302             return {}
1303 
1304     @staticmethod
1305     def _parse_distro_release_content(line: str) -> Dict[str, str]:
1306         """ 
1307         Parse a line from a distro release file. 
1308  
1309         Parameters: 
1310         * line: Line from the distro release file. Must be a unicode string 
1311                 or a UTF-8 encoded byte string. 
1312  
1313         Returns: 
1314             A dictionary containing all information items. 
1315         """
1316         matches = _DISTRO_RELEASE_CONTENT_REVERSED_PATTERN.match(line.strip()[::-1])
1317         distro_info = {}
1318         if matches:
1319             # regexp ensures non-None
1320             distro_info["name"] = matches.group(3)[::-1]
1321             if matches.group(2):
1322                 distro_info["version_id"] = matches.group(2)[::-1]
1323             if matches.group(1):
1324                 distro_info["codename"] = matches.group(1)[::-1]
1325         elif line:
1326             distro_info["name"] = line.strip()
1327         return distro_info
