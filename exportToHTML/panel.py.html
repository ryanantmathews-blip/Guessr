16   class Panel(JupyterMixin):
17       """A console renderable that draws a border around its contents. 
18    
19       Example: 
20           >>> console.print(Panel("Hello, World!")) 
21    
22       Args: 
23           renderable (RenderableType): A console renderable object. 
24           box (Box, optional): A Box instance that defines the look of the border (see :ref:`appendix_box`. 
25               Defaults to box.ROUNDED. 
26           safe_box (bool, optional): Disable box characters that don't display on windows legacy terminal with *raster* fonts. Defaults to True. 
27           expand (bool, optional): If True the panel will stretch to fill the console 
28               width, otherwise it will be sized to fit the contents. Defaults to True. 
29           style (str, optional): The style of the panel (border and contents). Defaults to "none". 
30           border_style (str, optional): The style of the border. Defaults to "none". 
31           width (Optional[int], optional): Optional width of panel. Defaults to None to auto-detect. 
32           height (Optional[int], optional): Optional height of panel. Defaults to None to auto-detect. 
33           padding (Optional[PaddingDimensions]): Optional padding around renderable. Defaults to 0. 
34           highlight (bool, optional): Enable automatic highlighting of panel title (if str). Defaults to False. 
35       """

102      @property
103      def _title(self) -> Optional[Text]:
104          if self.title:
105              title_text = (
106                  Text.from_markup(self.title)
107                  if isinstance(self.title, str)
108                  else self.title.copy()
109              )
110              title_text.end = ""
111              title_text.plain = title_text.plain.replace("\n", " ")
112              title_text.no_wrap = True
113              title_text.expand_tabs()
114              title_text.pad(1)
115              return title_text
116          return None

122                  Text.from_markup(self.subtitle)
123                  if isinstance(self.subtitle, str)
124                  else self.subtitle.copy()
125              )
126              subtitle_text.end = ""
127              subtitle_text.plain = subtitle_text.plain.replace("\n", " ")
128              subtitle_text.no_wrap = True
129              subtitle_text.expand_tabs()
130              subtitle_text.pad(1)
131              return subtitle_text
132          return None

134      def __rich_console__(
135          self, console: "Console", options: "ConsoleOptions"
136      ) -> "RenderResult":
137          _padding = Padding.unpack(self.padding)
138          renderable = (
139              Padding(self.renderable, _padding) if any(_padding) else self.renderable
140          )
141          style = console.get_style(self.style)
142          border_style = style + console.get_style(self.border_style)
143          width = (
144              options.max_width
145              if self.width is None
146              else min(options.max_width, self.width)
147          )
148  
149          safe_box: bool = console.safe_box if self.safe_box is None else self.safe_box
150          box = self.box.substitute(options, safe=safe_box)
151  
152          title_text = self._title
153          if title_text is not None:
154              title_text.style = border_style
155  
156          child_width = (
157              width - 2
158              if self.expand
159              else console.measure(
160                  renderable, options=options.update_width(width - 2)
161              ).maximum
162          )
163          child_height = self.height or options.height or None
164          if child_height:
165              child_height -= 2
166          if title_text is not None:
167              child_width = min(
168                  options.max_width - 2, max(child_width, title_text.cell_len + 2)
169              )
170  
171          width = child_width + 2
172          child_options = options.update(
173              width=child_width, height=child_height, highlight=self.highlight
174          )
175          lines = console.render_lines(renderable, child_options, style=style)
176  
177          line_start = Segment(box.mid_left, border_style)
178          line_end = Segment(f"{box.mid_right}", border_style)
179          new_line = Segment.line()
180          if title_text is None or width <= 4:
181              yield Segment(box.get_top([width - 2]), border_style)
182          else:
183              title_text.align(self.title_align, width - 4, character=box.top)
184              yield Segment(box.top_left + box.top, border_style)
185              yield from console.render(title_text, child_options.update_width(width - 4))
186              yield Segment(box.top + box.top_right, border_style)
187  
188          yield new_line
189          for line in lines:
190              yield line_start
191              yield from line
192              yield line_end
193              yield new_line
194  
195          subtitle_text = self._subtitle
196          if subtitle_text is not None:
197              subtitle_text.style = border_style
198  
199          if subtitle_text is None or width <= 4:
200              yield Segment(box.get_bottom([width - 2]), border_style)
201          else:
202              subtitle_text.align(self.subtitle_align, width - 4, character=box.bottom)
203              yield Segment(box.bottom_left + box.bottom, border_style)
204              yield from console.render(
205                  subtitle_text, child_options.update_width(width - 4)
206              )
207              yield Segment(box.bottom + box.bottom_right, border_style)
208  
209          yield new_line
210  
211      def __rich_measure__(
212          self, console: "Console", options: "ConsoleOptions"
213      ) -> "Measurement":
214          _title = self._title
215          _, right, _, left = Padding.unpack(self.padding)
216          padding = left + right
217          renderables = [self.renderable, _title] if _title else [self.renderable]
218  
219          if self.width is None:
220              width = (
221                  measure_renderables(
222                      console,
223                      options.update_width(options.max_width - padding - 2),
224                      renderables,
225                  ).maximum
226                  + padding
227                  + 2
228              )
229          else:
230              width = self.width
231          return Measurement(width, width)
