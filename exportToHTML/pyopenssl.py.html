1    """ 
2    TLS with SNI_-support for Python 2. Follow these instructions if you would 
3    like to verify TLS certificates in Python 2. Note, the default libraries do 
4    *not* do certificate checking; you need to do additional work to validate 
5    certificates yourself. 
6     
7    This needs the following packages installed: 
8     
9    * `pyOpenSSL`_ (tested with 16.0.0) 
10   * `cryptography`_ (minimum 1.3.4, from pyopenssl) 
11   * `idna`_ (minimum 2.0, from cryptography) 
12    
13   However, pyopenssl depends on cryptography, which depends on idna, so while we 
14   use all three directly here we end up having relatively few packages required. 
15    
16   You can install them with the following command: 
17    
18   .. code-block:: bash 
19    
20       $ python -m pip install pyopenssl cryptography idna 
21    
22   To activate certificate checking, call 
23   :func:`~urllib3.contrib.pyopenssl.inject_into_urllib3` from your Python code 
24   before you begin making HTTP requests. This can be done in a ``sitecustomize`` 
25   module, or at any other time before your application begins using ``urllib3``, 
26   like this: 
27    
28   .. code-block:: python 
29    
30       try: 
31           import pip._vendor.urllib3.contrib.pyopenssl as pyopenssl 
32           pyopenssl.inject_into_urllib3() 
33       except ImportError: 
34           pass 
35    
36   Now you can use :mod:`urllib3` as you normally would, and it will support SNI 
37   when the required modules are installed. 
38    
39   Activating this module also has the positive side effect of disabling SSL/TLS 
40   compression in Python 2 (see `CRIME attack`_). 
41    
42   .. _sni: https://en.wikipedia.org/wiki/Server_Name_Indication 
43   .. _crime attack: https://en.wikipedia.org/wiki/CRIME_(security_exploit) 
44   .. _pyopenssl: https://www.pyopenssl.org 
45   .. _cryptography: https://cryptography.io 
46   .. _idna: https://github.com/kjd/idna 
47   """

154  def _validate_dependencies_met():
155      """ 
156      Verifies that PyOpenSSL's package-level dependencies have been met. 
157      Throws `ImportError` if they are not met. 
158      """
159      # Method added in `cryptography==1.1`; not available in older versions
160      from cryptography.x509.extensions import Extensions
161  
162      if getattr(Extensions, "get_extension_for_class", None) is None:
163          raise ImportError(
164              "'cryptography' module missing required functionality.  "
165              "Try upgrading to v1.3.4 or newer."
166          )
167  
168      # pyOpenSSL 0.14 and above use cryptography for OpenSSL bindings. The _x509
169      # attribute is only present on those versions.
170      from OpenSSL.crypto import X509
171  
172      x509 = X509()
173      if getattr(x509, "_x509", None) is None:
174          raise ImportError(
175              "'pyOpenSSL' module missing required functionality. "
176              "Try upgrading to v0.14 or newer."
177          )

180  def _dnsname_to_stdlib(name):
181      """ 
182      Converts a dNSName SubjectAlternativeName field to the form used by the 
183      standard library on the given Python version. 
184   
185      Cryptography produces a dNSName as a unicode string that was idna-decoded 
186      from ASCII bytes. We need to idna-encode that string to get it back, and 
187      then on Python 3 we also need to convert to unicode via UTF-8 (the stdlib 
188      uses PyUnicode_FromStringAndSize on it, which decodes via UTF-8). 
189   
190      If the name cannot be idna-encoded then we return None signalling that 
191      the name given should be skipped. 
192      """
193  
194      def idna_encode(name):
195          """ 
196          Borrowed wholesale from the Python Cryptography Project. It turns out 
197          that we can't just safely call `idna.encode`: it can explode for 
198          wildcard names. This avoids that problem. 
199          """
200          from pip._vendor import idna
201  
202          try:
203              for prefix in [u"*.", u"."]:
204                  if name.startswith(prefix):
205                      name = name[len(prefix) :]
206                      return prefix.encode("ascii") + idna.encode(name)
207              return idna.encode(name)
208          except idna.core.IDNAError:
209              return None
210  
211      # Don't send IPv6 addresses through the IDNA encoder.
212      if ":" in name:
213          return name
214  
215      name = idna_encode(name)
216      if name is None:
217          return None
218      elif sys.version_info >= (3, 0):
219          name = name.decode("utf-8")
220      return name

223  def get_subj_alt_name(peer_cert):
224      """ 
225      Given an PyOpenSSL certificate, provides all the subject alternative names. 
226      """
227      # Pass the cert to cryptography, which has much better APIs for this.
228      if hasattr(peer_cert, "to_cryptography"):
229          cert = peer_cert.to_cryptography()
230      else:
231          # This is technically using private APIs, but should work across all
232          # relevant versions before PyOpenSSL got a proper API for this.
233          cert = _Certificate(openssl_backend, peer_cert._x509)
234  
235      # We want to find the SAN extension. Ask Cryptography to locate it (it's
236      # faster than looping in Python)
237      try:
238          ext = cert.extensions.get_extension_for_class(x509.SubjectAlternativeName).value
239      except x509.ExtensionNotFound:
240          # No such extension, return the empty list.
241          return []
242      except (
243          x509.DuplicateExtension,
244          UnsupportedExtension,
245          x509.UnsupportedGeneralNameType,
246          UnicodeError,
247      ) as e:
248          # A problem has been found with the quality of the certificate. Assume
249          # no SAN field is present.
250          log.warning(
251              "A problem was encountered with the certificate that prevented "
252              "urllib3 from finding the SubjectAlternativeName field. This can "
253              "affect certificate validation. The error was %s",
254              e,
255          )
256          return []
257  
258      # We want to return dNSName and iPAddress fields. We need to cast the IPs
259      # back to strings because the match_hostname function wants them as
260      # strings.
261      # Sadly the DNS names need to be idna encoded and then, on Python 3, UTF-8
262      # decoded. This is pretty frustrating, but that's what the standard library
263      # does with certificates, and so we need to attempt to do the same.
264      # We also want to skip over names which cannot be idna encoded.
265      names = [
266          ("DNS", name)
267          for name in map(_dnsname_to_stdlib, ext.get_values_for_type(x509.DNSName))
268          if name is not None
269      ]
270      names.extend(
271          ("IP Address", str(name)) for name in ext.get_values_for_type(x509.IPAddress)
272      )
273  
274      return names

277  class WrappedSocket(object):
278      """API-compatibility wrapper for Python OpenSSL's Connection-class. 
279   
280      Note: _makefile_refs, _drop() and _reuse() are needed for the garbage 
281      collector of pypy. 
282      """
283  
284      def __init__(self, connection, socket, suppress_ragged_eofs=True):
285          self.connection = connection
286          self.socket = socket
287          self.suppress_ragged_eofs = suppress_ragged_eofs
288          self._makefile_refs = 0
289          self._closed = False
290  
291      def fileno(self):
292          return self.socket.fileno()
293  
294      # Copy-pasted from Python 3.5 source code
295      def _decref_socketios(self):
296          if self._makefile_refs > 0:
297              self._makefile_refs -= 1
298          if self._closed:
299              self.close()
300  
301      def recv(self, *args, **kwargs):
302          try:
303              data = self.connection.recv(*args, **kwargs)
304          except OpenSSL.SSL.SysCallError as e:
305              if self.suppress_ragged_eofs and e.args == (-1, "Unexpected EOF"):
306                  return b""
307              else:
308                  raise SocketError(str(e))
309          except OpenSSL.SSL.ZeroReturnError:
310              if self.connection.get_shutdown() == OpenSSL.SSL.RECEIVED_SHUTDOWN:
311                  return b""
312              else:
313                  raise
314          except OpenSSL.SSL.WantReadError:
315              if not util.wait_for_read(self.socket, self.socket.gettimeout()):
316                  raise timeout("The read operation timed out")
317              else:
318                  return self.recv(*args, **kwargs)
319  
320          # TLS 1.3 post-handshake authentication
321          except OpenSSL.SSL.Error as e:
322              raise ssl.SSLError("read error: %r" % e)
323          else:
324              return data
325  
326      def recv_into(self, *args, **kwargs):
327          try:
328              return self.connection.recv_into(*args, **kwargs)
329          except OpenSSL.SSL.SysCallError as e:
330              if self.suppress_ragged_eofs and e.args == (-1, "Unexpected EOF"):
331                  return 0
332              else:
333                  raise SocketError(str(e))
334          except OpenSSL.SSL.ZeroReturnError:
335              if self.connection.get_shutdown() == OpenSSL.SSL.RECEIVED_SHUTDOWN:
336                  return 0
337              else:
338                  raise
339          except OpenSSL.SSL.WantReadError:
340              if not util.wait_for_read(self.socket, self.socket.gettimeout()):
341                  raise timeout("The read operation timed out")
342              else:
343                  return self.recv_into(*args, **kwargs)
344  
345          # TLS 1.3 post-handshake authentication
346          except OpenSSL.SSL.Error as e:
347              raise ssl.SSLError("read error: %r" % e)
348  
349      def settimeout(self, timeout):
350          return self.socket.settimeout(timeout)
351  
352      def _send_until_done(self, data):
353          while True:
354              try:
355                  return self.connection.send(data)
356              except OpenSSL.SSL.WantWriteError:
357                  if not util.wait_for_write(self.socket, self.socket.gettimeout()):
358                      raise timeout()
359                  continue
360              except OpenSSL.SSL.SysCallError as e:
361                  raise SocketError(str(e))
362  
363      def sendall(self, data):
364          total_sent = 0
365          while total_sent < len(data):
366              sent = self._send_until_done(
367                  data[total_sent : total_sent + SSL_WRITE_BLOCKSIZE]
368              )
369              total_sent += sent
370  
371      def shutdown(self):
372          # FIXME rethrow compatible exceptions should we ever use this
373          self.connection.shutdown()
374  
375      def close(self):
376          if self._makefile_refs < 1:
377              try:
378                  self._closed = True
379                  return self.connection.close()
380              except OpenSSL.SSL.Error:
381                  return
382          else:
383              self._makefile_refs -= 1
384  
385      def getpeercert(self, binary_form=False):
386          x509 = self.connection.get_peer_certificate()
387  
388          if not x509:
389              return x509
390  
391          if binary_form:
392              return OpenSSL.crypto.dump_certificate(OpenSSL.crypto.FILETYPE_ASN1, x509)
393  
394          return {
395              "subject": ((("commonName", x509.get_subject().CN),),),
396              "subjectAltName": get_subj_alt_name(x509),
397          }
398  
399      def version(self):
400          return self.connection.get_protocol_version_name()
401  
402      def _reuse(self):
403          self._makefile_refs += 1
404  
405      def _drop(self):
406          if self._makefile_refs < 1:
407              self.close()
408          else:
409              self._makefile_refs -= 1
410

424  class PyOpenSSLContext(object):
425      """ 
426      I am a wrapper class for the PyOpenSSL ``Context`` object. I am responsible 
427      for translating the interface of the standard library ``SSLContext`` object 
428      to calls into PyOpenSSL. 
429      """
430  
431      def __init__(self, protocol):
432          self.protocol = _openssl_versions[protocol]
433          self._ctx = OpenSSL.SSL.Context(self.protocol)
434          self._options = 0
435          self.check_hostname = False
436  
437      @property
438      def options(self):
439          return self._options
440  
441      @options.setter
442      def options(self, value):
443          self._options = value
444          self._ctx.set_options(value)
445  
446      @property
447      def verify_mode(self):
448          return _openssl_to_stdlib_verify[self._ctx.get_verify_mode()]
449  
450      @verify_mode.setter
451      def verify_mode(self, value):
452          self._ctx.set_verify(_stdlib_to_openssl_verify[value], _verify_callback)
453  
454      def set_default_verify_paths(self):
455          self._ctx.set_default_verify_paths()
456  
457      def set_ciphers(self, ciphers):
458          if isinstance(ciphers, six.text_type):
459              ciphers = ciphers.encode("utf-8")
460          self._ctx.set_cipher_list(ciphers)
461  
462      def load_verify_locations(self, cafile=None, capath=None, cadata=None):
463          if cafile is not None:
464              cafile = cafile.encode("utf-8")
465          if capath is not None:
466              capath = capath.encode("utf-8")
467          try:
468              self._ctx.load_verify_locations(cafile, capath)
469              if cadata is not None:
470                  self._ctx.load_verify_locations(BytesIO(cadata))
471          except OpenSSL.SSL.Error as e:
472              raise ssl.SSLError("unable to load trusted certificates: %r" % e)
473  
474      def load_cert_chain(self, certfile, keyfile=None, password=None):
475          self._ctx.use_certificate_chain_file(certfile)
476          if password is not None:
477              if not isinstance(password, six.binary_type):
478                  password = password.encode("utf-8")
479              self._ctx.set_passwd_cb(lambda *_: password)
480          self._ctx.use_privatekey_file(keyfile or certfile)
481  
482      def set_alpn_protocols(self, protocols):
483          protocols = [six.ensure_binary(p) for p in protocols]
484          return self._ctx.set_alpn_protos(protocols)
485  
486      def wrap_socket(
487          self,
488          sock,
489          server_side=False,
490          do_handshake_on_connect=True,
491          suppress_ragged_eofs=True,
492          server_hostname=None,
493      ):
494          cnx = OpenSSL.SSL.Connection(self._ctx, sock)
495  
496          if isinstance(server_hostname, six.text_type):  # Platform-specific: Python 3
497              server_hostname = server_hostname.encode("utf-8")
498  
499          if server_hostname is not None:
500              cnx.set_tlsext_host_name(server_hostname)
501  
502          cnx.set_connect_state()
503  
504          while True:
505              try:
506                  cnx.do_handshake()
507              except OpenSSL.SSL.WantReadError:
508                  if not util.wait_for_read(sock, sock.gettimeout()):
509                      raise timeout("select timed out")
510                  continue
511              except OpenSSL.SSL.Error as e:
512                  raise ssl.SSLError("bad handshake: %r" % e)
513              break
514  
515          return WrappedSocket(cnx, sock)
516
