39   import codecs
40   import logging
41   import re
42   
43   from .charsetgroupprober import CharSetGroupProber
44   from .enums import InputState, LanguageFilter, ProbingState
45   from .escprober import EscCharSetProber
46   from .latin1prober import Latin1Prober
47   from .mbcsgroupprober import MBCSGroupProber
48   from .sbcsgroupprober import SBCSGroupProber
49   from .utf1632prober import UTF1632Prober

51   
52   class UniversalDetector:
53       """ 
54       The ``UniversalDetector`` class underlies the ``chardet.detect`` function 
55       and coordinates all of the different charset probers. 
56    
57       To get a ``dict`` containing an encoding and its confidence, you can simply 
58       run: 
59    
60       .. code:: 
61    
62               u = UniversalDetector() 
63               u.feed(some_bytes) 
64               u.close() 
65               detected = u.result 
66    
67       """
68   
69       MINIMUM_THRESHOLD = 0.20
70       HIGH_BYTE_DETECTOR = re.compile(b"[\x80-\xFF]")
71       ESC_DETECTOR = re.compile(b"(\033|~{)")
72       WIN_BYTE_DETECTOR = re.compile(b"[\x80-\x9F]")
73       ISO_WIN_MAP = {
74           "iso-8859-1": "Windows-1252",
75           "iso-8859-2": "Windows-1250",
76           "iso-8859-5": "Windows-1251",
77           "iso-8859-6": "Windows-1256",
78           "iso-8859-7": "Windows-1253",
79           "iso-8859-8": "Windows-1255",
80           "iso-8859-9": "Windows-1254",
81           "iso-8859-13": "Windows-1257",
82       }
83   
84       def __init__(self, lang_filter=LanguageFilter.ALL):
85           self._esc_charset_prober = None
86           self._utf1632_prober = None
87           self._charset_probers = []
88           self.result = None
89           self.done = None
90           self._got_data = None
91           self._input_state = None
92           self._last_char = None
93           self.lang_filter = lang_filter
94           self.logger = logging.getLogger(__name__)
95           self._has_win_bytes = None
96           self.reset()
97   
98       @property
99       def input_state(self):
100          return self._input_state
101  
102      @property
103      def has_win_bytes(self):
104          return self._has_win_bytes
105  
106      @property
107      def charset_probers(self):
108          return self._charset_probers
109  
110      def reset(self):
111          """ 
112          Reset the UniversalDetector and all of its probers back to their 
113          initial states.  This is called by ``__init__``, so you only need to 
114          call this directly in between analyses of different documents. 
115          """
116          self.result = {"encoding": None, "confidence": 0.0, "language": None}
117          self.done = False
118          self._got_data = False
119          self._has_win_bytes = False
120          self._input_state = InputState.PURE_ASCII
121          self._last_char = b""
122          if self._esc_charset_prober:
123              self._esc_charset_prober.reset()
124          if self._utf1632_prober:
125              self._utf1632_prober.reset()
126          for prober in self._charset_probers:
127              prober.reset()
128  
129      def feed(self, byte_str):
130          """ 
131          Takes a chunk of a document and feeds it through all of the relevant 
132          charset probers. 
133   
134          After calling ``feed``, you can check the value of the ``done`` 
135          attribute to see if you need to continue feeding the 
136          ``UniversalDetector`` more data, or if it has made a prediction 
137          (in the ``result`` attribute). 
138   
139          .. note:: 
140             You should always call ``close`` when you're done feeding in your 
141             document if ``done`` is not already ``True``. 
142          """
143          if self.done:
144              return
145  
146          if not byte_str:
147              return
148  
149          if not isinstance(byte_str, bytearray):
150              byte_str = bytearray(byte_str)
151  
152          # First check for known BOMs, since these are guaranteed to be correct
153          if not self._got_data:
154              # If the data starts with BOM, we know it is UTF
155              if byte_str.startswith(codecs.BOM_UTF8):
156                  # EF BB BF  UTF-8 with BOM
157                  self.result = {
158                      "encoding": "UTF-8-SIG",
159                      "confidence": 1.0,
160                      "language": "",
161                  }
162              elif byte_str.startswith((codecs.BOM_UTF32_LE, codecs.BOM_UTF32_BE)):
163                  # FF FE 00 00  UTF-32, little-endian BOM
164                  # 00 00 FE FF  UTF-32, big-endian BOM
165                  self.result = {"encoding": "UTF-32", "confidence": 1.0, "language": ""}
166              elif byte_str.startswith(b"\xFE\xFF\x00\x00"):
167                  # FE FF 00 00  UCS-4, unusual octet order BOM (3412)
168                  self.result = {
169                      "encoding": "X-ISO-10646-UCS-4-3412",
170                      "confidence": 1.0,
171                      "language": "",
172                  }
173              elif byte_str.startswith(b"\x00\x00\xFF\xFE"):
174                  # 00 00 FF FE  UCS-4, unusual octet order BOM (2143)
175                  self.result = {
176                      "encoding": "X-ISO-10646-UCS-4-2143",
177                      "confidence": 1.0,
178                      "language": "",
179                  }
180              elif byte_str.startswith((codecs.BOM_LE, codecs.BOM_BE)):
181                  # FF FE  UTF-16, little endian BOM
182                  # FE FF  UTF-16, big endian BOM
183                  self.result = {"encoding": "UTF-16", "confidence": 1.0, "language": ""}
184  
185              self._got_data = True
186              if self.result["encoding"] is not None:
187                  self.done = True
188                  return
189  
190          # If none of those matched and we've only see ASCII so far, check
191          # for high bytes and escape sequences
192          if self._input_state == InputState.PURE_ASCII:
193              if self.HIGH_BYTE_DETECTOR.search(byte_str):
194                  self._input_state = InputState.HIGH_BYTE
195              elif (
196                  self._input_state == InputState.PURE_ASCII
197                  and self.ESC_DETECTOR.search(self._last_char + byte_str)
198              ):
199                  self._input_state = InputState.ESC_ASCII
200  
201          self._last_char = byte_str[-1:]
202  
203          # next we will look to see if it is appears to be either a UTF-16 or
204          # UTF-32 encoding
205          if not self._utf1632_prober:
206              self._utf1632_prober = UTF1632Prober()
207  
208          if self._utf1632_prober.state == ProbingState.DETECTING:
209              if self._utf1632_prober.feed(byte_str) == ProbingState.FOUND_IT:
210                  self.result = {
211                      "encoding": self._utf1632_prober.charset_name,
212                      "confidence": self._utf1632_prober.get_confidence(),
213                      "language": "",
214                  }
215                  self.done = True
216                  return
217  
218          # If we've seen escape sequences, use the EscCharSetProber, which
219          # uses a simple state machine to check for known escape sequences in
220          # HZ and ISO-2022 encodings, since those are the only encodings that
221          # use such sequences.
222          if self._input_state == InputState.ESC_ASCII:
223              if not self._esc_charset_prober:
224                  self._esc_charset_prober = EscCharSetProber(self.lang_filter)
225              if self._esc_charset_prober.feed(byte_str) == ProbingState.FOUND_IT:
226                  self.result = {
227                      "encoding": self._esc_charset_prober.charset_name,
228                      "confidence": self._esc_charset_prober.get_confidence(),
229                      "language": self._esc_charset_prober.language,
230                  }
231                  self.done = True
232          # If we've seen high bytes (i.e., those with values greater than 127),
233          # we need to do more complicated checks using all our multi-byte and
234          # single-byte probers that are left.  The single-byte probers
235          # use character bigram distributions to determine the encoding, whereas
236          # the multi-byte probers use a combination of character unigram and
237          # bigram distributions.
238          elif self._input_state == InputState.HIGH_BYTE:
239              if not self._charset_probers:
240                  self._charset_probers = [MBCSGroupProber(self.lang_filter)]
241                  # If we're checking non-CJK encodings, use single-byte prober
242                  if self.lang_filter & LanguageFilter.NON_CJK:
243                      self._charset_probers.append(SBCSGroupProber())
244                  self._charset_probers.append(Latin1Prober())
245              for prober in self._charset_probers:
246                  if prober.feed(byte_str) == ProbingState.FOUND_IT:
247                      self.result = {
248                          "encoding": prober.charset_name,
249                          "confidence": prober.get_confidence(),
250                          "language": prober.language,
251                      }
252                      self.done = True
253                      break
254              if self.WIN_BYTE_DETECTOR.search(byte_str):
255                  self._has_win_bytes = True
256  
257      def close(self):
258          """ 
259          Stop analyzing the current document and come up with a final 
260          prediction. 
261   
262          :returns:  The ``result`` attribute, a ``dict`` with the keys 
263                     `encoding`, `confidence`, and `language`. 
264          """
265          # Don't bother with checks if we're already done
266          if self.done:
267              return self.result
268          self.done = True
269  
270          if not self._got_data:
271              self.logger.debug("no data received!")
272  
273          # Default to ASCII if it is all we've seen so far
274          elif self._input_state == InputState.PURE_ASCII:
275              self.result = {"encoding": "ascii", "confidence": 1.0, "language": ""}
276  
277          # If we have seen non-ASCII, return the best that met MINIMUM_THRESHOLD
278          elif self._input_state == InputState.HIGH_BYTE:
279              prober_confidence = None
280              max_prober_confidence = 0.0
281              max_prober = None
282              for prober in self._charset_probers:
283                  if not prober:
284                      continue
285                  prober_confidence = prober.get_confidence()
286                  if prober_confidence > max_prober_confidence:
287                      max_prober_confidence = prober_confidence
288                      max_prober = prober
289              if max_prober and (max_prober_confidence > self.MINIMUM_THRESHOLD):
290                  charset_name = max_prober.charset_name
291                  lower_charset_name = max_prober.charset_name.lower()
292                  confidence = max_prober.get_confidence()
293                  # Use Windows encoding name instead of ISO-8859 if we saw any
294                  # extra Windows-specific bytes
295                  if lower_charset_name.startswith("iso-8859"):
296                      if self._has_win_bytes:
297                          charset_name = self.ISO_WIN_MAP.get(
298                              lower_charset_name, charset_name
299                          )
300                  self.result = {
301                      "encoding": charset_name,
302                      "confidence": confidence,
303                      "language": max_prober.language,
304                  }
305  
306          # Log all prober confidences if none met MINIMUM_THRESHOLD
307          if self.logger.getEffectiveLevel() <= logging.DEBUG:
308              if self.result["encoding"] is None:
309                  self.logger.debug("no probers hit minimum threshold")
310                  for group_prober in self._charset_probers:
311                      if not group_prober:
312                          continue
313                      if isinstance(group_prober, CharSetGroupProber):
314                          for prober in group_prober.probers:
315                              self.logger.debug(
316                                  "%s %s confidence = %s",
317                                  prober.charset_name,
318                                  prober.language,
319                                  prober.get_confidence(),
320                              )
321                      else:
322                          self.logger.debug(
323                              "%s %s confidence = %s",
324                              group_prober.charset_name,
325                              group_prober.language,
326                              group_prober.get_confidence(),
327                          )
328          return self.result
