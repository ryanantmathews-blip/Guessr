70   class InstallRequirement:
71       """ 
72       Represents something that may be installed later on, may have information 
73       about where to fetch the relevant requirement and also contains logic for 
74       installing the said requirement. 
75       """

77       def __init__(
78           self,
79           req: Optional[Requirement],
80           comes_from: Optional[Union[str, "InstallRequirement"]],
81           editable: bool = False,
82           link: Optional[Link] = None,
83           markers: Optional[Marker] = None,
84           use_pep517: Optional[bool] = None,
85           isolated: bool = False,
86           install_options: Optional[List[str]] = None,
87           global_options: Optional[List[str]] = None,
88           hash_options: Optional[Dict[str, List[str]]] = None,
89           config_settings: Optional[Dict[str, str]] = None,
90           constraint: bool = False,
91           extras: Collection[str] = (),
92           user_supplied: bool = False,
93           permit_editable_wheels: bool = False,
94       ) -> None:
95           assert req is None or isinstance(req, Requirement), req
96           self.req = req
97           self.comes_from = comes_from
98           self.constraint = constraint
99           self.editable = editable
100          self.permit_editable_wheels = permit_editable_wheels
101          self.legacy_install_reason: Optional[LegacyInstallReason] = None
102  
103          # source_dir is the local directory where the linked requirement is
104          # located, or unpacked. In case unpacking is needed, creating and
105          # populating source_dir is done by the RequirementPreparer. Note this
106          # is not necessarily the directory where pyproject.toml or setup.py is
107          # located - that one is obtained via unpacked_source_directory.
108          self.source_dir: Optional[str] = None
109          if self.editable:
110              assert link
111              if link.is_file:
112                  self.source_dir = os.path.normpath(os.path.abspath(link.file_path))
113  
114          if link is None and req and req.url:
115              # PEP 508 URL requirement
116              link = Link(req.url)
117          self.link = self.original_link = link
118          self.original_link_is_in_wheel_cache = False
119  
120          # Information about the location of the artifact that was downloaded . This
121          # property is guaranteed to be set in resolver results.
122          self.download_info: Optional[DirectUrl] = None
123  
124          # Path to any downloaded or already-existing package.
125          self.local_file_path: Optional[str] = None
126          if self.link and self.link.is_file:
127              self.local_file_path = self.link.file_path
128  
129          if extras:
130              self.extras = extras
131          elif req:
132              self.extras = {safe_extra(extra) for extra in req.extras}
133          else:
134              self.extras = set()
135          if markers is None and req:
136              markers = req.marker
137          self.markers = markers
138  
139          # This holds the Distribution object if this requirement is already installed.
140          self.satisfied_by: Optional[BaseDistribution] = None
141          # Whether the installation process should try to uninstall an existing
142          # distribution before installing this requirement.
143          self.should_reinstall = False
144          # Temporary build location
145          self._temp_build_dir: Optional[TempDirectory] = None
146          # Set to True after successful installation
147          self.install_succeeded: Optional[bool] = None
148          # Supplied options
149          self.install_options = install_options if install_options else []
150          self.global_options = global_options if global_options else []
151          self.hash_options = hash_options if hash_options else {}
152          self.config_settings = config_settings
153          # Set to True after successful preparation of this requirement
154          self.prepared = False
155          # User supplied requirement are explicitly requested for installation
156          # by the user via CLI arguments or requirements files, as opposed to,
157          # e.g. dependencies, extras or constraints.
158          self.user_supplied = user_supplied
159  
160          self.isolated = isolated
161          self.build_env: BuildEnvironment = NoOpBuildEnvironment()
162  
163          # For PEP 517, the directory where we request the project metadata
164          # gets stored. We need this to pass to build_wheel, so the backend
165          # can ensure that the wheel matches the metadata (see the PEP for
166          # details).
167          self.metadata_directory: Optional[str] = None
168  
169          # The static build requirements (from pyproject.toml)
170          self.pyproject_requires: Optional[List[str]] = None
171  
172          # Build requirements that we will check are available
173          self.requirements_to_check: List[str] = []
174  
175          # The PEP 517 backend we should use to build the project
176          self.pep517_backend: Optional[Pep517HookCaller] = None
177  
178          # Are we using PEP 517 for this requirement?
179          # After pyproject.toml has been loaded, the only valid values are True
180          # and False. Before loading, None is valid (meaning "use the default").
181          # Setting an explicit value before loading pyproject.toml is supported,
182          # but after loading this flag should be treated as read only.
183          self.use_pep517 = use_pep517
184  
185          # This requirement needs more preparation before it can be built
186          self.needs_more_preparation = False

188      def __str__(self) -> str:
189          if self.req:
190              s = str(self.req)
191              if self.link:
192                  s += " from {}".format(redact_auth_from_url(self.link.url))
193          elif self.link:
194              s = redact_auth_from_url(self.link.url)
195          else:
196              s = "<InstallRequirement>"
197          if self.satisfied_by is not None:
198              s += " in {}".format(display_path(self.satisfied_by.location))
199          if self.comes_from:
200              if isinstance(self.comes_from, str):
201                  comes_from: Optional[str] = self.comes_from
202              else:
203                  comes_from = self.comes_from.from_path()
204              if comes_from:
205                  s += f" (from {comes_from})"
206          return s

213      def format_debug(self) -> str:
214          """An un-tested helper for getting state, for debugging."""
215          attributes = vars(self)
216          names = sorted(attributes)
217  
218          state = ("{}={!r}".format(attr, attributes[attr]) for attr in sorted(names))
219          return "<{name} object: {{{state}}}>".format(
220              name=self.__class__.__name__,
221              state=", ".join(state),
222          )

231      @functools.lru_cache()  # use cached_property in python 3.8+
232      def supports_pyproject_editable(self) -> bool:
233          if not self.use_pep517:
234              return False
235          assert self.pep517_backend
236          with self.build_env:
237              runner = runner_with_spinner_message(
238                  "Checking if build backend supports build_editable"
239              )
240              with self.pep517_backend.subprocess_runner(runner):
241                  return "build_editable" in self.pep517_backend._supported_features()

249          """Return whether I am pinned to an exact version. 
250   
251          For example, some-package==1.2 is pinned; some-package>1.2 is not.

256      def match_markers(self, extras_requested: Optional[Iterable[str]] = None) -> bool:
257          if not extras_requested:
258              # Provide an extra to safely evaluate the markers
259              # without matching any extra
260              extras_requested = ("",)
261          if self.markers is not None:
262              return any(
263                  self.markers.evaluate({"extra": extra}) for extra in extras_requested
264              )
265          else:
266              return True

278      def hashes(self, trust_internet: bool = True) -> Hashes:
279          """Return a hash-comparer that considers my option- and URL-based 
280          hashes to be known-good. 
281   
282          Hashes in URLs--ones embedded in the requirements file, not ones 
283          downloaded from an index server--are almost peers with ones from 
284          flags. They satisfy --require-hashes (whether it was implicitly or 
285          explicitly activated) but do not activate it. md5 and sha224 are not 
286          allowed in flags, which should nudge people toward good algos. We 
287          always OR all hashes together, even ones from URLs. 
288   
289          :param trust_internet: Whether to trust URL-based (#md5=...) hashes 
290              downloaded from the internet, as by populate_link() 
291   
292          """
293          good_hashes = self.hash_options.copy()
294          link = self.link if trust_internet else self.original_link
295          if link and link.hash:
296              good_hashes.setdefault(link.hash_name, []).append(link.hash)
297          return Hashes(good_hashes)

299      def from_path(self) -> Optional[str]:
300          """Format a nice indicator to show where this "comes from" """
301          if self.req is None:
302              return None
303          s = str(self.req)
304          if self.comes_from:
305              if isinstance(self.comes_from, str):
306                  comes_from = self.comes_from
307              else:
308                  comes_from = self.comes_from.from_path()
309              if comes_from:
310                  s += "->" + comes_from
311          return s

313      def ensure_build_location(
314          self, build_dir: str, autodelete: bool, parallel_builds: bool
315      ) -> str:
316          assert build_dir is not None
317          if self._temp_build_dir is not None:
318              assert self._temp_build_dir.path
319              return self._temp_build_dir.path
320          if self.req is None:
321              # Some systems have /tmp as a symlink which confuses custom
322              # builds (such as numpy). Thus, we ensure that the real path
323              # is returned.
324              self._temp_build_dir = TempDirectory(
325                  kind=tempdir_kinds.REQ_BUILD, globally_managed=True
326              )
327  
328              return self._temp_build_dir.path
329  
330          # This is the only remaining place where we manually determine the path
331          # for the temporary directory. It is only needed for editables where
332          # it is the value of the --src option.
333  
334          # When parallel builds are enabled, add a UUID to the build directory
335          # name so multiple builds do not interfere with each other.
336          dir_name: str = canonicalize_name(self.name)
337          if parallel_builds:
338              dir_name = f"{dir_name}_{uuid.uuid4().hex}"
339  
340          # FIXME: Is there a better place to create the build_dir? (hg and bzr
341          # need this)
342          if not os.path.exists(build_dir):
343              logger.debug("Creating directory %s", build_dir)
344              os.makedirs(build_dir)
345          actual_build_dir = os.path.join(build_dir, dir_name)
346          # `None` indicates that we respect the globally-configured deletion
347          # settings, which is what we actually want when auto-deleting.
348          delete_arg = None if autodelete else False
349          return TempDirectory(
350              path=actual_build_dir,
351              delete=delete_arg,
352              kind=tempdir_kinds.REQ_BUILD,
353              globally_managed=True,
354          ).path

356      def _set_requirement(self) -> None:
357          """Set requirement after generating metadata."""
358          assert self.req is None
359          assert self.metadata is not None
360          assert self.source_dir is not None
361  
362          # Construct a Requirement object from the generated metadata
363          if isinstance(parse_version(self.metadata["Version"]), Version):
364              op = "=="
365          else:
366              op = "==="
367  
368          self.req = Requirement(
369              "".join(
370                  [
371                      self.metadata["Name"],
372                      op,
373                      self.metadata["Version"],
374                  ]
375              )
376          )

378      def warn_on_mismatching_name(self) -> None:
379          metadata_name = canonicalize_name(self.metadata["Name"])
380          if canonicalize_name(self.req.name) == metadata_name:
381              # Everything is fine.
382              return
383  
384          # If we're here, there's a mismatch. Log a warning about it.
385          logger.warning(
386              "Generating metadata for package %s "
387              "produced metadata for project name %s. Fix your "
388              "#egg=%s fragments.",
389              self.name,
390              metadata_name,
391              self.name,
392          )
393          self.req = Requirement(metadata_name)

395      def check_if_exists(self, use_user_site: bool) -> None:
396          """Find an installed distribution that satisfies or conflicts 
397          with this requirement, and set self.satisfied_by or 
398          self.should_reinstall appropriately. 
399          """
400          if self.req is None:
401              return
402          existing_dist = get_default_environment().get_distribution(self.req.name)
403          if not existing_dist:
404              return
405  
406          version_compatible = self.req.specifier.contains(
407              existing_dist.version,
408              prereleases=True,
409          )
410          if not version_compatible:
411              self.satisfied_by = None
412              if use_user_site:
413                  if existing_dist.in_usersite:
414                      self.should_reinstall = True
415                  elif running_under_virtualenv() and existing_dist.in_site_packages:
416                      raise InstallationError(
417                          f"Will not install to the user site because it will "
418                          f"lack sys.path precedence to {existing_dist.raw_name} "
419                          f"in {existing_dist.location}"
420                      )
421              else:
422                  self.should_reinstall = True
423          else:
424              if self.editable:
425                  self.should_reinstall = True
426                  # when installing editables, nothing pre-existing should ever
427                  # satisfy
428                  self.satisfied_by = None
429              else:
430                  self.satisfied_by = existing_dist

465      def load_pyproject_toml(self) -> None:
466          """Load the pyproject.toml file. 
467   
468          After calling this routine, all of the attributes related to PEP 517 
469          processing for this requirement have been set. In particular, the 
470          use_pep517 attribute can be used to determine whether we should 
471          follow the PEP 517 or legacy (setup.py) code path. 
472          """
473          pyproject_toml_data = load_pyproject_toml(
474              self.use_pep517, self.pyproject_toml_path, self.setup_py_path, str(self)
475          )
476  
477          if pyproject_toml_data is None:
478              self.use_pep517 = False
479              return
480  
481          self.use_pep517 = True
482          requires, backend, check, backend_path = pyproject_toml_data
483          self.requirements_to_check = check
484          self.pyproject_requires = requires
485          self.pep517_backend = ConfiguredPep517HookCaller(
486              self,
487              self.unpacked_source_directory,
488              backend,
489              backend_path=backend_path,
490          )

492      def isolated_editable_sanity_check(self) -> None:
493          """Check that an editable requirement if valid for use with PEP 517/518. 
494   
495          This verifies that an editable that has a pyproject.toml either supports PEP 660 
496          or as a setup.py or a setup.cfg 
497          """
498          if (
499              self.editable
500              and self.use_pep517
501              and not self.supports_pyproject_editable()
502              and not os.path.isfile(self.setup_py_path)
503              and not os.path.isfile(self.setup_cfg_path)
504          ):
505              raise InstallationError(
506                  f"Project {self} has a 'pyproject.toml' and its build "
507                  f"backend is missing the 'build_editable' hook. Since it does not "
508                  f"have a 'setup.py' nor a 'setup.cfg', "
509                  f"it cannot be installed in editable mode. "
510                  f"Consider using a build backend that supports PEP 660."
511              )

513      def prepare_metadata(self) -> None:
514          """Ensure that project metadata is available. 
515   
516          Under PEP 517 and PEP 660, call the backend hook to prepare the metadata. 
517          Under legacy processing, call setup.py egg-info. 
518          """
519          assert self.source_dir
520          details = self.name or f"from {self.link}"
521  
522          if self.use_pep517:
523              assert self.pep517_backend is not None
524              if (
525                  self.editable
526                  and self.permit_editable_wheels
527                  and self.supports_pyproject_editable()
528              ):
529                  self.metadata_directory = generate_editable_metadata(
530                      build_env=self.build_env,
531                      backend=self.pep517_backend,
532                      details=details,
533                  )
534              else:
535                  self.metadata_directory = generate_metadata(
536                      build_env=self.build_env,
537                      backend=self.pep517_backend,
538                      details=details,
539                  )
540          else:
541              self.metadata_directory = generate_metadata_legacy(
542                  build_env=self.build_env,
543                  setup_py_path=self.setup_py_path,
544                  source_dir=self.unpacked_source_directory,
545                  isolated=self.isolated,
546                  details=details,
547              )
548  
549          # Act on the newly generated metadata, based on the name and version.
550          if not self.name:
551              self._set_requirement()
552          else:
553              self.warn_on_mismatching_name()
554  
555          self.assert_source_matches_version()

557      @property

564      def get_dist(self) -> BaseDistribution:
565          if self.metadata_directory:
566              return get_directory_distribution(self.metadata_directory)
567          elif self.local_file_path and self.is_wheel:
568              return get_wheel_distribution(
569                  FilesystemWheel(self.local_file_path), canonicalize_name(self.name)
570              )
571          raise AssertionError(
572              f"InstallRequirement {self} has no metadata directory and no wheel: "
573              f"can't make a distribution."
574          )

576      def assert_source_matches_version(self) -> None:
577          assert self.source_dir
578          version = self.metadata["version"]
579          if self.req.specifier and version not in self.req.specifier:
580              logger.warning(
581                  "Requested %s, but installing version %s",
582                  self,
583                  version,
584              )
585          else:
586              logger.debug(
587                  "Source in %s has version %s, which satisfies requirement %s",
588                  display_path(self.source_dir),
589                  version,
590                  self,
591              )

594      def ensure_has_source_dir(
595          self,
596          parent_dir: str,
597          autodelete: bool = False,
598          parallel_builds: bool = False,
599      ) -> None:
600          """Ensure that a source_dir is set. 
601   
602          This will create a temporary build dir if the name of the requirement 
603          isn't known yet. 
604   
605          :param parent_dir: The ideal pip parent_dir for the source_dir. 
606              Generally src_dir for editables and build_dir for sdists. 
607          :return: self.source_dir 
608          """
609          if self.source_dir is None:
610              self.source_dir = self.ensure_build_location(
611                  parent_dir,
612                  autodelete=autodelete,
613                  parallel_builds=parallel_builds,
614              )

616      # For editable installations
617      def update_editable(self) -> None:
618          if not self.link:
619              logger.debug(
620                  "Cannot update repository at %s; repository location is unknown",
621                  self.source_dir,
622              )
623              return
624          assert self.editable
625          assert self.source_dir
626          if self.link.scheme == "file":
627              # Static paths don't get updated
628              return
629          vcs_backend = vcs.get_backend_for_scheme(self.link.scheme)
630          # Editable requirements are validated in Requirement constructors.
631          # So here, if it's neither a path nor a valid VCS URL, it's a bug.
632          assert vcs_backend, f"Unsupported VCS URL {self.link.url}"
633          hidden_url = hide_url(self.link.url)
634          vcs_backend.obtain(self.source_dir, url=hidden_url, verbosity=0)

636      # Top-level Actions
637      def uninstall(
638          self, auto_confirm: bool = False, verbose: bool = False
639      ) -> Optional[UninstallPathSet]:
640          """ 
641          Uninstall the distribution currently satisfying this requirement. 
642   
643          Prompts before removing or modifying files unless 
644          ``auto_confirm`` is True. 
645   
646          Refuses to delete or modify files outside of ``sys.prefix`` - 
647          thus uninstallation within a virtual environment can only 
648          modify that virtual environment, even if the virtualenv is 
649          linked to global site-packages. 
650   
651          """
652          assert self.req
653          dist = get_default_environment().get_distribution(self.req.name)
654          if not dist:
655              logger.warning("Skipping %s as it is not installed.", self.name)
656              return None
657          logger.info("Found existing installation: %s", dist)
658  
659          uninstalled_pathset = UninstallPathSet.from_dist(dist)
660          uninstalled_pathset.remove(auto_confirm, verbose)
661          return uninstalled_pathset

663      def _get_archive_name(self, path: str, parentdir: str, rootdir: str) -> str:
664          def _clean_zip_name(name: str, prefix: str) -> str:
665              assert name.startswith(
666                  prefix + os.path.sep
667              ), f"name {name!r} doesn't start with prefix {prefix!r}"
668              name = name[len(prefix) + 1 :]
669              name = name.replace(os.path.sep, "/")
670              return name
671  
672          path = os.path.join(parentdir, path)
673          name = _clean_zip_name(path, rootdir)
674          return self.name + "/" + name

676      def archive(self, build_dir: Optional[str]) -> None:
677          """Saves archive to provided build_dir. 
678   
679          Used for saving downloaded VCS requirements as part of `pip download`. 
680          """
681          assert self.source_dir
682          if build_dir is None:
683              return
684  
685          create_archive = True
686          archive_name = "{}-{}.zip".format(self.name, self.metadata["version"])
687          archive_path = os.path.join(build_dir, archive_name)
688  
689          if os.path.exists(archive_path):
690              response = ask_path_exists(
691                  "The file {} exists. (i)gnore, (w)ipe, "
692                  "(b)ackup, (a)bort ".format(display_path(archive_path)),
693                  ("i", "w", "b", "a"),
694              )
695              if response == "i":
696                  create_archive = False
697              elif response == "w":
698                  logger.warning("Deleting %s", display_path(archive_path))
699                  os.remove(archive_path)
700              elif response == "b":
701                  dest_file = backup_dir(archive_path)
702                  logger.warning(
703                      "Backing up %s to %s",
704                      display_path(archive_path),
705                      display_path(dest_file),
706                  )
707                  shutil.move(archive_path, dest_file)
708              elif response == "a":
709                  sys.exit(-1)
710  
711          if not create_archive:
712              return
713  
714          zip_output = zipfile.ZipFile(
715              archive_path,
716              "w",
717              zipfile.ZIP_DEFLATED,
718              allowZip64=True,
719          )
720          with zip_output:
721              dir = os.path.normcase(os.path.abspath(self.unpacked_source_directory))
722              for dirpath, dirnames, filenames in os.walk(dir):
723                  for dirname in dirnames:
724                      dir_arcname = self._get_archive_name(
725                          dirname,
726                          parentdir=dirpath,
727                          rootdir=dir,
728                      )
729                      zipdir = zipfile.ZipInfo(dir_arcname + "/")
730                      zipdir.external_attr = 0x1ED << 16  # 0o755
731                      zip_output.writestr(zipdir, "")
732                  for filename in filenames:
733                      file_arcname = self._get_archive_name(
734                          filename,
735                          parentdir=dirpath,
736                          rootdir=dir,
737                      )
738                      filename = os.path.join(dirpath, filename)
739                      zip_output.write(filename, file_arcname)
740  
741          logger.info("Saved %s", display_path(archive_path))

743      def install(
744          self,
745          install_options: List[str],
746          global_options: Optional[Sequence[str]] = None,
747          root: Optional[str] = None,
748          home: Optional[str] = None,
749          prefix: Optional[str] = None,
750          warn_script_location: bool = True,
751          use_user_site: bool = False,
752          pycompile: bool = True,
753      ) -> None:
754          scheme = get_scheme(
755              self.name,
756              user=use_user_site,
757              home=home,
758              root=root,
759              isolated=self.isolated,
760              prefix=prefix,
761          )
762  
763          global_options = global_options if global_options is not None else []
764          if self.editable and not self.is_wheel:
765              install_editable_legacy(
766                  install_options,
767                  global_options,
768                  prefix=prefix,
769                  home=home,
770                  use_user_site=use_user_site,
771                  name=self.name,
772                  setup_py_path=self.setup_py_path,
773                  isolated=self.isolated,
774                  build_env=self.build_env,
775                  unpacked_source_directory=self.unpacked_source_directory,
776              )
777              self.install_succeeded = True
778              return
779  
780          if self.is_wheel:
781              assert self.local_file_path
782              direct_url = None
783              # TODO this can be refactored to direct_url = self.download_info
784              if self.editable:
785                  direct_url = direct_url_for_editable(self.unpacked_source_directory)
786              elif self.original_link:
787                  direct_url = direct_url_from_link(
788                      self.original_link,
789                      self.source_dir,
790                      self.original_link_is_in_wheel_cache,
791                  )
792              install_wheel(
793                  self.name,
794                  self.local_file_path,
795                  scheme=scheme,
796                  req_description=str(self.req),
797                  pycompile=pycompile,
798                  warn_script_location=warn_script_location,
799                  direct_url=direct_url,
800                  requested=self.user_supplied,
801              )
802              self.install_succeeded = True
803              return
804  
805          # TODO: Why don't we do this for editable installs?
806  
807          # Extend the list of global and install options passed on to
808          # the setup.py call with the ones from the requirements file.
809          # Options specified in requirements file override those
810          # specified on the command line, since the last option given
811          # to setup.py is the one that is used.
812          global_options = list(global_options) + self.global_options
813          install_options = list(install_options) + self.install_options
814  
815          try:
816              if (
817                  self.legacy_install_reason is not None
818                  and self.legacy_install_reason.emit_before_install
819              ):
820                  self.legacy_install_reason.emit_deprecation(self.name)
821              success = install_legacy(
822                  install_options=install_options,
823                  global_options=global_options,
824                  root=root,
825                  home=home,
826                  prefix=prefix,
827                  use_user_site=use_user_site,
828                  pycompile=pycompile,
829                  scheme=scheme,
830                  setup_py_path=self.setup_py_path,
831                  isolated=self.isolated,
832                  req_name=self.name,
833                  build_env=self.build_env,
834                  unpacked_source_directory=self.unpacked_source_directory,
835                  req_description=str(self.req),
836              )
837          except LegacyInstallFailure as exc:
838              self.install_succeeded = False
839              raise exc
840          except Exception:
841              self.install_succeeded = True
842              raise
843  
844          self.install_succeeded = success
845  
846          if (
847              success
848              and self.legacy_install_reason is not None
849              and self.legacy_install_reason.emit_after_success
850          ):
851              self.legacy_install_reason.emit_deprecation(self.name)

854  def check_invalid_constraint_type(req: InstallRequirement) -> str:
855  
856      # Check for unsupported forms
857      problem = ""
858      if not req.name:
859          problem = "Unnamed requirements are not allowed as constraints"
860      elif req.editable:
861          problem = "Editable requirements are not allowed as constraints"
862      elif req.extras:
863          problem = "Constraints cannot have extras"
864  
865      if problem:
866          deprecated(
867              reason=(
868                  "Constraints are only allowed to take the form of a package "
869                  "name and a version specifier. Other forms were originally "
870                  "permitted as an accident of the implementation, but were "
871                  "undocumented. The new implementation of the resolver no "
872                  "longer supports these forms."
873              ),
874              replacement="replacing the constraint with a requirement",
875              # No plan yet for when the new resolver becomes default
876              gone_in=None,
877              issue=8210,
878          )
879  
880      return problem

883  def _has_option(options: Values, reqs: List[InstallRequirement], option: str) -> bool:
884      if getattr(options, option, None):
885          return True
886      for req in reqs:
887          if getattr(req, option, None):
888              return True
889      return False

892  def _install_option_ignored(
893      install_options: List[str], reqs: List[InstallRequirement]
894  ) -> bool:
895      for req in reqs:
896          if (install_options or req.install_options) and not req.use_pep517:
897              return False
898      return True

907  def check_legacy_setup_py_options(
908      options: Values,
909      reqs: List[InstallRequirement],
910      mode: LegacySetupPyOptionsCheckMode,
911  ) -> None:
912      has_install_options = _has_option(options, reqs, "install_options")
913      has_build_options = _has_option(options, reqs, "build_options")
914      has_global_options = _has_option(options, reqs, "global_options")
915      legacy_setup_py_options_present = (
916          has_install_options or has_build_options or has_global_options
917      )
918      if not legacy_setup_py_options_present:
919          return
920  
921      options.format_control.disallow_binaries()
922      logger.warning(
923          "Implying --no-binary=:all: due to the presence of "
924          "--build-option / --global-option / --install-option. "
925          "Consider using --config-settings for more flexibility.",
926      )
927      if mode == LegacySetupPyOptionsCheckMode.INSTALL and has_install_options:
928          if _install_option_ignored(options.install_options, reqs):
929              logger.warning(
930                  "Ignoring --install-option when building using PEP 517",
931              )
932          else:
933              deprecated(
934                  reason=(
935                      "--install-option is deprecated because "
936                      "it forces pip to use the 'setup.py install' "
937                      "command which is itself deprecated."
938                  ),
939                  issue=11358,
940                  replacement="to use --config-settings",
941                  gone_in="23.1",
942              )
