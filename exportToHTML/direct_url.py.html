21   class DirectUrlValidationError(Exception):
22       pass

25   def _get(
26       d: Dict[str, Any], expected_type: Type[T], key: str, default: Optional[T] = None
27   ) -> Optional[T]:
28       """Get value from dictionary and verify expected type."""
29       if key not in d:
30           return default
31       value = d[key]
32       if not isinstance(value, expected_type):
33           raise DirectUrlValidationError(
34               "{!r} has unexpected type for {} (expected {})".format(
35                   value, key, expected_type
36               )
37           )
38       return value
39   
40   
41   def _get_required(
42       d: Dict[str, Any], expected_type: Type[T], key: str, default: Optional[T] = None
43   ) -> T:
44       value = _get(d, expected_type, key, default)
45       if value is None:
46           raise DirectUrlValidationError(f"{key} must have a value")
47       return value

50   def _exactly_one_of(infos: Iterable[Optional["InfoType"]]) -> "InfoType":
51       infos = [info for info in infos if info is not None]
52       if not infos:
53           raise DirectUrlValidationError(
54               "missing one of archive_info, dir_info, vcs_info"
55           )
56       if len(infos) > 1:
57           raise DirectUrlValidationError(
58               "more than one of archive_info, dir_info, vcs_info"
59           )
60       assert infos[0] is not None
61       return infos[0]
62

64   def _filter_none(**kwargs: Any) -> Dict[str, Any]:
65       """Make dict excluding None values."""
66       return {k: v for k, v in kwargs.items() if v is not None}

69   class VcsInfo:
70       name = "vcs_info"
71   
72       def __init__(
73           self,
74           vcs: str,
75           commit_id: str,
76           requested_revision: Optional[str] = None,
77       ) -> None:
78           self.vcs = vcs
79           self.requested_revision = requested_revision
80           self.commit_id = commit_id
81   
82       @classmethod
83       def _from_dict(cls, d: Optional[Dict[str, Any]]) -> Optional["VcsInfo"]:
84           if d is None:
85               return None
86           return cls(
87               vcs=_get_required(d, str, "vcs"),
88               commit_id=_get_required(d, str, "commit_id"),
89               requested_revision=_get(d, str, "requested_revision"),
90           )
91   
92       def _to_dict(self) -> Dict[str, Any]:
93           return _filter_none(
94               vcs=self.vcs,
95               requested_revision=self.requested_revision,
96               commit_id=self.commit_id,
97           )

141  class DirectUrl:
142      def __init__(
143          self,
144          url: str,
145          info: InfoType,
146          subdirectory: Optional[str] = None,
147      ) -> None:
148          self.url = url
149          self.info = info
150          self.subdirectory = subdirectory
151  
152      def _remove_auth_from_netloc(self, netloc: str) -> str:
153          if "@" not in netloc:
154              return netloc
155          user_pass, netloc_no_user_pass = netloc.split("@", 1)
156          if (
157              isinstance(self.info, VcsInfo)
158              and self.info.vcs == "git"
159              and user_pass == "git"
160          ):
161              return netloc
162          if ENV_VAR_RE.match(user_pass):
163              return netloc
164          return netloc_no_user_pass
165  
166      @property
167      def redacted_url(self) -> str:
168          """url with user:password part removed unless it is formed with 
169          environment variables as specified in PEP 610, or it is ``git`` 
170          in the case of a git URL. 
171          """
172          purl = urllib.parse.urlsplit(self.url)
173          netloc = self._remove_auth_from_netloc(purl.netloc)
174          surl = urllib.parse.urlunsplit(
175              (purl.scheme, netloc, purl.path, purl.query, purl.fragment)
176          )
177          return surl
178  
179      def validate(self) -> None:
180          self.from_dict(self.to_dict())
181  
182      @classmethod
183      def from_dict(cls, d: Dict[str, Any]) -> "DirectUrl":
184          return DirectUrl(
185              url=_get_required(d, str, "url"),
186              subdirectory=_get(d, str, "subdirectory"),
187              info=_exactly_one_of(
188                  [
189                      ArchiveInfo._from_dict(_get(d, dict, "archive_info")),
190                      DirInfo._from_dict(_get(d, dict, "dir_info")),
191                      VcsInfo._from_dict(_get(d, dict, "vcs_info")),
192                  ]
193              ),
194          )
195  
196      def to_dict(self) -> Dict[str, Any]:
197          res = _filter_none(
198              url=self.redacted_url,
199              subdirectory=self.subdirectory,
200          )
201          res[self.info.name] = self.info._to_dict()
202          return res
203  
204      @classmethod
205      def from_json(cls, s: str) -> "DirectUrl":
206          return cls.from_dict(json.loads(s))
207  
208      def to_json(self) -> str:
209          return json.dumps(self.to_dict(), sort_keys=True)
210  
211      def is_local_editable(self) -> bool:
212          return isinstance(self.info, DirInfo) and self.info.editable
