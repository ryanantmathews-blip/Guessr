_in_process.py 
1    """This is invoked in a subprocess to call the build backend hooks. 
2     
3    It expects: 
4    - Command line args: hook_name, control_dir 
5    - Environment variables: 
6          PEP517_BUILD_BACKEND=entry.point:spec 
7          PEP517_BACKEND_PATH=paths (separated with os.pathsep) 
8    - control_dir/input.json: 
9      - {"kwargs": {...}} 
10    
11   Results: 
12   - control_dir/output.json 
13     - {"return_val": ...} 
14   """
15   import json
16   import os
17   import os.path
18   import re
19   import shutil
20   import sys
21   import traceback
22   from glob import glob
23   from importlib import import_module
24   from os.path import join as pjoin

30   def write_json(obj, path, **kwargs):
31       with open(path, 'w', encoding='utf-8') as f:
32           json.dump(obj, f, **kwargs)
33   
34   
35   def read_json(path):
36       with open(path, encoding='utf-8') as f:
37           return json.load(f)

40   class BackendUnavailable(Exception):
41       """Raised if we cannot import the backend"""
42       def __init__(self, traceback):
43           self.traceback = traceback

48       def __init__(self, message):

52   class HookMissing(Exception):
53       """Raised if a hook is missing and we are not executing the fallback"""
54       def __init__(self, hook_name=None):
55           super().__init__(hook_name)
56           self.hook_name = hook_name

59   def contained_in(filename, directory):
60       """Test if a file is located within the given directory."""
61       filename = os.path.normcase(os.path.abspath(filename))
62       directory = os.path.normcase(os.path.abspath(directory))
63       return os.path.commonprefix([filename, directory]) == directory

66   def _build_backend():
67       """Find and load the build backend"""
68       # Add in-tree backend directories to the front of sys.path.
69       backend_path = os.environ.get('PEP517_BACKEND_PATH')
70       if backend_path:
71           extra_pathitems = backend_path.split(os.pathsep)
72           sys.path[:0] = extra_pathitems
73   
74       ep = os.environ['PEP517_BUILD_BACKEND']
75       mod_path, _, obj_path = ep.partition(':')
76       try:
77           obj = import_module(mod_path)
78       except ImportError:
79           raise BackendUnavailable(traceback.format_exc())
80   
81       if backend_path:
82           if not any(
83               contained_in(obj.__file__, path)
84               for path in extra_pathitems
85           ):
86               raise BackendInvalid("Backend was not loaded from backend-path")
87   
88       if obj_path:
89           for path_part in obj_path.split('.'):
90               obj = getattr(obj, path_part)
91       return obj

94   def _supported_features():
95       """Return the list of options features supported by the backend. 
96    
97       Returns a list of strings. 
98       The only possible value is 'build_editable'. 
99       """
100      backend = _build_backend()
101      features = []
102      if hasattr(backend, "build_editable"):
103          features.append("build_editable")
104      return features

112      backend = _build_backend()

135  def prepare_metadata_for_build_wheel(
136          metadata_directory, config_settings, _allow_fallback):
137      """Invoke optional prepare_metadata_for_build_wheel 
138   
139      Implements a fallback by building a wheel if the hook isn't defined, 
140      unless _allow_fallback is False in which case HookMissing is raised. 
141      """
142      backend = _build_backend()
143      try:
144          hook = backend.prepare_metadata_for_build_wheel
145      except AttributeError:
146          if not _allow_fallback:
147              raise HookMissing()
148          whl_basename = backend.build_wheel(metadata_directory, config_settings)
149          return _get_wheel_metadata_from_wheel(whl_basename, metadata_directory,
150                                                config_settings)
151      else:
152          return hook(metadata_directory, config_settings)

155  def prepare_metadata_for_build_editable(
156          metadata_directory, config_settings, _allow_fallback):
157      """Invoke optional prepare_metadata_for_build_editable 
158   
159      Implements a fallback by building an editable wheel if the hook isn't 
160      defined, unless _allow_fallback is False in which case HookMissing is 
161      raised. 
162      """
163      backend = _build_backend()
164      try:
165          hook = backend.prepare_metadata_for_build_editable
166      except AttributeError:
167          if not _allow_fallback:
168              raise HookMissing()
169          try:
170              build_hook = backend.build_editable
171          except AttributeError:
172              raise HookMissing(hook_name='build_editable')
173          else:
174              whl_basename = build_hook(metadata_directory, config_settings)
175              return _get_wheel_metadata_from_wheel(whl_basename,
176                                                    metadata_directory,
177                                                    config_settings)
178      else:
179          return hook(metadata_directory, config_settings)

182  WHEEL_BUILT_MARKER = 'PEP517_ALREADY_BUILT_WHEEL'

185  def _dist_info_files(whl_zip):
186      """Identify the .dist-info folder inside a wheel ZipFile."""
187      res = []
188      for path in whl_zip.namelist():
189          m = re.match(r'[^/\\]+-[^/\\]+\.dist-info/', path)
190          if m:
191              res.append(path)
192      if res:
193          return res
194      raise Exception("No .dist-info folder found in wheel")

197  def _get_wheel_metadata_from_wheel(
198          whl_basename, metadata_directory, config_settings):
199      """Extract the metadata from a wheel. 
200   
201      Fallback for when the build backend does not 
202      define the 'get_wheel_metadata' hook. 
203      """
204      from zipfile import ZipFile
205      with open(os.path.join(metadata_directory, WHEEL_BUILT_MARKER), 'wb'):
206          pass  # Touch marker file
207  
208      whl_file = os.path.join(metadata_directory, whl_basename)
209      with ZipFile(whl_file) as zipf:
210          dist_info = _dist_info_files(zipf)
211          zipf.extractall(path=metadata_directory, members=dist_info)
212      return dist_info[0].split('/')[0]

215  def _find_already_built_wheel(metadata_directory):
216      """Check for a wheel already built during the get_wheel_metadata hook. 
217      """
218      if not metadata_directory:
219          return None
220      metadata_parent = os.path.dirname(metadata_directory)
221      if not os.path.isfile(pjoin(metadata_parent, WHEEL_BUILT_MARKER)):
222          return None
223  
224      whl_files = glob(os.path.join(metadata_parent, '*.whl'))
225      if not whl_files:
226          print('Found wheel built marker, but no .whl files')
227          return None
228      if len(whl_files) > 1:
229          print('Found multiple .whl files; unspecified behaviour. '
230                'Will call build_wheel.')
231          return None
232  
233      # Exactly one .whl file
234      return whl_files[0]

237  def build_wheel(wheel_directory, config_settings, metadata_directory=None):
238      """Invoke the mandatory build_wheel hook. 
239   
240      If a wheel was already built in the 
241      prepare_metadata_for_build_wheel fallback, this 
242      will copy it rather than rebuilding the wheel. 
243      """
244      prebuilt_whl = _find_already_built_wheel(metadata_directory)
245      if prebuilt_whl:
246          shutil.copy2(prebuilt_whl, wheel_directory)
247          return os.path.basename(prebuilt_whl)
248  
249      return _build_backend().build_wheel(wheel_directory, config_settings,
250                                          metadata_directory)

253  def build_editable(wheel_directory, config_settings, metadata_directory=None):
254      """Invoke the optional build_editable hook. 
255   
256      If a wheel was already built in the 
257      prepare_metadata_for_build_editable fallback, this 
258      will copy it rather than rebuilding the wheel. 
259      """
260      backend = _build_backend()
261      try:
262          hook = backend.build_editable
263      except AttributeError:
264          raise HookMissing()
265      else:
266          prebuilt_whl = _find_already_built_wheel(metadata_directory)
267          if prebuilt_whl:
268              shutil.copy2(prebuilt_whl, wheel_directory)
269              return os.path.basename(prebuilt_whl)
270  
271          return hook(wheel_directory, config_settings, metadata_directory)

292  class GotUnsupportedOperation(Exception):
293      """For internal use when backend raises UnsupportedOperation"""
294      def __init__(self, traceback):
295          self.traceback = traceback

298  def build_sdist(sdist_directory, config_settings):
299      """Invoke the mandatory build_sdist hook."""
300      backend = _build_backend()
301      try:
302          return backend.build_sdist(sdist_directory, config_settings)
303      except getattr(backend, 'UnsupportedOperation', _DummyException):
304          raise GotUnsupportedOperation(traceback.format_exc())

307  HOOK_NAMES = {
308      'get_requires_for_build_wheel',
309      'prepare_metadata_for_build_wheel',
310      'build_wheel',
311      'get_requires_for_build_editable',
312      'prepare_metadata_for_build_editable',
313      'build_editable',
314      'get_requires_for_build_sdist',
315      'build_sdist',
316      '_supported_features',
317  }

320  def main():
321      if len(sys.argv) < 3:
322          sys.exit("Needs args: hook_name, control_dir")
323      hook_name = sys.argv[1]
324      control_dir = sys.argv[2]
325      if hook_name not in HOOK_NAMES:
326          sys.exit("Unknown hook: %s" % hook_name)
327      hook = globals()[hook_name]
328  
329      hook_input = read_json(pjoin(control_dir, 'input.json'))
330  
331      json_out = {'unsupported': False, 'return_val': None}
332      try:
333          json_out['return_val'] = hook(**hook_input['kwargs'])
334      except BackendUnavailable as e:
335          json_out['no_backend'] = True
336          json_out['traceback'] = e.traceback
337      except BackendInvalid as e:
338          json_out['backend_invalid'] = True
339          json_out['backend_error'] = e.message
340      except GotUnsupportedOperation as e:
341          json_out['unsupported'] = True
342          json_out['traceback'] = e.traceback
343      except HookMissing as e:
344          json_out['hook_missing'] = True
345          json_out['missing_hook_name'] = e.hook_name or hook_name
346  
347      write_json(json_out, pjoin(control_dir, 'output.json'), indent=2)
348  
349  
350  if __name__ == '__main__':
351      main()
