40   def auto(
41       cls: Optional[Type[T]] = None, *, angular: Optional[bool] = None
42   ) -> Union[Type[T], Callable[[Type[T]], Type[T]]]:
43       """Class decorator to create __repr__ from __rich_repr__"""
44   
45       def do_replace(cls: Type[T], angular: Optional[bool] = None) -> Type[T]:
46           def auto_repr(self: T) -> str:
47               """Create repr string from __rich_repr__"""
48               repr_str: List[str] = []
49               append = repr_str.append
50   
51               angular: bool = getattr(self.__rich_repr__, "angular", False)  # type: ignore[attr-defined]
52               for arg in self.__rich_repr__():  # type: ignore[attr-defined]
53                   if isinstance(arg, tuple):
54                       if len(arg) == 1:
55                           append(repr(arg[0]))
56                       else:
57                           key, value, *default = arg
58                           if key is None:
59                               append(repr(value))
60                           else:
61                               if len(default) and default[0] == value:
62                                   continue
63                               append(f"{key}={value!r}")
64                   else:
65                       append(repr(arg))
66               if angular:
67                   return f"<{self.__class__.__name__} {' '.join(repr_str)}>"
68               else:
69                   return f"{self.__class__.__name__}({', '.join(repr_str)})"
70   
71           def auto_rich_repr(self: Type[T]) -> Result:
72               """Auto generate __rich_rep__ from signature of __init__"""
73               try:
74                   signature = inspect.signature(self.__init__)
75                   for name, param in signature.parameters.items():
76                       if param.kind == param.POSITIONAL_ONLY:
77                           yield getattr(self, name)
78                       elif param.kind in (
79                           param.POSITIONAL_OR_KEYWORD,
80                           param.KEYWORD_ONLY,
81                       ):
82                           if param.default == param.empty:
83                               yield getattr(self, param.name)
84                           else:
85                               yield param.name, getattr(self, param.name), param.default
86               except Exception as error:
87                   raise ReprError(
88                       f"Failed to auto generate __rich_repr__; {error}"
89                   ) from None
90   
91           if not hasattr(cls, "__rich_repr__"):
92               auto_rich_repr.__doc__ = "Build a rich repr"
93               cls.__rich_repr__ = auto_rich_repr  # type: ignore[attr-defined]
94   
95           auto_repr.__doc__ = "Return repr(self)"
96           cls.__repr__ = auto_repr  # type: ignore[assignment]
97           if angular is not None:
98               cls.__rich_repr__.angular = angular  # type: ignore[attr-defined]
99           return cls
100  
101      if cls is None:
102          return partial(do_replace, angular=angular)
103      else:
104          return do_replace(cls, angular=angular)
