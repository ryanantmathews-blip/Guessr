16   class StreamWrapper(object):
17       ''' 
18       Wraps a stream (such as stdout), acting as a transparent proxy for all 
19       attribute access apart from method 'write()', which is delegated to our 
20       Converter instance. 
21       '''
22       def __init__(self, wrapped, converter):
23           # double-underscore everything to prevent clashes with names of
24           # attributes on the wrapped stream object.
25           self.__wrapped = wrapped
26           self.__convertor = converter
27   
28       def __getattr__(self, name):
29           return getattr(self.__wrapped, name)
30   
31       def __enter__(self, *args, **kwargs):
32           # special method lookup bypasses __getattr__/__getattribute__, see
33           # https://stackoverflow.com/questions/12632894/why-doesnt-getattr-work-with-exit
34           # thus, contextlib magic methods are not proxied via __getattr__
35           return self.__wrapped.__enter__(*args, **kwargs)
36   
37       def __exit__(self, *args, **kwargs):
38           return self.__wrapped.__exit__(*args, **kwargs)
39   
40       def __setstate__(self, state):
41           self.__dict__ = state
42   
43       def __getstate__(self):
44           return self.__dict__
45   
46       def write(self, text):
47           self.__convertor.write(text)
48   
49       def isatty(self):
50           stream = self.__wrapped
51           if 'PYCHARM_HOSTED' in os.environ:
52               if stream is not None and (stream is sys.__stdout__ or stream is sys.__stderr__):
53                   return True
54           try:
55               stream_isatty = stream.isatty
56           except AttributeError:
57               return False
58           else:
59               return stream_isatty()
60   
61       @property
62       def closed(self):
63           stream = self.__wrapped
64           try:
65               return stream.closed
66           # AttributeError in the case that the stream doesn't support being closed
67           # ValueError for the case that the stream has already been detached when atexit runs
68           except (AttributeError, ValueError):
69               return True

72   class AnsiToWin32(object):
73       ''' 
74       Implements a 'write()' method which, on Windows, will strip ANSI character 
75       sequences from the text, and if outputting to a tty, will convert them into 
76       win32 function calls. 
77       '''
78       ANSI_CSI_RE = re.compile('\001?\033\\[((?:\\d|;)*)([a-zA-Z])\002?')   # Control Sequence Introducer
79       ANSI_OSC_RE = re.compile('\001?\033\\]([^\a]*)(\a)\002?')             # Operating System Command
80   
81       def __init__(self, wrapped, convert=None, strip=None, autoreset=False):
82           # The wrapped stream (normally sys.stdout or sys.stderr)
83           self.wrapped = wrapped
84   
85           # should we reset colors to defaults after every .write()
86           self.autoreset = autoreset
87   
88           # create the proxy wrapping our output stream
89           self.stream = StreamWrapper(wrapped, self)
90   
91           on_windows = os.name == 'nt'
92           # We test if the WinAPI works, because even if we are on Windows
93           # we may be using a terminal that doesn't support the WinAPI
94           # (e.g. Cygwin Terminal). In this case it's up to the terminal
95           # to support the ANSI codes.
96           conversion_supported = on_windows and winapi_test()
97   
98           # should we strip ANSI sequences from our output?
99           if strip is None:
100              strip = conversion_supported or (not self.stream.closed and not self.stream.isatty())
101          self.strip = strip
102  
103          # should we should convert ANSI sequences into win32 calls?
104          if convert is None:
105              convert = conversion_supported and not self.stream.closed and self.stream.isatty()
106          self.convert = convert
107  
108          # dict of ansi codes to win32 functions and parameters
109          self.win32_calls = self.get_win32_calls()
110  
111          # are we wrapping stderr?
112          self.on_stderr = self.wrapped is sys.stderr
113  
114      def should_wrap(self):
115          ''' 
116          True if this class is actually needed. If false, then the output 
117          stream will not be affected, nor will win32 calls be issued, so 
118          wrapping stdout is not actually required. This will generally be 
119          False on non-Windows platforms, unless optional functionality like 
120          autoreset has been requested using kwargs to init() 
121          '''
122          return self.convert or self.strip or self.autoreset
123

168      def write(self, text):
169          if self.strip or self.convert:
170              self.write_and_convert(text)
171          else:
172              self.wrapped.write(text)
173              self.wrapped.flush()
174          if self.autoreset:
175              self.reset_all()
176

178      def reset_all(self):
179          if self.convert:
180              self.call_win32('m', (0,))
181          elif not self.strip and not self.stream.closed:
182              self.wrapped.write(Style.RESET_ALL)

185      def write_and_convert(self, text):
186          ''' 
187          Write the given text to our wrapped stream, stripping any ANSI 
188          sequences from the text, and optionally converting them into win32 
189          calls. 
190          '''
191          cursor = 0
192          text = self.convert_osc(text)
193          for match in self.ANSI_CSI_RE.finditer(text):
194              start, end = match.span()
195              self.write_plain_text(text, cursor, start)
196              self.convert_ansi(*match.groups())
197              cursor = end
198          self.write_plain_text(text, cursor, len(text))
199  
200  
201      def write_plain_text(self, text, start, end):
202          if start < end:
203              self.wrapped.write(text[start:end])
204              self.wrapped.flush()

207      def convert_ansi(self, paramstring, command):
208          if self.convert:
209              params = self.extract_params(command, paramstring)
210              self.call_win32(command, params)
211

213      def extract_params(self, command, paramstring):
214          if command in 'Hf':
215              params = tuple(int(p) if len(p) != 0 else 1 for p in paramstring.split(';'))
216              while len(params) < 2:
217                  # defaults:
218                  params = params + (1,)
219          else:
220              params = tuple(int(p) for p in paramstring.split(';') if len(p) != 0)
221              if len(params) == 0:
222                  # defaults:
223                  if command in 'JKm':
224                      params = (0,)
225                  elif command in 'ABCD':
226                      params = (1,)
227  
228          return params
229  
230  
231      def call_win32(self, command, params):
232          if command == 'm':
233              for param in params:
234                  if param in self.win32_calls:
235                      func_args = self.win32_calls[param]
236                      func = func_args[0]
237                      args = func_args[1:]
238                      kwargs = dict(on_stderr=self.on_stderr)
239                      func(*args, **kwargs)
240          elif command in 'J':
241              winterm.erase_screen(params[0], on_stderr=self.on_stderr)
242          elif command in 'K':
243              winterm.erase_line(params[0], on_stderr=self.on_stderr)
244          elif command in 'Hf':     # cursor position - absolute
245              winterm.set_cursor_position(params, on_stderr=self.on_stderr)
246          elif command in 'ABCD':   # cursor position - relative
247              n = params[0]
248              # A - up, B - down, C - forward, D - back
249              x, y = {'A': (0, -n), 'B': (0, n), 'C': (n, 0), 'D': (-n, 0)}[command]
250              winterm.cursor_adjust(x, y, on_stderr=self.on_stderr)

253      def convert_osc(self, text):
254          for match in self.ANSI_OSC_RE.finditer(text):
255              start, end = match.span()
256              text = text[:start] + text[end:]
257              paramstring, command = match.groups()
258              if command == BEL:
259                  if paramstring.count(";") == 1:
260                      params = paramstring.split(";")
261                      # 0 - change title and icon (we will only change title)
262                      # 1 - change icon (we don't support this)
263                      # 2 - change title
264                      if params[0] in '02':
265                          winterm.set_title(params[1])
266          return text
