26   def find_ordinal(pos_num: int) -> str:
27       # See: https://en.wikipedia.org/wiki/English_numerals#Ordinal_numbers
28       if pos_num == 0:
29           return "th"
30       elif pos_num == 1:
31           return "st"
32       elif pos_num == 2:
33           return "nd"
34       elif pos_num == 3:
35           return "rd"
36       elif 4 <= pos_num <= 20:
37           return "th"
38       else:
39           return find_ordinal(pos_num % 10)

42   def to_ordinal(pos_num: int) -> str:
43       return f"{pos_num}{find_ordinal(pos_num)}"

46   def get_callback_name(cb: typing.Callable[..., typing.Any]) -> str:
47       """Get a callback fully-qualified name. 
48    
49       If no name can be produced ``repr(cb)`` is called and returned. 
50       """
51       segments = []
52       try:
53           segments.append(cb.__qualname__)
54       except AttributeError:
55           try:
56               segments.append(cb.__name__)
57           except AttributeError:
58               pass
59       if not segments:
60           return repr(cb)
61       else:
62           try:
63               # When running under sphinx it appears this can be none?
64               if cb.__module__:
65                   segments.insert(0, cb.__module__)
66           except AttributeError:
67               pass
68           return ".".join(segments)
