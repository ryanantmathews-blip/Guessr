6    def dir_to_zipfile(root):

8        buffer = io.BytesIO()

12               fs_path = os.path.join(root, path)
13               rel_path = os.path.relpath(fs_path, root)
14               zip_file.writestr(rel_path + '/', '')
15           for path in files:
16               fs_path = os.path.join(root, path)
17               rel_path = os.path.relpath(fs_path, root)
18               zip_file.write(fs_path, rel_path)
