78   def parse_bdist_wininst(name):
79       """Return (base,pyversion) or (None,None) for possible .exe name"""
80   
81       lower = name.lower()
82       base, py_ver, plat = None, None, None
83   
84       if lower.endswith('.exe'):
85           if lower.endswith('.win32.exe'):
86               base = name[:-10]
87               plat = 'win32'
88           elif lower.startswith('.win32-py', -16):
89               py_ver = name[-7:-4]
90               base = name[:-16]
91               plat = 'win32'
92           elif lower.endswith('.win-amd64.exe'):
93               base = name[:-14]
94               plat = 'win-amd64'
95           elif lower.startswith('.win-amd64-py', -20):
96               py_ver = name[-7:-4]
97               base = name[:-20]
98               plat = 'win-amd64'
99       return base, py_ver, plat

102  def egg_info_for_url(url):
103      parts = urllib.parse.urlparse(url)
104      scheme, server, path, parameters, query, fragment = parts
105      base = urllib.parse.unquote(path.split('/')[-1])
106      if server == 'sourceforge.net' and base == 'download':  # XXX Yuck
107          base = urllib.parse.unquote(path.split('/')[-2])
108      if '#' in base:
109          base, fragment = base.split('#', 1)
110      return base, fragment

113  def distros_for_url(url, metadata=None):
114      """Yield egg or source distribution objects that might be found at a URL"""
115      base, fragment = egg_info_for_url(url)
116      for dist in distros_for_location(url, base, metadata):
117          yield dist
118      if fragment:
119          match = EGG_FRAGMENT.match(fragment)
120          if match:
121              for dist in interpret_distro_name(
122                  url, match.group(1), metadata, precedence=CHECKOUT_DIST
123              ):
124                  yield dist

127  def distros_for_location(location, basename, metadata=None):
128      """Yield egg or source distribution objects based on basename"""
129      if basename.endswith('.egg.zip'):
130          basename = basename[:-4]  # strip the .zip
131      if basename.endswith('.egg') and '-' in basename:
132          # only one, unambiguous interpretation
133          return [Distribution.from_location(location, basename, metadata)]
134      if basename.endswith('.whl') and '-' in basename:
135          wheel = Wheel(basename)
136          if not wheel.is_compatible():
137              return []
138          return [
139              Distribution(
140                  location=location,
141                  project_name=wheel.project_name,
142                  version=wheel.version,
143                  # Increase priority over eggs.
144                  precedence=EGG_DIST + 1,
145              )
146          ]
147      if basename.endswith('.exe'):
148          win_base, py_ver, platform = parse_bdist_wininst(basename)
149          if win_base is not None:
150              return interpret_distro_name(
151                  location, win_base, metadata, py_ver, BINARY_DIST, platform
152              )
153      # Try source distro extensions (.zip, .tgz, etc.)
154      #
155      for ext in EXTENSIONS:
156          if basename.endswith(ext):
157              basename = basename[: -len(ext)]
158              return interpret_distro_name(location, basename, metadata)
159      return []  # no extension matched

162  def distros_for_filename(filename, metadata=None):

164      return distros_for_location(

169  def interpret_distro_name(
170      location, basename, metadata, py_version=None, precedence=SOURCE_DIST, platform=None
171  ):
172      """Generate alternative interpretations of a source distro name 
173   
174      Note: if `location` is a filesystem filename, you should call 
175      ``pkg_resources.normalize_path()`` on it before passing it to this 
176      routine! 
177      """
178      # Generate alternative interpretations of a source distro name
179      # Because some packages are ambiguous as to name/versions split
180      # e.g. "adns-python-1.1.0", "egenix-mx-commercial", etc.
181      # So, we generate each possible interpretation (e.g. "adns, python-1.1.0"
182      # "adns-python, 1.1.0", and "adns-python-1.1.0, no version").  In practice,
183      # the spurious interpretations should be ignored, because in the event
184      # there's also an "adns" package, the spurious "python-1.1.0" version will
185      # compare lower than any numeric version number, and is therefore unlikely
186      # to match a request for it.  It's still a potential problem, though, and
187      # in the long run PyPI and the distutils should go for "safe" names and
188      # versions in distribution archive names (sdist and bdist).
189  
190      parts = basename.split('-')
191      if not py_version and any(re.match(r'py\d\.\d$', p) for p in parts[2:]):
192          # it is a bdist_dumb, not an sdist -- bail out
193          return
194  
195      for p in range(1, len(parts) + 1):
196          yield Distribution(
197              location,
198              metadata,
199              '-'.join(parts[:p]),
200              '-'.join(parts[p:]),
201              py_version=py_version,
202              precedence=precedence,
203              platform=platform,
204          )

207  def unique_values(func):
208      """ 
209      Wrap a function returning an iterable such that the resulting iterable 
210      only ever yields unique items. 
211      """
212  
213      @wraps(func)
214      def wrapper(*args, **kwargs):
215          return unique_everseen(func(*args, **kwargs))
216  
217      return wrapper

226  @unique_values
227  def find_external_links(url, page):
228      """Find rel="homepage" and rel="download" links in `page`, yielding URLs"""
229  
230      for match in REL.finditer(page):
231          tag, rel = match.groups()
232          rels = set(map(str.strip, rel.lower().split(',')))
233          if 'homepage' in rels or 'download' in rels:
234              for match in HREF.finditer(tag):
235                  yield urllib.parse.urljoin(url, htmldecode(match.group(1)))
236  
237      for tag in ("<th>Home Page", "<th>Download URL"):
238          pos = page.find(tag)
239          if pos != -1:
240              match = HREF.search(page, pos)
241              if match:
242                  yield urllib.parse.urljoin(url, htmldecode(match.group(1)))

270  class HashChecker(ContentChecker):
271      pattern = re.compile(
272          r'(?P<hash_name>sha1|sha224|sha384|sha256|sha512|md5)='
273          r'(?P<expected>[a-f0-9]+)'
274      )
275  
276      def __init__(self, hash_name, expected):
277          self.hash_name = hash_name
278          self.hash = hashlib.new(hash_name)
279          self.expected = expected
280  
281      @classmethod
282      def from_url(cls, url):
283          "Construct a (possibly null) ContentChecker from a URL"
284          fragment = urllib.parse.urlparse(url)[-1]
285          if not fragment:
286              return ContentChecker()
287          match = cls.pattern.search(fragment)
288          if not match:
289              return ContentChecker()
290          return cls(**match.groupdict())
291  
292      def feed(self, block):
293          self.hash.update(block)
294  
295      def is_valid(self):
296          return self.hash.hexdigest() == self.expected
297  
298      def report(self, reporter, template):
299          msg = template % self.hash_name
300          return reporter(msg)

304      """A distribution index that scans web pages for download URLs"""

306      def __init__(
307          self,
308          index_url="https://pypi.org/simple/",
309          hosts=('*',),
310          ca_bundle=None,
311          verify_ssl=True,
312          *args,
313          **kw
314      ):
315          super().__init__(*args, **kw)
316          self.index_url = index_url + "/"[: not index_url.endswith('/')]
317          self.scanned_urls = {}
318          self.fetched_urls = {}
319          self.package_pages = {}
320          self.allows = re.compile('|'.join(map(translate, hosts))).match
321          self.to_scan = []
322          self.opener = urllib.request.urlopen

324      def add(self, dist):
325          # ignore invalid versions
326          try:
327              parse_version(dist.version)
328          except Exception:
329              return
330          return super().add(dist)

332      # FIXME: 'PackageIndex.process_url' is too complex (14)
333      def process_url(self, url, retrieve=False):  # noqa: C901
334          """Evaluate a URL as a possible download, and maybe retrieve it"""
335          if url in self.scanned_urls and not retrieve:
336              return
337          self.scanned_urls[url] = True
338          if not URL_SCHEME(url):
339              self.process_filename(url)
340              return
341          else:
342              dists = list(distros_for_url(url))
343              if dists:
344                  if not self.url_ok(url):
345                      return
346                  self.debug("Found link: %s", url)
347  
348          if dists or not retrieve or url in self.fetched_urls:
349              list(map(self.add, dists))
350              return  # don't need the actual page
351  
352          if not self.url_ok(url):
353              self.fetched_urls[url] = True
354              return
355  
356          self.info("Reading %s", url)
357          self.fetched_urls[url] = True  # prevent multiple fetch attempts
358          tmpl = "Download error on %s: %%s -- Some packages may not be found!"
359          f = self.open_url(url, tmpl % url)
360          if f is None:
361              return
362          if isinstance(f, urllib.error.HTTPError) and f.code == 401:
363              self.info("Authentication error: %s" % f.msg)
364          self.fetched_urls[f.url] = True
365          if 'html' not in f.headers.get('content-type', '').lower():
366              f.close()  # not html, we can't process it
367              return
368  
369          base = f.url  # handle redirects
370          page = f.read()
371          if not isinstance(page, str):
372              # In Python 3 and got bytes but want str.
373              if isinstance(f, urllib.error.HTTPError):
374                  # Errors have no charset, assume latin1:
375                  charset = 'latin-1'
376              else:
377                  charset = f.headers.get_param('charset') or 'latin-1'
378              page = page.decode(charset, "ignore")
379          f.close()
380          for match in HREF.finditer(page):
381              link = urllib.parse.urljoin(base, htmldecode(match.group(1)))
382              self.process_url(link)
383          if url.startswith(self.index_url) and getattr(f, 'code', None) != 404:
384              page = self.process_index(url, page)
385  
386      def process_filename(self, fn, nested=False):
387          # process filenames or directories
388          if not os.path.exists(fn):
389              self.warn("Not found: %s", fn)
390              return
391  
392          if os.path.isdir(fn) and not nested:
393              path = os.path.realpath(fn)
394              for item in os.listdir(path):
395                  self.process_filename(os.path.join(path, item), True)
396  
397          dists = distros_for_filename(fn)
398          if dists:
399              self.debug("Found: %s", fn)
400              list(map(self.add, dists))

402      def url_ok(self, url, fatal=False):
403          s = URL_SCHEME(url)
404          is_file = s and s.group(1).lower() == 'file'
405          if is_file or self.allows(urllib.parse.urlparse(url)[1]):
406              return True
407          msg = (
408              "\nNote: Bypassing %s (disallowed host; see "
409              "http://bit.ly/2hrImnY for details).\n"
410          )
411          if fatal:
412              raise DistutilsError(msg % url)
413          else:
414              self.warn(msg, url)

416      def scan_egg_links(self, search_path):
417          dirs = filter(os.path.isdir, search_path)
418          egg_links = (
419              (path, entry)
420              for path in dirs
421              for entry in os.listdir(path)
422              if entry.endswith('.egg-link')
423          )
424          list(itertools.starmap(self.scan_egg_link, egg_links))

426      def scan_egg_link(self, path, entry):
427          with open(os.path.join(path, entry)) as raw_lines:
428              # filter non-empty lines
429              lines = list(filter(None, map(str.strip, raw_lines)))
430  
431          if len(lines) != 2:
432              # format is not recognized; punt
433              return
434  
435          egg_path, setup_path = lines
436  
437          for dist in find_distributions(os.path.join(path, egg_path)):
438              dist.location = os.path.join(path, *lines)
439              dist.precedence = SOURCE_DIST
440              self.add(dist)

442      def _scan(self, link):
443          # Process a URL to see if it's for a package page
444          NO_MATCH_SENTINEL = None, None
445          if not link.startswith(self.index_url):
446              return NO_MATCH_SENTINEL
447  
448          parts = list(map(urllib.parse.unquote, link[len(self.index_url) :].split('/')))
449          if len(parts) != 2 or '#' in parts[1]:
450              return NO_MATCH_SENTINEL
451  
452          # it's a package page, sanitize and index it
453          pkg = safe_name(parts[0])
454          ver = safe_version(parts[1])
455          self.package_pages.setdefault(pkg.lower(), {})[link] = True
456          return to_filename(pkg), to_filename(ver)

458      def process_index(self, url, page):
459          """Process the contents of a PyPI page"""
460  
461          # process an index page into the package-page index
462          for match in HREF.finditer(page):
463              try:
464                  self._scan(urllib.parse.urljoin(url, htmldecode(match.group(1))))
465              except ValueError:
466                  pass
467  
468          pkg, ver = self._scan(url)  # ensure this page is in the page index
469          if not pkg:
470              return ""  # no sense double-scanning non-package pages
471  
472          # process individual package page
473          for new_url in find_external_links(url, page):
474              # Process the found URL
475              base, frag = egg_info_for_url(new_url)
476              if base.endswith('.py') and not frag:
477                  if ver:
478                      new_url += '#egg=%s-%s' % (pkg, ver)
479                  else:
480                      self.need_version_info(url)
481              self.scan_url(new_url)
482  
483          return PYPI_MD5.sub(
484              lambda m: '<a href="%s#md5=%s">%s</a>' % m.group(1, 3, 2), page
485          )

487      def need_version_info(self, url):
488          self.scan_all(
489              "Page at %s links to .py file(s) without version info; an index "
490              "scan is required.",
491              url,
492          )

494      def scan_all(self, msg=None, *args):
495          if self.index_url not in self.fetched_urls:
496              if msg:
497                  self.warn(msg, *args)
498              self.info("Scanning index of all packages (this may take a while)")
499          self.scan_url(self.index_url)

501      def find_packages(self, requirement):
502          self.scan_url(self.index_url + requirement.unsafe_name + '/')
503  
504          if not self.package_pages.get(requirement.key):
505              # Fall back to safe version of the name
506              self.scan_url(self.index_url + requirement.project_name + '/')
507  
508          if not self.package_pages.get(requirement.key):
509              # We couldn't find the target package, so search the index page too
510              self.not_found_in_index(requirement)
511  
512          for url in list(self.package_pages.get(requirement.key, ())):
513              # scan each page that might be related to the desired package
514              self.scan_url(url)
515  
516      def obtain(self, requirement, installer=None):
517          self.prescan()
518          self.find_packages(requirement)
519          for dist in self[requirement.key]:
520              if dist in requirement:
521                  return dist
522              self.debug("%s does not match %s", requirement, dist)
523          return super(PackageIndex, self).obtain(requirement, installer)

525      def check_hash(self, checker, filename, tfp):
526          """ 
527          checker is a ContentChecker 
528          """
529          checker.report(self.debug, "Validating %%s checksum for %s" % filename)
530          if not checker.is_valid():
531              tfp.close()
532              os.unlink(filename)
533              raise DistutilsError(
534                  "%s validation failed for %s; "
535                  "possible download problem?"
536                  % (checker.hash.name, os.path.basename(filename))
537              )

539      def add_find_links(self, urls):
540          """Add `urls` to the list that will be prescanned for searches"""
541          for url in urls:
542              if (
543                  self.to_scan is None  # if we have already "gone online"
544                  or not URL_SCHEME(url)  # or it's a local file/directory
545                  or url.startswith('file:')
546                  or list(distros_for_url(url))  # or a direct package link
547              ):
548                  # then go ahead and process it now
549                  self.scan_url(url)
550              else:
551                  # otherwise, defer retrieval till later
552                  self.to_scan.append(url)
553  
554      def prescan(self):
555          """Scan urls scheduled for prescanning (e.g. --find-links)"""
556          if self.to_scan:
557              list(map(self.scan_url, self.to_scan))
558          self.to_scan = None  # from now on, go ahead and process immediately

560      def not_found_in_index(self, requirement):
561          if self[requirement.key]:  # we've seen at least one distro
562              meth, msg = self.info, "Couldn't retrieve index page for %r"
563          else:  # no distros seen for this name, might be misspelled
564              meth, msg = (
565                  self.warn,
566                  "Couldn't find index page for %r (maybe misspelled?)",
567              )
568          meth(msg, requirement.unsafe_name)
569          self.scan_all()

571      def download(self, spec, tmpdir):
572          """Locate and/or download `spec` to `tmpdir`, returning a local path 
573   
574          `spec` may be a ``Requirement`` object, or a string containing a URL, 
575          an existing local filename, or a project/version requirement spec 
576          (i.e. the string form of a ``Requirement`` object).  If it is the URL 
577          of a .py file with an unambiguous ``#egg=name-version`` tag (i.e., one 
578          that escapes ``-`` as ``_`` throughout), a trivial ``setup.py`` is 
579          automatically created alongside the downloaded file. 
580   
581          If `spec` is a ``Requirement`` object or a string containing a 
582          project/version requirement spec, this method returns the location of 
583          a matching distribution (possibly after downloading it to `tmpdir`). 
584          If `spec` is a locally existing file or directory name, it is simply 
585          returned unchanged.  If `spec` is a URL, it is downloaded to a subpath 
586          of `tmpdir`, and the local filename is returned.  Various errors may be 
587          raised if a problem occurs during downloading. 
588          """
589          if not isinstance(spec, Requirement):
590              scheme = URL_SCHEME(spec)
591              if scheme:
592                  # It's a url, download it to tmpdir
593                  found = self._download_url(scheme.group(1), spec, tmpdir)
594                  base, fragment = egg_info_for_url(spec)
595                  if base.endswith('.py'):
596                      found = self.gen_setup(found, fragment, tmpdir)
597                  return found
598              elif os.path.exists(spec):
599                  # Existing file or directory, just return it
600                  return spec
601              else:
602                  spec = parse_requirement_arg(spec)
603          return getattr(self.fetch_distribution(spec, tmpdir), 'location', None)

605      def fetch_distribution(  # noqa: C901  # is too complex (14)  # FIXME
606          self,
607          requirement,
608          tmpdir,
609          force_scan=False,
610          source=False,
611          develop_ok=False,
612          local_index=None,
613      ):
614          """Obtain a distribution suitable for fulfilling `requirement` 
615   
616          `requirement` must be a ``pkg_resources.Requirement`` instance. 
617          If necessary, or if the `force_scan` flag is set, the requirement is 
618          searched for in the (online) package index as well as the locally 
619          installed packages.  If a distribution matching `requirement` is found, 
620          the returned distribution's ``location`` is the value you would have 
621          gotten from calling the ``download()`` method with the matching 
622          distribution's URL or filename.  If no matching distribution is found, 
623          ``None`` is returned. 
624   
625          If the `source` flag is set, only source distributions and source 
626          checkout links will be considered.  Unless the `develop_ok` flag is 
627          set, development and system eggs (i.e., those using the ``.egg-info`` 
628          format) will be ignored. 
629          """
630          # process a Requirement
631          self.info("Searching for %s", requirement)
632          skipped = {}
633          dist = None
634  
635          def find(req, env=None):
636              if env is None:
637                  env = self
638              # Find a matching distribution; may be called more than once
639  
640              for dist in env[req.key]:
641  
642                  if dist.precedence == DEVELOP_DIST and not develop_ok:
643                      if dist not in skipped:
644                          self.warn(
645                              "Skipping development or system egg: %s",
646                              dist,
647                          )
648                          skipped[dist] = 1
649                      continue
650  
651                  test = dist in req and (dist.precedence <= SOURCE_DIST or not source)
652                  if test:
653                      loc = self.download(dist.location, tmpdir)
654                      dist.download_location = loc
655                      if os.path.exists(dist.download_location):
656                          return dist
657  
658          if force_scan:
659              self.prescan()
660              self.find_packages(requirement)
661              dist = find(requirement)
662  
663          if not dist and local_index is not None:
664              dist = find(requirement, local_index)
665  
666          if dist is None:
667              if self.to_scan is not None:
668                  self.prescan()
669              dist = find(requirement)
670  
671          if dist is None and not force_scan:
672              self.find_packages(requirement)
673              dist = find(requirement)
674  
675          if dist is None:
676              self.warn(
677                  "No local packages or working download links found for %s%s",
678                  (source and "a source distribution of " or ""),
679                  requirement,
680              )
681          else:
682              self.info("Best match: %s", dist)
683              return dist.clone(location=dist.download_location)

698      def gen_setup(self, filename, fragment, tmpdir):
699          match = EGG_FRAGMENT.match(fragment)
700          dists = (
701              match
702              and [
703                  d
704                  for d in interpret_distro_name(filename, match.group(1), None)
705                  if d.version
706              ]
707              or []
708          )

714              if os.path.dirname(filename) != tmpdir:

716                  if not (os.path.exists(dst) and os.path.samefile(filename, dst)):
717                      shutil.copy2(filename, dst)
718                      filename = dst

726                          dists[0].version,

728                      )
729                  )
730              return filename
731  
732          elif match:
733              raise DistutilsError(
734                  "Can't unambiguously interpret project/version identifier %r; "
735                  "any dashes in the name or version should be escaped using "
736                  "underscores. %r" % (fragment, dists)

746      def _download_to(self, url, filename):
747          self.info("Downloading %s", url)
748          # Download the file
749          fp = None
750          try:
751              checker = HashChecker.from_url(url)
752              fp = self.open_url(url)
753              if isinstance(fp, urllib.error.HTTPError):
754                  raise DistutilsError(
755                      "Can't download %s: %s %s" % (url, fp.code, fp.msg)
756                  )
757              headers = fp.info()
758              blocknum = 0
759              bs = self.dl_blocksize
760              size = -1
761              if "content-length" in headers:
762                  # Some servers return multiple Content-Length headers :(
763                  sizes = headers.get_all('Content-Length')
764                  size = max(map(int, sizes))
765                  self.reporthook(url, filename, blocknum, bs, size)
766              with open(filename, 'wb') as tfp:
767                  while True:
768                      block = fp.read(bs)
769                      if block:
770                          checker.feed(block)
771                          tfp.write(block)
772                          blocknum += 1
773                          self.reporthook(url, filename, blocknum, bs, size)
774                      else:
775                          break
776                  self.check_hash(checker, filename, tfp)
777              return headers
778          finally:
779              if fp:
780                  fp.close()

820      def _download_url(self, scheme, url, tmpdir):
821          # Determine download filename
822          #
823          name, fragment = egg_info_for_url(url)
824          if name:
825              while '..' in name:
826                  name = name.replace('..', '.').replace('\\', '_')
827          else:
828              name = "__downloaded__"  # default if URL has no path contents
829  
830          if name.endswith('.egg.zip'):
831              name = name[:-4]  # strip the extra .zip before download
832  
833          filename = os.path.join(tmpdir, name)
834  
835          # Download the file
836          #
837          if scheme == 'svn' or scheme.startswith('svn+'):
838              return self._download_svn(url, filename)
839          elif scheme == 'git' or scheme.startswith('git+'):
840              return self._download_git(url, filename)
841          elif scheme.startswith('hg+'):
842              return self._download_hg(url, filename)
843          elif scheme == 'file':
844              return urllib.request.url2pathname(urllib.parse.urlparse(url)[2])
845          else:
846              self.url_ok(url, True)  # raises error if not allowed
847              return self._attempt_download(url, filename)

859      def _download_html(self, url, headers, filename):
860          file = open(filename)
861          for line in file:
862              if line.strip():
863                  # Check for a subversion index page
864                  if re.search(r'<title>([^- ]+ - )?Revision \d+:', line):
865                      # it's a subversion index page:
866                      file.close()
867                      os.unlink(filename)
868                      return self._download_svn(url, filename)
869                  break  # not an index page
870          file.close()
871          os.unlink(filename)
872          raise DistutilsError("Unexpected HTML page found at " + url)

874      def _download_svn(self, url, filename):
875          warnings.warn("SVN download support is deprecated", UserWarning)
876          url = url.split('#', 1)[0]  # remove any fragment for svn's sake
877          creds = ''
878          if url.lower().startswith('svn:') and '@' in url:
879              scheme, netloc, path, p, q, f = urllib.parse.urlparse(url)
880              if not netloc and path.startswith('//') and '/' in path[2:]:
881                  netloc, path = path[2:].split('/', 1)
882                  auth, host = _splituser(netloc)
883                  if auth:
884                      if ':' in auth:
885                          user, pw = auth.split(':', 1)
886                          creds = " --username=%s --password=%s" % (user, pw)
887                      else:
888                          creds = " --username=" + auth
889                      netloc = host
890                      parts = scheme, netloc, url, p, q, f
891                      url = urllib.parse.urlunparse(parts)
892          self.info("Doing subversion checkout from %s to %s", url, filename)
893          os.system("svn checkout%s -q %s %s" % (creds, url, filename))
894          return filename

999  def _encode_auth(auth):
1000     """ 
1001     Encode auth from a URL suitable for an HTTP header. 
1002     >>> str(_encode_auth('username%3Apassword')) 
1003     'dXNlcm5hbWU6cGFzc3dvcmQ=' 
1004  
1005     Long auth strings should not cause a newline to be inserted. 
1006     >>> long_auth = 'username:' + 'password'*10 
1007     >>> chr(10) in str(_encode_auth(long_auth)) 
1008     False 
1009     """
1010     auth_s = urllib.parse.unquote(auth)
1011     # convert to bytes
1012     auth_bytes = auth_s.encode()
1013     encoded_bytes = base64.b64encode(auth_bytes)
1014     # convert back to a string
1015     encoded = encoded_bytes.decode()
1016     # strip the trailing carriage return
1017     return encoded.replace('\n', '')

1076 def open_with_auth(url, opener=urllib.request.urlopen):
1077     """Open a urllib2 request, handling HTTP authentication"""
1078 
1079     parsed = urllib.parse.urlparse(url)
1080     scheme, netloc, path, params, query, frag = parsed
1081 
1082     # Double scheme does not raise on macOS as revealed by a
1083     # failing test. We would expect "nonnumeric port". Refs #20.
1084     if netloc.endswith(':'):
1085         raise http.client.InvalidURL("nonnumeric port: ''")
1086 
1087     if scheme in ('http', 'https'):
1088         auth, address = _splituser(netloc)
1089     else:
1090         auth = None
1091 
1092     if not auth:
1093         cred = PyPIConfig().find_credential(url)
1094         if cred:
1095             auth = str(cred)
1096             info = cred.username, url
1097             log.info('Authenticating as %s for %s (from .pypirc)', *info)
1098 
1099     if auth:
1100         auth = "Basic " + _encode_auth(auth)
1101         parts = scheme, address, path, params, query, frag
1102         new_url = urllib.parse.urlunparse(parts)
1103         request = urllib.request.Request(new_url)
1104         request.add_header("Authorization", auth)
1105     else:
1106         request = urllib.request.Request(url)
1107 
1108     request.add_header('User-Agent', user_agent)
1109     fp = opener(request)
1110 
1111     if auth:
1112         # Put authentication info back into request URL if same host,
1113         # so that links found on the page will work
1114         s2, h2, path2, param2, query2, frag2 = urllib.parse.urlparse(fp.url)
1115         if s2 == scheme and h2 == address:
1116             parts = s2, netloc, path2, param2, query2, frag2
1117             fp.url = urllib.parse.urlunparse(parts)
1118 
1119     return fp

1138 def local_open(url):
1139     """Read a local path, with special support for directories"""
1140     scheme, server, path, param, query, frag = urllib.parse.urlparse(url)
1141     filename = urllib.request.url2pathname(path)
1142     if os.path.isfile(filename):
1143         return urllib.request.urlopen(url)
1144     elif path.endswith('/') and os.path.isdir(filename):
1145         files = []
1146         for f in os.listdir(filename):
1147             filepath = os.path.join(filename, f)
1148             if f == 'index.html':
1149                 with open(filepath, 'r') as fp:
1150                     body = fp.read()
1151                 break
1152             elif os.path.isdir(filepath):
1153                 f += '/'
1154             files.append('<a href="{name}">{name}</a>'.format(name=f))
1155         else:
1156             tmpl = (
1157                 "<html><head><title>{url}</title>" "</head><body>{files}</body></html>"
1158             )
1159             body = tmpl.format(url=url, files='\n'.join(files))
1160         status, message = 200, "OK"
1161     else:
1162         status, message, body = 404, "Path not found", "Not found"
1163 
1164     headers = {'content-type': 'text/html'}
1165     body_stream = io.StringIO(body)
1166     return urllib.error.HTTPError(url, status, message, headers, body_stream)
