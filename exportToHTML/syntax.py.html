135  class PygmentsSyntaxTheme(SyntaxTheme):
136      """Syntax theme that delegates to Pygments theme."""
137  
138      def __init__(self, theme: Union[str, Type[PygmentsStyle]]) -> None:
139          self._style_cache: Dict[TokenType, Style] = {}
140          if isinstance(theme, str):
141              try:
142                  self._pygments_style_class = get_style_by_name(theme)
143              except ClassNotFound:
144                  self._pygments_style_class = get_style_by_name("default")
145          else:
146              self._pygments_style_class = theme
147  
148          self._background_color = self._pygments_style_class.background_color
149          self._background_style = Style(bgcolor=self._background_color)
150  
151      def get_style_for_token(self, token_type: TokenType) -> Style:
152          """Get a style from a Pygments class."""
153          try:
154              return self._style_cache[token_type]
155          except KeyError:
156              try:
157                  pygments_style = self._pygments_style_class.style_for_token(token_type)
158              except KeyError:
159                  style = Style.null()
160              else:
161                  color = pygments_style["color"]
162                  bgcolor = pygments_style["bgcolor"]
163                  style = Style(
164                      color="#" + color if color else "#000000",
165                      bgcolor="#" + bgcolor if bgcolor else self._background_color,
166                      bold=pygments_style["bold"],
167                      italic=pygments_style["italic"],
168                      underline=pygments_style["underline"],
169                  )
170              self._style_cache[token_type] = style
171          return style
172  
173      def get_background_style(self) -> Style:
174          return self._background_style

177  class ANSISyntaxTheme(SyntaxTheme):
178      """Syntax theme to use standard colors."""
179  
180      def __init__(self, style_map: Dict[TokenType, Style]) -> None:
181          self.style_map = style_map
182          self._missing_style = Style.null()
183          self._background_style = Style.null()
184          self._style_cache: Dict[TokenType, Style] = {}
185  
186      def get_style_for_token(self, token_type: TokenType) -> Style:
187          """Look up style in the style map."""
188          try:
189              return self._style_cache[token_type]
190          except KeyError:
191              # Styles form a hierarchy
192              # We need to go from most to least specific
193              # e.g. ("foo", "bar", "baz") to ("foo", "bar")  to ("foo",)
194              get_style = self.style_map.get
195              token = tuple(token_type)
196              style = self._missing_style
197              while token:
198                  _style = get_style(token)
199                  if _style is not None:
200                      style = _style
201                      break
202                  token = token[:-1]
203              self._style_cache[token_type] = style
204              return style
205  
206      def get_background_style(self) -> Style:
207          return self._background_style

363      @classmethod
364      def guess_lexer(cls, path: str, code: Optional[str] = None) -> str:
365          """Guess the alias of the Pygments lexer to use based on a path and an optional string of code. 
366          If code is supplied, it will use a combination of the code and the filename to determine the 
367          best lexer to use. For example, if the file is ``index.html`` and the file contains Django 
368          templating syntax, then "html+django" will be returned. If the file is ``index.html``, and no 
369          templating language is used, the "html" lexer will be used. If no string of code 
370          is supplied, the lexer will be chosen based on the file extension.. 
371   
372          Args: 
373               path (AnyStr): The path to the file containing the code you wish to know the lexer for. 
374               code (str, optional): Optional string of code that will be used as a fallback if no lexer 
375                  is found for the supplied path. 
376   
377          Returns: 
378              str: The name of the Pygments lexer that best matches the supplied path/code. 
379          """
380          lexer: Optional[Lexer] = None
381          lexer_name = "default"
382          if code:
383              try:
384                  lexer = guess_lexer_for_filename(path, code)
385              except ClassNotFound:
386                  pass
387  
388          if not lexer:
389              try:
390                  _, ext = os.path.splitext(path)
391                  if ext:
392                      extension = ext.lstrip(".").lower()
393                      lexer = get_lexer_by_name(extension)
394              except ClassNotFound:
395                  pass
396  
397          if lexer:
398              if lexer.aliases:
399                  lexer_name = lexer.aliases[0]
400              else:
401                  lexer_name = lexer.name
402  
403          return lexer_name

441      def highlight(
442          self,
443          code: str,
444          line_range: Optional[Tuple[Optional[int], Optional[int]]] = None,
445      ) -> Text:
446          """Highlight code and return a Text instance. 
447   
448          Args: 
449              code (str): Code to highlight. 
450              line_range(Tuple[int, int], optional): Optional line range to highlight. 
451   
452          Returns: 
453              Text: A text instance containing highlighted syntax. 
454          """
455  
456          base_style = self._get_base_style()
457          justify: JustifyMethod = (
458              "default" if base_style.transparent_background else "left"
459          )
460  
461          text = Text(
462              justify=justify,
463              style=base_style,
464              tab_size=self.tab_size,
465              no_wrap=not self.word_wrap,
466          )
467          _get_theme_style = self._theme.get_style_for_token
468  
469          lexer = self.lexer
470  
471          if lexer is None:
472              text.append(code)
473          else:
474              if line_range:
475                  # More complicated path to only stylize a portion of the code
476                  # This speeds up further operations as there are less spans to process
477                  line_start, line_end = line_range
478  
479                  def line_tokenize() -> Iterable[Tuple[Any, str]]:
480                      """Split tokens to one per line."""
481                      assert lexer  # required to make MyPy happy - we know lexer is not None at this point
482  
483                      for token_type, token in lexer.get_tokens(code):
484                          while token:
485                              line_token, new_line, token = token.partition("\n")
486                              yield token_type, line_token + new_line
487  
488                  def tokens_to_spans() -> Iterable[Tuple[str, Optional[Style]]]:
489                      """Convert tokens to spans."""
490                      tokens = iter(line_tokenize())
491                      line_no = 0
492                      _line_start = line_start - 1 if line_start else 0
493  
494                      # Skip over tokens until line start
495                      while line_no < _line_start:
496                          _token_type, token = next(tokens)
497                          yield (token, None)
498                          if token.endswith("\n"):
499                              line_no += 1
500                      # Generate spans until line end
501                      for token_type, token in tokens:
502                          yield (token, _get_theme_style(token_type))
503                          if token.endswith("\n"):
504                              line_no += 1
505                              if line_end and line_no >= line_end:
506                                  break
507  
508                  text.append_tokens(tokens_to_spans())
509  
510              else:
511                  text.append_tokens(
512                      (token, _get_theme_style(token_type))
513                      for token_type, token in lexer.get_tokens(code)
514                  )
515              if self.background_color is not None:
516                  text.stylize(f"on {self.background_color}")
517  
518          if self._stylized_ranges:
519              self._apply_stylized_ranges(text)
520  
521          return text

537      def _get_line_numbers_color(self, blend: float = 0.3) -> Color:
538          background_style = self._theme.get_background_style() + self.background_style
539          background_color = background_style.bgcolor
540          if background_color is None or background_color.is_system_defined:
541              return Color.default()
542          foreground_color = self._get_token_color(Token.Text)
543          if foreground_color is None or foreground_color.is_system_defined:
544              return foreground_color or Color.default()
545          new_color = blend_rgb(
546              background_color.get_truecolor(),
547              foreground_color.get_truecolor(),
548              cross_fade=blend,
549          )
550          return Color.from_triplet(new_color)

563      def _get_number_styles(self, console: Console) -> Tuple[Style, Style, Style]:
564          """Get background, number, and highlight styles for line numbers."""
565          background_style = self._get_base_style()
566          if background_style.transparent_background:
567              return Style.null(), Style(dim=True), Style.null()
568          if console.color_system in ("256", "truecolor"):
569              number_style = Style.chain(
570                  background_style,
571                  self._theme.get_style_for_token(Token.Text),
572                  Style(color=self._get_line_numbers_color()),
573                  self.background_style,
574              )
575              highlight_number_style = Style.chain(
576                  background_style,
577                  self._theme.get_style_for_token(Token.Text),
578                  Style(bold=True, color=self._get_line_numbers_color(0.9)),
579                  self.background_style,
580              )
581          else:
582              number_style = background_style + Style(dim=True)
583              highlight_number_style = background_style + Style(dim=False)
584          return background_style, number_style, highlight_number_style

586      def __rich_measure__(
587          self, console: "Console", options: "ConsoleOptions"
588      ) -> "Measurement":
589          _, right, _, left = Padding.unpack(self.padding)
590          if self.code_width is not None:
591              width = self.code_width + self._numbers_column_width + right + left
592              return Measurement(self._numbers_column_width, width)
593          return Measurement(self._numbers_column_width, options.max_width)

595      def __rich_console__(
596          self, console: Console, options: ConsoleOptions
597      ) -> RenderResult:
598          segments = Segments(self._get_syntax(console, options))
599          if self.padding:
600              yield Padding(
601                  segments, style=self._theme.get_background_style(), pad=self.padding
602              )
603          else:
604              yield segments

606      def _get_syntax(
607          self,
608          console: Console,
609          options: ConsoleOptions,
610      ) -> Iterable[Segment]:
611          """ 
612          Get the Segments for the Syntax object, excluding any vertical/horizontal padding 
613          """
614          transparent_background = self._get_base_style().transparent_background
615          code_width = (
616              (
617                  (options.max_width - self._numbers_column_width - 1)
618                  if self.line_numbers
619                  else options.max_width
620              )
621              if self.code_width is None
622              else self.code_width
623          )
624  
625          ends_on_nl, processed_code = self._process_code(self.code)
626          text = self.highlight(processed_code, self.line_range)
627  
628          if not self.line_numbers and not self.word_wrap and not self.line_range:
629              if not ends_on_nl:
630                  text.remove_suffix("\n")
631              # Simple case of just rendering text
632              style = (
633                  self._get_base_style()
634                  + self._theme.get_style_for_token(Comment)
635                  + Style(dim=True)
636                  + self.background_style
637              )
638              if self.indent_guides and not options.ascii_only:
639                  text = text.with_indent_guides(self.tab_size, style=style)
640                  text.overflow = "crop"
641              if style.transparent_background:
642                  yield from console.render(
643                      text, options=options.update(width=code_width)
644                  )
645              else:
646                  syntax_lines = console.render_lines(
647                      text,
648                      options.update(width=code_width, height=None, justify="left"),
649                      style=self.background_style,
650                      pad=True,
651                      new_lines=True,
652                  )
653                  for syntax_line in syntax_lines:
654                      yield from syntax_line
655              return
656  
657          start_line, end_line = self.line_range or (None, None)
658          line_offset = 0
659          if start_line:
660              line_offset = max(0, start_line - 1)
661          lines: Union[List[Text], Lines] = text.split("\n", allow_blank=ends_on_nl)
662          if self.line_range:
663              lines = lines[line_offset:end_line]
664  
665          if self.indent_guides and not options.ascii_only:
666              style = (
667                  self._get_base_style()
668                  + self._theme.get_style_for_token(Comment)
669                  + Style(dim=True)
670                  + self.background_style
671              )
672              lines = (
673                  Text("\n")
674                  .join(lines)
675                  .with_indent_guides(self.tab_size, style=style)
676                  .split("\n", allow_blank=True)
677              )
678  
679          numbers_column_width = self._numbers_column_width
680          render_options = options.update(width=code_width)
681  
682          highlight_line = self.highlight_lines.__contains__
683          _Segment = Segment
684          new_line = _Segment("\n")
685  
686          line_pointer = "> " if options.legacy_windows else " "
687  
688          (
689              background_style,
690              number_style,
691              highlight_number_style,
692          ) = self._get_number_styles(console)
693  
694          for line_no, line in enumerate(lines, self.start_line + line_offset):
695              if self.word_wrap:
696                  wrapped_lines = console.render_lines(
697                      line,
698                      render_options.update(height=None, justify="left"),
699                      style=background_style,
700                      pad=not transparent_background,
701                  )
702              else:
703                  segments = list(line.render(console, end=""))
704                  if options.no_wrap:
705                      wrapped_lines = [segments]
706                  else:
707                      wrapped_lines = [
708                          _Segment.adjust_line_length(
709                              segments,
710                              render_options.max_width,
711                              style=background_style,
712                              pad=not transparent_background,
713                          )
714                      ]
715  
716              if self.line_numbers:
717                  wrapped_line_left_pad = _Segment(
718                      " " * numbers_column_width + " ", background_style
719                  )
720                  for first, wrapped_line in loop_first(wrapped_lines):
721                      if first:
722                          line_column = str(line_no).rjust(numbers_column_width - 2) + " "
723                          if highlight_line(line_no):
724                              yield _Segment(line_pointer, Style(color="red"))
725                              yield _Segment(line_column, highlight_number_style)
726                          else:
727                              yield _Segment("  ", highlight_number_style)
728                              yield _Segment(line_column, number_style)
729                      else:
730                          yield wrapped_line_left_pad
731                      yield from wrapped_line
732                      yield new_line
733              else:
734                  for wrapped_line in wrapped_lines:
735                      yield from wrapped_line
736                      yield new_line

738      def _apply_stylized_ranges(self, text: Text) -> None:
739          """ 
740          Apply stylized ranges to a text instance, 
741          using the given code to determine the right portion to apply the style to. 
742   
743          Args: 
744              text (Text): Text instance to apply the style to. 
745          """
746          code = text.plain
747          newlines_offsets = [
748              # Let's add outer boundaries at each side of the list:
749              0,
750              # N.B. using "\n" here is much faster than using metacharacters such as "^" or "\Z":
751              *[
752                  match.start() + 1
753                  for match in re.finditer("\n", code, flags=re.MULTILINE)
754              ],
755              len(code) + 1,
756          ]
757  
758          for stylized_range in self._stylized_ranges:
759              start = _get_code_index_for_syntax_position(
760                  newlines_offsets, stylized_range.start
761              )
762              end = _get_code_index_for_syntax_position(
763                  newlines_offsets, stylized_range.end
764              )
765              if start is not None and end is not None:
766                  text.stylize(stylized_range.style, start, end)
767  
768      def _process_code(self, code: str) -> Tuple[bool, str]:
769          """ 
770          Applies various processing to a raw code string 
771          (normalises it so it always ends with a line return, dedents it if necessary, etc.) 
772   
773          Args: 
774              code (str): The raw code string to process 
775   
776          Returns: 
777              Tuple[bool, str]: the boolean indicates whether the raw code ends with a line return, 
778                  while the string is the processed code. 
779          """
780          ends_on_nl = code.endswith("\n")
781          processed_code = code if ends_on_nl else code + "\n"
782          processed_code = (
783              textwrap.dedent(processed_code) if self.dedent else processed_code
784          )
785          processed_code = processed_code.expandtabs(self.tab_size)
786          return ends_on_nl, processed_code

789  def _get_code_index_for_syntax_position(
790      newlines_offsets: Sequence[int], position: SyntaxPosition
791  ) -> Optional[int]:
792      """ 
793      Returns the index of the code string for the given positions. 
794   
795      Args: 
796          newlines_offsets (Sequence[int]): The offset of each newline character found in the code snippet. 
797          position (SyntaxPosition): The position to search for. 
798   
799      Returns: 
800          Optional[int]: The index of the code string for this position, or `None` 
801              if the given position's line number is out of range (if it's the column that is out of range 
802              we silently clamp its value so that it reaches the end of the line) 
803      """
804      lines_count = len(newlines_offsets)
805  
806      line_number, column_index = position
807      if line_number > lines_count or len(newlines_offsets) < (line_number + 1):
808          return None  # `line_number` is out of range
809      line_index = line_number - 1
810      line_length = newlines_offsets[line_index + 1] - newlines_offsets[line_index] - 1
811      # If `column_index` is out of range: let's silently clamp it:
812      column_index = min(line_length, column_index)
813      return newlines_offsets[line_index] + column_index
