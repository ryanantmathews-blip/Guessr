59   def _get_prepared_distribution(
60       req: InstallRequirement,
61       build_tracker: BuildTracker,
62       finder: PackageFinder,
63       build_isolation: bool,
64       check_build_deps: bool,
65   ) -> BaseDistribution:
66       """Prepare a distribution for installation."""
67       abstract_dist = make_distribution_for_install_requirement(req)
68       with build_tracker.track(req):
69           abstract_dist.prepare_distribution_metadata(
70               finder, build_isolation, check_build_deps
71           )
72       return abstract_dist.get_metadata_distribution()

75   def unpack_vcs_link(link: Link, location: str, verbosity: int) -> None:
76       vcs_backend = vcs.get_backend_for_scheme(link.scheme)
77       assert vcs_backend is not None
78       vcs_backend.unpack(location, url=hide_url(link.url), verbosity=verbosity)

81   class File:
82       def __init__(self, path: str, content_type: Optional[str]) -> None:
83           self.path = path
84           if content_type is None:
85               self.content_type = mimetypes.guess_type(path)[0]
86           else:
87               self.content_type = content_type

90   def get_http_url(
91       link: Link,
92       download: Downloader,
93       download_dir: Optional[str] = None,
94       hashes: Optional[Hashes] = None,
95   ) -> File:
96       temp_dir = TempDirectory(kind="unpack", globally_managed=True)
97       # If a download dir is specified, is the file already downloaded there?
98       already_downloaded_path = None
99       if download_dir:
100          already_downloaded_path = _check_download_dir(link, download_dir, hashes)
101  
102      if already_downloaded_path:
103          from_path = already_downloaded_path
104          content_type = None
105      else:
106          # let's download to a tmp dir
107          from_path, content_type = download(link, temp_dir.path)
108          if hashes:
109              hashes.check_against_path(from_path)
110  
111      return File(from_path, content_type)

114  def get_file_url(
115      link: Link, download_dir: Optional[str] = None, hashes: Optional[Hashes] = None
116  ) -> File:
117      """Get file and optionally check its hash."""
118      # If a download dir is specified, is the file already there and valid?
119      already_downloaded_path = None
120      if download_dir:
121          already_downloaded_path = _check_download_dir(link, download_dir, hashes)
122  
123      if already_downloaded_path:
124          from_path = already_downloaded_path
125      else:
126          from_path = link.file_path
127  
128      # If --require-hashes is off, `hashes` is either empty, the
129      # link's embedded hash, or MissingHashes; it is required to
130      # match. If --require-hashes is on, we are satisfied by any
131      # hash in `hashes` matching: a URL-based or an option-based
132      # one; no internet-sourced hash will be in `hashes`.
133      if hashes:
134          hashes.check_against_path(from_path)
135      return File(from_path, None)

138  def unpack_url(
139      link: Link,
140      location: str,
141      download: Downloader,
142      verbosity: int,
143      download_dir: Optional[str] = None,
144      hashes: Optional[Hashes] = None,
145  ) -> Optional[File]:
146      """Unpack link into location, downloading if required. 
147   
148      :param hashes: A Hashes object, one of whose embedded hashes must match, 
149          or HashMismatch will be raised. If the Hashes is empty, no matches are 
150          required, and unhashable types of requirements (like VCS ones, which 
151          would ordinarily raise HashUnsupported) are allowed. 
152      """
153      # non-editable vcs urls
154      if link.is_vcs:
155          unpack_vcs_link(link, location, verbosity=verbosity)
156          return None
157  
158      assert not link.is_existing_dir()
159  
160      # file urls
161      if link.is_file:
162          file = get_file_url(link, download_dir, hashes=hashes)
163  
164      # http urls
165      else:
166          file = get_http_url(
167              link,
168              download,
169              download_dir,
170              hashes=hashes,
171          )
172  
173      # unpack the archive to the build dir location. even when only downloading
174      # archives, they have to be unpacked to parse dependencies, except wheels
175      if not link.is_wheel:
176          unpack_file(file.path, location, file.content_type)
177  
178      return file
179

181  def _check_download_dir(
182      link: Link, download_dir: str, hashes: Optional[Hashes]
183  ) -> Optional[str]:
184      """Check download_dir for previously downloaded file with correct hash 
185      If a correct file is found return its path else None 
186      """
187      download_path = os.path.join(download_dir, link.filename)
188  
189      if not os.path.exists(download_path):
190          return None
191  
192      # If already downloaded, does its hash match?
193      logger.info("File was already downloaded %s", download_path)
194      if hashes:
195          try:
196              hashes.check_against_path(download_path)
197          except HashMismatch:
198              logger.warning(
199                  "Previously-downloaded file %s has bad hash. Re-downloading.",
200                  download_path,
201              )
202              os.unlink(download_path)
203              return None
204      return download_path

210      def __init__(
211          self,
212          build_dir: str,
213          download_dir: Optional[str],
214          src_dir: str,
215          build_isolation: bool,
216          check_build_deps: bool,
217          build_tracker: BuildTracker,
218          session: PipSession,
219          progress_bar: str,
220          finder: PackageFinder,
221          require_hashes: bool,
222          use_user_site: bool,
223          lazy_wheel: bool,
224          verbosity: int,
225      ) -> None:
226          super().__init__()
227  
228          self.src_dir = src_dir
229          self.build_dir = build_dir
230          self.build_tracker = build_tracker
231          self._session = session
232          self._download = Downloader(session, progress_bar)
233          self._batch_download = BatchDownloader(session, progress_bar)
234          self.finder = finder
235  
236          # Where still-packed archives should be written to. If None, they are
237          # not saved, and are deleted immediately after unpacking.
238          self.download_dir = download_dir
239  
240          # Is build isolation allowed?
241          self.build_isolation = build_isolation
242  
243          # Should check build dependencies?
244          self.check_build_deps = check_build_deps
245  
246          # Should hash-checking be required?
247          self.require_hashes = require_hashes
248  
249          # Should install in user site-packages?
250          self.use_user_site = use_user_site
251  
252          # Should wheels be downloaded lazily?
253          self.use_lazy_wheel = lazy_wheel
254  
255          # How verbose should underlying tooling be?
256          self.verbosity = verbosity
257  
258          # Memoized downloaded files, as mapping of url: path.
259          self._downloaded: Dict[str, str] = {}
260  
261          # Previous "header" printed for a link-based InstallRequirement
262          self._previous_requirement_header = ("", "")

264      def _log_preparing_link(self, req: InstallRequirement) -> None:
265          """Provide context for the requirement being prepared."""
266          if req.link.is_file and not req.original_link_is_in_wheel_cache:
267              message = "Processing %s"
268              information = str(display_path(req.link.file_path))
269          else:
270              message = "Collecting %s"
271              information = str(req.req or req)
272  
273          if (message, information) != self._previous_requirement_header:
274              self._previous_requirement_header = (message, information)
275              logger.info(message, information)
276  
277          if req.original_link_is_in_wheel_cache:
278              with indent_log():
279                  logger.info("Using cached %s", req.link.filename)

281      def _ensure_link_req_src_dir(
282          self, req: InstallRequirement, parallel_builds: bool
283      ) -> None:
284          """Ensure source_dir of a linked InstallRequirement."""
285          # Since source_dir is only set for editable requirements.
286          if req.link.is_wheel:
287              # We don't need to unpack wheels, so no need for a source
288              # directory.
289              return
290          assert req.source_dir is None
291          if req.link.is_existing_dir():
292              # build local directories in-tree
293              req.source_dir = req.link.file_path
294              return
295  
296          # We always delete unpacked sdists after pip runs.
297          req.ensure_has_source_dir(
298              self.build_dir,
299              autodelete=True,
300              parallel_builds=parallel_builds,
301          )
302  
303          # If a checkout exists, it's unwise to keep going.  version
304          # inconsistencies are logged later, but do not fail the
305          # installation.
306          # FIXME: this won't upgrade when there's an existing
307          # package unpacked in `req.source_dir`
308          # TODO: this check is now probably dead code
309          if is_installable_dir(req.source_dir):
310              raise PreviousBuildDirError(
311                  "pip can't proceed with requirements '{}' due to a"
312                  "pre-existing build directory ({}). This is likely "
313                  "due to a previous installation that failed . pip is "
314                  "being responsible and not assuming it can delete this. "
315                  "Please delete it and try again.".format(req, req.source_dir)
316              )

318      def _get_linked_req_hashes(self, req: InstallRequirement) -> Hashes:
319          # By the time this is called, the requirement's link should have
320          # been checked so we can tell what kind of requirements req is
321          # and raise some more informative errors than otherwise.
322          # (For example, we can raise VcsHashUnsupported for a VCS URL
323          # rather than HashMissing.)
324          if not self.require_hashes:
325              return req.hashes(trust_internet=True)
326  
327          # We could check these first 2 conditions inside unpack_url
328          # and save repetition of conditions, but then we would
329          # report less-useful error messages for unhashable
330          # requirements, complaining that there's no hash provided.
331          if req.link.is_vcs:
332              raise VcsHashUnsupported()
333          if req.link.is_existing_dir():
334              raise DirectoryUrlHashUnsupported()
335  
336          # Unpinned packages are asking for trouble when a new version
337          # is uploaded.  This isn't a security check, but it saves users
338          # a surprising hash mismatch in the future.
339          # file:/// URLs aren't pinnable, so don't complain about them
340          # not being pinned.
341          if req.original_link is None and not req.is_pinned:
342              raise HashUnpinned()
343  
344          # If known-good hashes are missing for this requirement,
345          # shim it with a facade object that will provoke hash
346          # computation and then raise a HashMissing exception
347          # showing the user what the hash should be.
348          return req.hashes(trust_internet=False) or MissingHashes()

350      def _fetch_metadata_only(
351          self,
352          req: InstallRequirement,
353      ) -> Optional[BaseDistribution]:
354          if self.require_hashes:
355              logger.debug(
356                  "Metadata-only fetching is not used as hash checking is required",
357              )
358              return None
359          # Try PEP 658 metadata first, then fall back to lazy wheel if unavailable.
360          return self._fetch_metadata_using_link_data_attr(
361              req
362          ) or self._fetch_metadata_using_lazy_wheel(req.link)
363  
364      def _fetch_metadata_using_link_data_attr(
365          self,
366          req: InstallRequirement,
367      ) -> Optional[BaseDistribution]:
368          """Fetch metadata from the data-dist-info-metadata attribute, if possible."""
369          # (1) Get the link to the metadata file, if provided by the backend.
370          metadata_link = req.link.metadata_link()
371          if metadata_link is None:
372              return None
373          assert req.req is not None
374          logger.info(
375              "Obtaining dependency information for %s from %s",
376              req.req,
377              metadata_link,
378          )
379          # (2) Download the contents of the METADATA file, separate from the dist itself.
380          metadata_file = get_http_url(
381              metadata_link,
382              self._download,
383              hashes=metadata_link.as_hashes(),
384          )
385          with open(metadata_file.path, "rb") as f:
386              metadata_contents = f.read()
387          # (3) Generate a dist just from those file contents.
388          metadata_dist = get_metadata_distribution(
389              metadata_contents,
390              req.link.filename,
391              req.req.name,
392          )
393          # (4) Ensure the Name: field from the METADATA file matches the name from the
394          #     install requirement.
395          #
396          #     NB: raw_name will fall back to the name from the install requirement if
397          #     the Name: field is not present, but it's noted in the raw_name docstring
398          #     that that should NEVER happen anyway.
399          if metadata_dist.raw_name != req.req.name:
400              raise MetadataInconsistent(
401                  req, "Name", req.req.name, metadata_dist.raw_name
402              )
403          return metadata_dist

405      def _fetch_metadata_using_lazy_wheel(
406          self,
407          link: Link,
408      ) -> Optional[BaseDistribution]:
409          """Fetch metadata using lazy wheel, if possible."""
410          # --use-feature=fast-deps must be provided.
411          if not self.use_lazy_wheel:
412              return None
413          if link.is_file or not link.is_wheel:
414              logger.debug(
415                  "Lazy wheel is not used as %r does not point to a remote wheel",
416                  link,
417              )
418              return None
419  
420          wheel = Wheel(link.filename)
421          name = canonicalize_name(wheel.name)
422          logger.info(
423              "Obtaining dependency information from %s %s",
424              name,
425              wheel.version,
426          )
427          url = link.url.split("#", 1)[0]
428          try:
429              return dist_from_wheel_url(name, url, self._session)
430          except HTTPRangeRequestUnsupported:
431              logger.debug("%s does not support range requests", url)
432              return None

434      def _complete_partial_requirements(
435          self,
436          partially_downloaded_reqs: Iterable[InstallRequirement],
437          parallel_builds: bool = False,
438      ) -> None:
439          """Download any requirements which were only fetched by metadata."""
440          # Download to a temporary directory. These will be copied over as
441          # needed for downstream 'download', 'wheel', and 'install' commands.
442          temp_dir = TempDirectory(kind="unpack", globally_managed=True).path
443  
444          # Map each link to the requirement that owns it. This allows us to set
445          # `req.local_file_path` on the appropriate requirement after passing
446          # all the links at once into BatchDownloader.
447          links_to_fully_download: Dict[Link, InstallRequirement] = {}
448          for req in partially_downloaded_reqs:
449              assert req.link
450              links_to_fully_download[req.link] = req
451  
452          batch_download = self._batch_download(
453              links_to_fully_download.keys(),
454              temp_dir,
455          )
456          for link, (filepath, _) in batch_download:
457              logger.debug("Downloading link %s to %s", link, filepath)
458              req = links_to_fully_download[link]
459              req.local_file_path = filepath
460  
461          # This step is necessary to ensure all lazy wheels are processed
462          # successfully by the 'download', 'wheel', and 'install' commands.
463          for req in partially_downloaded_reqs:
464              self._prepare_linked_requirement(req, parallel_builds)
465  
466      def prepare_linked_requirement(
467          self, req: InstallRequirement, parallel_builds: bool = False
468      ) -> BaseDistribution:
469          """Prepare a requirement to be obtained from req.link."""
470          assert req.link
471          self._log_preparing_link(req)
472          with indent_log():
473              # Check if the relevant file is already available
474              # in the download directory
475              file_path = None
476              if self.download_dir is not None and req.link.is_wheel:
477                  hashes = self._get_linked_req_hashes(req)
478                  file_path = _check_download_dir(req.link, self.download_dir, hashes)
479  
480              if file_path is not None:
481                  # The file is already available, so mark it as downloaded
482                  self._downloaded[req.link.url] = file_path
483              else:
484                  # The file is not available, attempt to fetch only metadata
485                  metadata_dist = self._fetch_metadata_only(req)
486                  if metadata_dist is not None:
487                      req.needs_more_preparation = True
488                      return metadata_dist
489  
490              # None of the optimizations worked, fully prepare the requirement
491              return self._prepare_linked_requirement(req, parallel_builds)
492  
493      def prepare_linked_requirements_more(
494          self, reqs: Iterable[InstallRequirement], parallel_builds: bool = False
495      ) -> None:
496          """Prepare linked requirements more, if needed."""
497          reqs = [req for req in reqs if req.needs_more_preparation]
498          for req in reqs:
499              # Determine if any of these requirements were already downloaded.
500              if self.download_dir is not None and req.link.is_wheel:
501                  hashes = self._get_linked_req_hashes(req)
502                  file_path = _check_download_dir(req.link, self.download_dir, hashes)
503                  if file_path is not None:
504                      self._downloaded[req.link.url] = file_path
505                      req.needs_more_preparation = False
506  
507          # Prepare requirements we found were already downloaded for some
508          # reason. The other downloads will be completed separately.
509          partially_downloaded_reqs: List[InstallRequirement] = []
510          for req in reqs:
511              if req.needs_more_preparation:
512                  partially_downloaded_reqs.append(req)
513              else:
514                  self._prepare_linked_requirement(req, parallel_builds)
515  
516          # TODO: separate this part out from RequirementPreparer when the v1
517          # resolver can be removed!
518          self._complete_partial_requirements(
519              partially_downloaded_reqs,
520              parallel_builds=parallel_builds,
521          )
522  
523      def _prepare_linked_requirement(
524          self, req: InstallRequirement, parallel_builds: bool
525      ) -> BaseDistribution:
526          assert req.link
527          link = req.link
528  
529          self._ensure_link_req_src_dir(req, parallel_builds)
530          hashes = self._get_linked_req_hashes(req)
531  
532          if link.is_existing_dir():
533              local_file = None
534          elif link.url not in self._downloaded:
535              try:
536                  local_file = unpack_url(
537                      link,
538                      req.source_dir,
539                      self._download,
540                      self.verbosity,
541                      self.download_dir,
542                      hashes,
543                  )
544              except NetworkConnectionError as exc:
545                  raise InstallationError(
546                      "Could not install requirement {} because of HTTP "
547                      "error {} for URL {}".format(req, exc, link)
548                  )
549          else:
550              file_path = self._downloaded[link.url]
551              if hashes:
552                  hashes.check_against_path(file_path)
553              local_file = File(file_path, content_type=None)
554  
555          # If download_info is set, we got it from the wheel cache.
556          if req.download_info is None:
557              # Editables don't go through this function (see
558              # prepare_editable_requirement).
559              assert not req.editable
560              req.download_info = direct_url_from_link(link, req.source_dir)
561              # Make sure we have a hash in download_info. If we got it as part of the
562              # URL, it will have been verified and we can rely on it. Otherwise we
563              # compute it from the downloaded file.
564              if (
565                  isinstance(req.download_info.info, ArchiveInfo)
566                  and not req.download_info.info.hash
567                  and local_file
568              ):
569                  hash = hash_file(local_file.path)[0].hexdigest()
570                  req.download_info.info.hash = f"sha256={hash}"
571  
572          # For use in later processing,
573          # preserve the file path on the requirement.
574          if local_file:
575              req.local_file_path = local_file.path
576  
577          dist = _get_prepared_distribution(
578              req,
579              self.build_tracker,
580              self.finder,
581              self.build_isolation,
582              self.check_build_deps,
583          )
584          return dist
585  
586      def save_linked_requirement(self, req: InstallRequirement) -> None:
587          assert self.download_dir is not None
588          assert req.link is not None
589          link = req.link
590          if link.is_vcs or (link.is_existing_dir() and req.editable):
591              # Make a .zip of the source_dir we already created.
592              req.archive(self.download_dir)
593              return
594  
595          if link.is_existing_dir():
596              logger.debug(
597                  "Not copying link to destination directory "
598                  "since it is a directory: %s",
599                  link,
600              )
601              return
602          if req.local_file_path is None:
603              # No distribution was downloaded for this requirement.
604              return
605  
606          download_location = os.path.join(self.download_dir, link.filename)
607          if not os.path.exists(download_location):
608              shutil.copy(req.local_file_path, download_location)
609              download_path = display_path(download_location)
610              logger.info("Saved %s", download_path)

612      def prepare_editable_requirement(
613          self,
614          req: InstallRequirement,
615      ) -> BaseDistribution:
616          """Prepare an editable requirement."""
617          assert req.editable, "cannot prepare a non-editable req as editable"
618  
619          logger.info("Obtaining %s", req)
620  
621          with indent_log():
622              if self.require_hashes:
623                  raise InstallationError(
624                      "The editable requirement {} cannot be installed when "
625                      "requiring hashes, because there is no single file to "
626                      "hash.".format(req)
627                  )
628              req.ensure_has_source_dir(self.src_dir)
629              req.update_editable()
630              assert req.source_dir
631              req.download_info = direct_url_for_editable(req.unpacked_source_directory)
632  
633              dist = _get_prepared_distribution(
634                  req,
635                  self.build_tracker,
636                  self.finder,
637                  self.build_isolation,
638                  self.check_build_deps,
639              )
640  
641              req.check_if_exists(self.use_user_site)
642  
643          return dist

645      def prepare_installed_requirement(
646          self,
647          req: InstallRequirement,
648          skip_reason: str,
649      ) -> BaseDistribution:
650          """Prepare an already-installed requirement."""
651          assert req.satisfied_by, "req should have been satisfied but isn't"
652          assert skip_reason is not None, (
653              "did not get skip reason skipped but req.satisfied_by "
654              "is set to {}".format(req.satisfied_by)
655          )
656          logger.info(
657              "Requirement %s: %s (%s)", skip_reason, req, req.satisfied_by.version
658          )
659          with indent_log():
660              if self.require_hashes:
661                  logger.debug(
662                      "Since it is already installed, we are trusting this "
663                      "package without checking its hash. To ensure a "
664                      "completely repeatable environment, install into an "
665                      "empty virtualenv."
666                  )
667              return InstalledDistribution(req).get_metadata_distribution()
668
