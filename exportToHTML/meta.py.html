1    """Build metadata for a project using PEP 517 hooks. 
2    """
3    import argparse
4    import functools
5    import logging
6    import os
7    import shutil
8    import tempfile
9    
10   try:
11       import importlib.metadata as imp_meta
12   except ImportError:
13       import importlib_metadata as imp_meta
14   
15   try:
16       from zipfile import Path
17   except ImportError:
18       from zipp import Path
19   
20   from .build import compat_system, load_system, validate_system
21   from .dirtools import dir_to_zipfile
22   from .envbuild import BuildEnvironment
23   from .wrappers import Pep517HookCaller, quiet_subprocess_runner
24   
25   log = logging.getLogger(__name__)
26   
27   
28   def _prep_meta(hooks, env, dest):
29       reqs = hooks.get_requires_for_build_wheel({})
30       log.info('Got build requires: %s', reqs)
31   
32       env.pip_install(reqs)
33       log.info('Installed dynamic build dependencies')
34   
35       with tempfile.TemporaryDirectory() as td:
36           log.info('Trying to build metadata in %s', td)
37           filename = hooks.prepare_metadata_for_build_wheel(td, {})
38           source = os.path.join(td, filename)
39           shutil.move(source, os.path.join(dest, os.path.basename(filename)))
40   
41   
42   def build(source_dir='.', dest=None, system=None):
43       system = system or load_system(source_dir)
44       dest = os.path.join(source_dir, dest or 'dist')
45       os.makedirs(dest, exist_ok=True)
46       validate_system(system)
47       hooks = Pep517HookCaller(
48           source_dir, system['build-backend'], system.get('backend-path')
49       )
50   
51       with hooks.subprocess_runner(quiet_subprocess_runner):
52           with BuildEnvironment() as env:
53               env.pip_install(system['requires'])
54               _prep_meta(hooks, env, dest)
55   
56   
57   def build_as_zip(builder=build):
58       with tempfile.TemporaryDirectory() as out_dir:
59           builder(dest=out_dir)
60           return dir_to_zipfile(out_dir)
61   
62   
63   def load(root):
64       """ 
65       Given a source directory (root) of a package, 
66       return an importlib.metadata.Distribution object 
67       with metadata build from that package. 
68       """
69       root = os.path.expanduser(root)
70       system = compat_system(root)
71       builder = functools.partial(build, source_dir=root, system=system)
72       path = Path(build_as_zip(builder))
73       return imp_meta.PathDistribution(path)
74   
75   
76   parser = argparse.ArgumentParser()
77   parser.add_argument(
78       'source_dir',
79       help="A directory containing pyproject.toml",
80   )
81   parser.add_argument(
82       '--out-dir', '-o',
83       help="Destination in which to save the builds relative to source dir",
84   )
85   
86   
87   def main():
88       args = parser.parse_args()
89       build(args.source_dir, args.out_dir)
90   
91   
92   if __name__ == '__main__':
93       main()
