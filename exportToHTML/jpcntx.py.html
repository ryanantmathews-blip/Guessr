119  class JapaneseContextAnalysis:
120      NUM_OF_CATEGORY = 6
121      DONT_KNOW = -1
122      ENOUGH_REL_THRESHOLD = 100
123      MAX_REL_THRESHOLD = 1000
124      MINIMUM_DATA_THRESHOLD = 4
125  
126      def __init__(self):
127          self._total_rel = None
128          self._rel_sample = None
129          self._need_to_skip_char_num = None
130          self._last_char_order = None
131          self._done = None
132          self.reset()
133  
134      def reset(self):
135          self._total_rel = 0  # total sequence received
136          # category counters, each integer counts sequence in its category
137          self._rel_sample = [0] * self.NUM_OF_CATEGORY
138          # if last byte in current buffer is not the last byte of a character,
139          # we need to know how many bytes to skip in next buffer
140          self._need_to_skip_char_num = 0
141          self._last_char_order = -1  # The order of previous char
142          # If this flag is set to True, detection is done and conclusion has
143          # been made
144          self._done = False
145  
146      def feed(self, byte_str, num_bytes):
147          if self._done:
148              return
149  
150          # The buffer we got is byte oriented, and a character may span in more than one
151          # buffers. In case the last one or two byte in last buffer is not
152          # complete, we record how many byte needed to complete that character
153          # and skip these bytes here.  We can choose to record those bytes as
154          # well and analyse the character once it is complete, but since a
155          # character will not make much difference, by simply skipping
156          # this character will simply our logic and improve performance.
157          i = self._need_to_skip_char_num
158          while i < num_bytes:
159              order, char_len = self.get_order(byte_str[i : i + 2])
160              i += char_len
161              if i > num_bytes:
162                  self._need_to_skip_char_num = i - num_bytes
163                  self._last_char_order = -1
164              else:
165                  if (order != -1) and (self._last_char_order != -1):
166                      self._total_rel += 1
167                      if self._total_rel > self.MAX_REL_THRESHOLD:
168                          self._done = True
169                          break
170                      self._rel_sample[
171                          jp2_char_context[self._last_char_order][order]
172                      ] += 1
173                  self._last_char_order = order
174  
175      def got_enough_data(self):
176          return self._total_rel > self.ENOUGH_REL_THRESHOLD
177  
178      def get_confidence(self):
179          # This is just one way to calculate confidence. It works well for me.
180          if self._total_rel > self.MINIMUM_DATA_THRESHOLD:
181              return (self._total_rel - self._rel_sample[0]) / self._total_rel
182          return self.DONT_KNOW
183  
184      def get_order(self, _):
185          return -1, 1

190          super().__init__()
191          self._charset_name = "SHIFT_JIS"
192  
193      @property
194      def charset_name(self):
195          return self._charset_name

197      def get_order(self, byte_str):
198          if not byte_str:
199              return -1, 1
200          # find out current char's byte length
201          first_char = byte_str[0]
202          if (0x81 <= first_char <= 0x9F) or (0xE0 <= first_char <= 0xFC):
203              char_len = 2
204              if (first_char == 0x87) or (0xFA <= first_char <= 0xFC):
205                  self._charset_name = "CP932"

207              char_len = 1

209          # return its order if it is hiragana
210          if len(byte_str) > 1:
211              second_char = byte_str[1]
212              if (first_char == 202) and (0x9F <= second_char <= 0xF1):
213                  return second_char - 0x9F, char_len
214  
215          return -1, char_len
