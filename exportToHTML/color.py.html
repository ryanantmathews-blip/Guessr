299  @rich_repr
300  class Color(NamedTuple):
301      """Terminal color definition."""
302  
303      name: str
304      """The name of the color (typically the input to Color.parse)."""
305      type: ColorType
306      """The type of the color."""
307      number: Optional[int] = None
308      """The color number, if a standard color, or None."""
309      triplet: Optional[ColorTriplet] = None
310      """A triplet of color components, if an RGB color."""
311  
312      def __rich__(self) -> "Text":
313          """Dispays the actual color if Rich printed."""
314          from .style import Style
315          from .text import Text
316  
317          return Text.assemble(
318              f"<color {self.name!r} ({self.type.name.lower()})",
319              ("", Style(color=self)),
320              " >",
321          )
322  
323      def __rich_repr__(self) -> Result:
324          yield self.name
325          yield self.type
326          yield "number", self.number, None
327          yield "triplet", self.triplet, None

329      @property
330      def system(self) -> ColorSystem:
331          """Get the native color system for this color."""
332          if self.type == ColorType.DEFAULT:
333              return ColorSystem.STANDARD
334          return ColorSystem(int(self.type))

336      @property
337      def is_system_defined(self) -> bool:
338          """Check if the color is ultimately defined by the system."""
339          return self.system not in (ColorSystem.EIGHT_BIT, ColorSystem.TRUECOLOR)
340  
341      @property
342      def is_default(self) -> bool:
343          """Check if the color is a default color."""
344          return self.type == ColorType.DEFAULT

346      def get_truecolor(
347          self, theme: Optional["TerminalTheme"] = None, foreground: bool = True
348      ) -> ColorTriplet:
349          """Get an equivalent color triplet for this color. 
350   
351          Args: 
352              theme (TerminalTheme, optional): Optional terminal theme, or None to use default. Defaults to None. 
353              foreground (bool, optional): True for a foreground color, or False for background. Defaults to True. 
354   
355          Returns: 
356              ColorTriplet: A color triplet containing RGB components. 
357          """
358  
359          if theme is None:
360              theme = DEFAULT_TERMINAL_THEME
361          if self.type == ColorType.TRUECOLOR:
362              assert self.triplet is not None
363              return self.triplet
364          elif self.type == ColorType.EIGHT_BIT:
365              assert self.number is not None
366              return EIGHT_BIT_PALETTE[self.number]
367          elif self.type == ColorType.STANDARD:
368              assert self.number is not None
369              return theme.ansi_colors[self.number]
370          elif self.type == ColorType.WINDOWS:
371              assert self.number is not None
372              return WINDOWS_PALETTE[self.number]
373          else:  # self.type == ColorType.DEFAULT:
374              assert self.number is None
375              return theme.foreground_color if foreground else theme.background_color

377      @classmethod
378      def from_ansi(cls, number: int) -> "Color":
379          """Create a Color number from it's 8-bit ansi number. 
380   
381          Args: 
382              number (int): A number between 0-255 inclusive. 
383   
384          Returns: 
385              Color: A new Color instance. 
386          """
387          return cls(
388              name=f"color({number})",
389              type=(ColorType.STANDARD if number < 16 else ColorType.EIGHT_BIT),
390              number=number,
391          )

394      def from_triplet(cls, triplet: "ColorTriplet") -> "Color":
395          """Create a truecolor RGB color from a triplet of values. 
396   
397          Args: 
398              triplet (ColorTriplet): A color triplet containing red, green and blue components. 
399

405      @classmethod
406      def from_rgb(cls, red: float, green: float, blue: float) -> "Color":
407          """Create a truecolor from three color components in the range(0->255). 
408   
409          Args: 
410              red (float): Red component in range 0-255. 
411              green (float): Green component in range 0-255. 
412              blue (float): Blue component in range 0-255. 
413   
414          Returns: 
415              Color: A new color object. 
416          """
417          return cls.from_triplet(ColorTriplet(int(red), int(green), int(blue)))

428      @classmethod
429      @lru_cache(maxsize=1024)
430      def parse(cls, color: str) -> "Color":
431          """Parse a color definition."""
432          original_color = color
433          color = color.lower().strip()
434  
435          if color == "default":
436              return cls(color, type=ColorType.DEFAULT)
437  
438          color_number = ANSI_COLOR_NAMES.get(color)
439          if color_number is not None:
440              return cls(
441                  color,
442                  type=(ColorType.STANDARD if color_number < 16 else ColorType.EIGHT_BIT),
443                  number=color_number,
444              )
445  
446          color_match = RE_COLOR.match(color)
447          if color_match is None:
448              raise ColorParseError(f"{original_color!r} is not a valid color")
449  
450          color_24, color_8, color_rgb = color_match.groups()
451          if color_24:
452              triplet = ColorTriplet(
453                  int(color_24[0:2], 16), int(color_24[2:4], 16), int(color_24[4:6], 16)
454              )
455              return cls(color, ColorType.TRUECOLOR, triplet=triplet)
456  
457          elif color_8:
458              number = int(color_8)
459              if number > 255:
460                  raise ColorParseError(f"color number must be <= 255 in {color!r}")
461              return cls(
462                  color,
463                  type=(ColorType.STANDARD if number < 16 else ColorType.EIGHT_BIT),
464                  number=number,
465              )
466  
467          else:  #  color_rgb:
468              components = color_rgb.split(",")
469              if len(components) != 3:
470                  raise ColorParseError(
471                      f"expected three components in {original_color!r}"
472                  )
473              red, green, blue = components
474              triplet = ColorTriplet(int(red), int(green), int(blue))
475              if not all(component <= 255 for component in triplet):
476                  raise ColorParseError(
477                      f"color components must be <= 255 in {original_color!r}"
478                  )
479              return cls(color, ColorType.TRUECOLOR, triplet=triplet)

481      @lru_cache(maxsize=1024)
482      def get_ansi_codes(self, foreground: bool = True) -> Tuple[str, ...]:
483          """Get the ANSI escape codes for this color."""
484          _type = self.type
485          if _type == ColorType.DEFAULT:
486              return ("39" if foreground else "49",)
487  
488          elif _type == ColorType.WINDOWS:
489              number = self.number
490              assert number is not None
491              fore, back = (30, 40) if number < 8 else (82, 92)
492              return (str(fore + number if foreground else back + number),)
493  
494          elif _type == ColorType.STANDARD:
495              number = self.number
496              assert number is not None
497              fore, back = (30, 40) if number < 8 else (82, 92)
498              return (str(fore + number if foreground else back + number),)
499  
500          elif _type == ColorType.EIGHT_BIT:
501              assert self.number is not None
502              return ("38" if foreground else "48", "5", str(self.number))
503  
504          else:  # self.standard == ColorStandard.TRUECOLOR:
505              assert self.triplet is not None
506              red, green, blue = self.triplet
507              return ("38" if foreground else "48", "2", str(red), str(green), str(blue))

509      @lru_cache(maxsize=1024)
510      def downgrade(self, system: ColorSystem) -> "Color":
511          """Downgrade a color system to a system with fewer colors."""
512  
513          if self.type in [ColorType.DEFAULT, system]:
514              return self
515          # Convert to 8-bit color from truecolor color
516          if system == ColorSystem.EIGHT_BIT and self.system == ColorSystem.TRUECOLOR:
517              assert self.triplet is not None
518              red, green, blue = self.triplet.normalized
519              _h, l, s = rgb_to_hls(red, green, blue)
520              # If saturation is under 10% assume it is grayscale
521              if s < 0.1:
522                  gray = round(l * 25.0)
523                  if gray == 0:
524                      color_number = 16
525                  elif gray == 25:
526                      color_number = 231
527                  else:
528                      color_number = 231 + gray
529                  return Color(self.name, ColorType.EIGHT_BIT, number=color_number)
530  
531              color_number = (
532                  16 + 36 * round(red * 5.0) + 6 * round(green * 5.0) + round(blue * 5.0)
533              )
534              return Color(self.name, ColorType.EIGHT_BIT, number=color_number)
535  
536          # Convert to standard from truecolor or 8-bit
537          elif system == ColorSystem.STANDARD:
538              if self.system == ColorSystem.TRUECOLOR:
539                  assert self.triplet is not None
540                  triplet = self.triplet
541              else:  # self.system == ColorSystem.EIGHT_BIT
542                  assert self.number is not None
543                  triplet = ColorTriplet(*EIGHT_BIT_PALETTE[self.number])
544  
545              color_number = STANDARD_PALETTE.match(triplet)
546              return Color(self.name, ColorType.STANDARD, number=color_number)
547  
548          elif system == ColorSystem.WINDOWS:
549              if self.system == ColorSystem.TRUECOLOR:
550                  assert self.triplet is not None
551                  triplet = self.triplet
552              else:  # self.system == ColorSystem.EIGHT_BIT
553                  assert self.number is not None
554                  if self.number < 16:
555                      return Color(self.name, ColorType.WINDOWS, number=self.number)
556                  triplet = ColorTriplet(*EIGHT_BIT_PALETTE[self.number])
557  
558              color_number = WINDOWS_PALETTE.match(triplet)
559              return Color(self.name, ColorType.WINDOWS, number=color_number)
560  
561          return self

564  def parse_rgb_hex(hex_color: str) -> ColorTriplet:
565      """Parse six hex characters in to RGB triplet."""
566      assert len(hex_color) == 6, "must be 6 characters"
567      color = ColorTriplet(
568          int(hex_color[0:2], 16), int(hex_color[2:4], 16), int(hex_color[4:6], 16)
569      )
570      return color

573  def blend_rgb(
574      color1: ColorTriplet, color2: ColorTriplet, cross_fade: float = 0.5
575  ) -> ColorTriplet:
576      """Blend one RGB color in to another."""
577      r1, g1, b1 = color1
578      r2, g2, b2 = color2
579      new_color = ColorTriplet(
580          int(r1 + (r2 - r1) * cross_fade),
581          int(g1 + (g2 - g1) * cross_fade),
582          int(b1 + (b2 - b1) * cross_fade),
583      )
584      return new_color
