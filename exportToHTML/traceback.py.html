36   def install(
37       *,
38       console: Optional[Console] = None,
39       width: Optional[int] = 100,
40       extra_lines: int = 3,
41       theme: Optional[str] = None,
42       word_wrap: bool = False,
43       show_locals: bool = False,
44       indent_guides: bool = True,
45       suppress: Iterable[Union[str, ModuleType]] = (),
46       max_frames: int = 100,
47   ) -> Callable[[Type[BaseException], BaseException, Optional[TracebackType]], Any]:
48       """Install a rich traceback handler. 
49    
50       Once installed, any tracebacks will be printed with syntax highlighting and rich formatting. 
51    
52    
53       Args: 
54           console (Optional[Console], optional): Console to write exception to. Default uses internal Console instance. 
55           width (Optional[int], optional): Width (in characters) of traceback. Defaults to 100. 
56           extra_lines (int, optional): Extra lines of code. Defaults to 3. 
57           theme (Optional[str], optional): Pygments theme to use in traceback. Defaults to ``None`` which will pick 
58               a theme appropriate for the platform. 
59           word_wrap (bool, optional): Enable word wrapping of long lines. Defaults to False. 
60           show_locals (bool, optional): Enable display of local variables. Defaults to False. 
61           indent_guides (bool, optional): Enable indent guides in code and locals. Defaults to True. 
62           suppress (Sequence[Union[str, ModuleType]]): Optional sequence of modules or paths to exclude from traceback. 
63    
64       Returns: 
65           Callable: The previous exception handler that was replaced. 
66    
67       """
68       traceback_console = Console(file=sys.stderr) if console is None else console

70       def excepthook(
71           type_: Type[BaseException],
72           value: BaseException,
73           traceback: Optional[TracebackType],
74       ) -> None:
75           traceback_console.print(
76               Traceback.from_exception(
77                   type_,
78                   value,
79                   traceback,
80                   width=width,
81                   extra_lines=extra_lines,
82                   theme=theme,
83                   word_wrap=word_wrap,
84                   show_locals=show_locals,
85                   indent_guides=indent_guides,
86                   suppress=suppress,
87                   max_frames=max_frames,
88               )
89           )

91       def ipy_excepthook_closure(ip: Any) -> None:  # pragma: no cover
92           tb_data = {}  # store information about showtraceback call
93           default_showtraceback = ip.showtraceback  # keep reference of default traceback
94   
95           def ipy_show_traceback(*args: Any, **kwargs: Any) -> None:
96               """wrap the default ip.showtraceback to store info for ip._showtraceback"""
97               nonlocal tb_data
98               tb_data = kwargs
99               default_showtraceback(*args, **kwargs)
100  
101          def ipy_display_traceback(
102              *args: Any, is_syntax: bool = False, **kwargs: Any
103          ) -> None:
104              """Internally called traceback from ip._showtraceback"""
105              nonlocal tb_data
106              exc_tuple = ip._get_exc_info()
107  
108              # do not display trace on syntax error
109              tb: Optional[TracebackType] = None if is_syntax else exc_tuple[2]
110  
111              # determine correct tb_offset
112              compiled = tb_data.get("running_compiled_code", False)
113              tb_offset = tb_data.get("tb_offset", 1 if compiled else 0)
114              # remove ipython internal frames from trace with tb_offset
115              for _ in range(tb_offset):
116                  if tb is None:
117                      break
118                  tb = tb.tb_next
119  
120              excepthook(exc_tuple[0], exc_tuple[1], tb)
121              tb_data = {}  # clear data upon usage
122  
123          # replace _showtraceback instead of showtraceback to allow ipython features such as debugging to work
124          # this is also what the ipython docs recommends to modify when subclassing InteractiveShell
125          ip._showtraceback = ipy_display_traceback
126          # add wrapper to capture tb_data
127          ip.showtraceback = ipy_show_traceback
128          ip.showsyntaxerror = lambda *args, **kwargs: ipy_display_traceback(
129              *args, is_syntax=True, **kwargs
130          )
131  
132      try:  # pragma: no cover
133          # if within ipython, use customized traceback
134          ip = get_ipython()  # type: ignore[name-defined]
135          ipy_excepthook_closure(ip)
136          return sys.excepthook
137      except Exception:
138          # otherwise use default system hook
139          old_excepthook = sys.excepthook
140          sys.excepthook = excepthook
141          return old_excepthook

182   
183      Args: 
184          trace (Trace, optional): A `Trace` object produced from `extract`. Defaults to None, which uses

186          width (Optional[int], optional): Number of characters used to traceback. Defaults to 100.

192          locals_max_length (int, optional): Maximum length of containers before abbreviating, or None for no abbreviation. 
193              Defaults to 10. 
194          locals_max_string (int, optional): Maximum length of string before truncating, or None to disable. Defaults to 80. 
195          suppress (Sequence[Union[str, ModuleType]]): Optional sequence of modules or paths to exclude from traceback. 
196          max_frames (int): Maximum number of frames to show in a traceback, 0 for no maximum. Defaults to 100.

208      def __init__(
209          self,
210          trace: Optional[Trace] = None,
211          width: Optional[int] = 100,
212          extra_lines: int = 3,
213          theme: Optional[str] = None,
214          word_wrap: bool = False,
215          show_locals: bool = False,
216          indent_guides: bool = True,
217          locals_max_length: int = LOCALS_MAX_LENGTH,
218          locals_max_string: int = LOCALS_MAX_STRING,
219          suppress: Iterable[Union[str, ModuleType]] = (),
220          max_frames: int = 100,
221      ):
222          if trace is None:
223              exc_type, exc_value, traceback = sys.exc_info()
224              if exc_type is None or exc_value is None or traceback is None:
225                  raise ValueError(
226                      "Value for 'trace' required if not called in except: block"
227                  )
228              trace = self.extract(
229                  exc_type, exc_value, traceback, show_locals=show_locals
230              )
231          self.trace = trace
232          self.width = width
233          self.extra_lines = extra_lines
234          self.theme = Syntax.get_theme(theme or "ansi_dark")
235          self.word_wrap = word_wrap
236          self.show_locals = show_locals
237          self.indent_guides = indent_guides
238          self.locals_max_length = locals_max_length
239          self.locals_max_string = locals_max_string
240  
241          self.suppress: Sequence[str] = []
242          for suppress_entity in suppress:
243              if not isinstance(suppress_entity, str):
244                  assert (
245                      suppress_entity.__file__ is not None
246                  ), f"{suppress_entity!r} must be a module with '__file__' attribute"
247                  path = os.path.dirname(suppress_entity.__file__)
248              else:
249                  path = suppress_entity
250              path = os.path.normpath(os.path.abspath(path))
251              self.suppress.append(path)
252          self.max_frames = max(4, max_frames) if max_frames > 0 else 0

274              exc_type (Type[BaseException]): Exception type. 
275              exc_value (BaseException): Exception value. 
276              traceback (TracebackType): Python Traceback object. 
277              width (Optional[int], optional): Number of characters used to traceback. Defaults to 100. 
278              extra_lines (int, optional): Additional lines of code to render. Defaults to 3. 
279              theme (str, optional): Override pygments theme used in traceback. 
280              word_wrap (bool, optional): Enable word wrapping of long lines. Defaults to False. 
281              show_locals (bool, optional): Enable display of local variables. Defaults to False. 
282              indent_guides (bool, optional): Enable indent guides in code and locals. Defaults to True. 
283              locals_max_length (int, optional): Maximum length of containers before abbreviating, or None for no abbreviation. 
284                  Defaults to 10. 
285              locals_max_string (int, optional): Maximum length of string before truncating, or None to disable. Defaults to 80. 
286              suppress (Iterable[Union[str, ModuleType]]): Optional sequence of modules or paths to exclude from traceback. 
287              max_frames (int): Maximum number of frames to show in a traceback, 0 for no maximum. Defaults to 100. 
288   
289          Returns: 
290              Traceback: A Traceback instance that may be printed. 
291          """
292          rich_traceback = cls.extract(
293              exc_type, exc_value, traceback, show_locals=show_locals
294          )
295          return cls(
296              rich_traceback,
297              width=width,
298              extra_lines=extra_lines,
299              theme=theme,
300              word_wrap=word_wrap,
301              show_locals=show_locals,

305              suppress=suppress,
306              max_frames=max_frames,

309      @classmethod
310      def extract(
311          cls,
312          exc_type: Type[BaseException],
313          exc_value: BaseException,
314          traceback: Optional[TracebackType],
315          show_locals: bool = False,
316          locals_max_length: int = LOCALS_MAX_LENGTH,
317          locals_max_string: int = LOCALS_MAX_STRING,
318      ) -> Trace:
319          """Extract traceback information. 
320   
321          Args: 
322              exc_type (Type[BaseException]): Exception type. 
323              exc_value (BaseException): Exception value. 
324              traceback (TracebackType): Python Traceback object. 
325              show_locals (bool, optional): Enable display of local variables. Defaults to False. 
326              locals_max_length (int, optional): Maximum length of containers before abbreviating, or None for no abbreviation. 
327                  Defaults to 10. 
328              locals_max_string (int, optional): Maximum length of string before truncating, or None to disable. Defaults to 80. 
329   
330          Returns: 
331              Trace: A Trace instance which you can use to construct a `Traceback`. 
332          """
333  
334          stacks: List[Stack] = []
335          is_cause = False
336  
337          from pip._vendor.rich import _IMPORT_CWD
338  
339          def safe_str(_object: Any) -> str:
340              """Don't allow exceptions from __str__ to propegate."""
341              try:
342                  return str(_object)
343              except Exception:
344                  return "<exception str() failed>"
345  
346          while True:
347              stack = Stack(
348                  exc_type=safe_str(exc_type.__name__),
349                  exc_value=safe_str(exc_value),
350                  is_cause=is_cause,
351              )
352  
353              if isinstance(exc_value, SyntaxError):
354                  stack.syntax_error = _SyntaxError(
355                      offset=exc_value.offset or 0,
356                      filename=exc_value.filename or "?",
357                      lineno=exc_value.lineno or 0,
358                      line=exc_value.text or "",
359                      msg=exc_value.msg,
360                  )
361  
362              stacks.append(stack)
363              append = stack.frames.append
364  
365              for frame_summary, line_no in walk_tb(traceback):
366                  filename = frame_summary.f_code.co_filename
367                  if filename and not filename.startswith("<"):
368                      if not os.path.isabs(filename):
369                          filename = os.path.join(_IMPORT_CWD, filename)
370                  if frame_summary.f_locals.get("_rich_traceback_omit", False):
371                      continue
372                  frame = Frame(
373                      filename=filename or "?",
374                      lineno=line_no,
375                      name=frame_summary.f_code.co_name,
376                      locals={
377                          key: pretty.traverse(
378                              value,
379                              max_length=locals_max_length,
380                              max_string=locals_max_string,
381                          )
382                          for key, value in frame_summary.f_locals.items()
383                      }
384                      if show_locals
385                      else None,
386                  )
387                  append(frame)
388                  if frame_summary.f_locals.get("_rich_traceback_guard", False):
389                      del stack.frames[:]
390  
391              cause = getattr(exc_value, "__cause__", None)
392              if cause and cause.__traceback__:
393                  exc_type = cause.__class__
394                  exc_value = cause
395                  traceback = cause.__traceback__
396                  is_cause = True
397                  continue
398  
399              cause = exc_value.__context__
400              if (
401                  cause
402                  and cause.__traceback__
403                  and not getattr(exc_value, "__suppress_context__", False)
404              ):
405                  exc_type = cause.__class__
406                  exc_value = cause
407                  traceback = cause.__traceback__
408                  is_cause = False
409                  continue
410              # No cover, code is reached but coverage doesn't recognize it.
411              break  # pragma: no cover
412  
413          trace = Trace(stacks=stacks)
414          return trace

416      def __rich_console__(
417          self, console: Console, options: ConsoleOptions
418      ) -> RenderResult:
419          theme = self.theme
420          background_style = theme.get_background_style()
421          token_style = theme.get_style_for_token
422  
423          traceback_theme = Theme(
424              {
425                  "pretty": token_style(TextToken),
426                  "pygments.text": token_style(Token),
427                  "pygments.string": token_style(String),
428                  "pygments.function": token_style(Name.Function),
429                  "pygments.number": token_style(Number),
430                  "repr.indent": token_style(Comment) + Style(dim=True),
431                  "repr.str": token_style(String),
432                  "repr.brace": token_style(TextToken) + Style(bold=True),
433                  "repr.number": token_style(Number),
434                  "repr.bool_true": token_style(Keyword.Constant),
435                  "repr.bool_false": token_style(Keyword.Constant),
436                  "repr.none": token_style(Keyword.Constant),
437                  "scope.border": token_style(String.Delimiter),
438                  "scope.equals": token_style(Operator),
439                  "scope.key": token_style(Name),
440                  "scope.key.special": token_style(Name.Constant) + Style(dim=True),
441              },
442              inherit=False,
443          )
444  
445          highlighter = ReprHighlighter()
446          for last, stack in loop_last(reversed(self.trace.stacks)):
447              if stack.frames:
448                  stack_renderable: ConsoleRenderable = Panel(
449                      self._render_stack(stack),
450                      title="[traceback.title]Traceback [dim](most recent call last)",
451                      style=background_style,
452                      border_style="traceback.border",
453                      expand=True,
454                      padding=(0, 1),
455                  )
456                  stack_renderable = Constrain(stack_renderable, self.width)
457                  with console.use_theme(traceback_theme):
458                      yield stack_renderable
459              if stack.syntax_error is not None:
460                  with console.use_theme(traceback_theme):
461                      yield Constrain(
462                          Panel(
463                              self._render_syntax_error(stack.syntax_error),
464                              style=background_style,
465                              border_style="traceback.border.syntax_error",
466                              expand=True,
467                              padding=(0, 1),
468                              width=self.width,
469                          ),
470                          self.width,
471                      )
472                  yield Text.assemble(
473                      (f"{stack.exc_type}: ", "traceback.exc_type"),
474                      highlighter(stack.syntax_error.msg),
475                  )
476              elif stack.exc_value:
477                  yield Text.assemble(
478                      (f"{stack.exc_type}: ", "traceback.exc_type"),
479                      highlighter(stack.exc_value),
480                  )
481              else:
482                  yield Text.assemble((f"{stack.exc_type}", "traceback.exc_type"))
483  
484              if not last:
485                  if stack.is_cause:
486                      yield Text.from_markup(
487                          "\n[i]The above exception was the direct cause of the following exception:\n",
488                      )
489                  else:
490                      yield Text.from_markup(
491                          "\n[i]During handling of the above exception, another exception occurred:\n",
492                      )

494      @group()
495      def _render_syntax_error(self, syntax_error: _SyntaxError) -> RenderResult:
496          highlighter = ReprHighlighter()
497          path_highlighter = PathHighlighter()
498          if syntax_error.filename != "<stdin>":
499              text = Text.assemble(
500                  (f" {syntax_error.filename}", "pygments.string"),
501                  (":", "pygments.text"),
502                  (str(syntax_error.lineno), "pygments.number"),
503                  style="pygments.text",
504              )
505              yield path_highlighter(text)
506          syntax_error_text = highlighter(syntax_error.line.rstrip())
507          syntax_error_text.no_wrap = True
508          offset = min(syntax_error.offset - 1, len(syntax_error_text))
509          syntax_error_text.stylize("bold underline", offset, offset)
510          syntax_error_text += Text.from_markup(
511              "\n" + " " * offset + "[traceback.offset][/]",
512              style="pygments.text",
513          )
514          yield syntax_error_text

516      @classmethod
517      def _guess_lexer(cls, filename: str, code: str) -> str:
518          ext = os.path.splitext(filename)[-1]
519          if not ext:
520              # No extension, look at first line to see if it is a hashbang
521              # Note, this is an educated guess and not a guarantee
522              # If it fails, the only downside is that the code is highlighted strangely
523              new_line_index = code.index("\n")
524              first_line = code[:new_line_index] if new_line_index != -1 else code
525              if first_line.startswith("#!") and "python" in first_line.lower():
526                  return "python"
527          try:
528              return cls.LEXERS.get(ext) or guess_lexer_for_filename(filename, code).name
529          except ClassNotFound:
530              return "text"

532      @group()
533      def _render_stack(self, stack: Stack) -> RenderResult:
534          path_highlighter = PathHighlighter()
535          theme = self.theme
536          code_cache: Dict[str, str] = {}
537  
538          def read_code(filename: str) -> str:
539              """Read files, and cache results on filename. 
540   
541              Args: 
542                  filename (str): Filename to read 
543   
544              Returns: 
545                  str: Contents of file 
546              """
547              code = code_cache.get(filename)
548              if code is None:
549                  with open(
550                      filename, "rt", encoding="utf-8", errors="replace"
551                  ) as code_file:
552                      code = code_file.read()
553                  code_cache[filename] = code
554              return code
555  
556          def render_locals(frame: Frame) -> Iterable[ConsoleRenderable]:
557              if frame.locals:
558                  yield render_scope(
559                      frame.locals,
560                      title="locals",
561                      indent_guides=self.indent_guides,
562                      max_length=self.locals_max_length,
563                      max_string=self.locals_max_string,
564                  )
565  
566          exclude_frames: Optional[range] = None
567          if self.max_frames != 0:
568              exclude_frames = range(
569                  self.max_frames // 2,
570                  len(stack.frames) - self.max_frames // 2,
571              )
572  
573          excluded = False
574          for frame_index, frame in enumerate(stack.frames):
575  
576              if exclude_frames and frame_index in exclude_frames:
577                  excluded = True
578                  continue
579  
580              if excluded:
581                  assert exclude_frames is not None
582                  yield Text(
583                      f"\n... {len(exclude_frames)} frames hidden ...",
584                      justify="center",
585                      style="traceback.error",
586                  )
587                  excluded = False
588  
589              first = frame_index == 0
590              frame_filename = frame.filename
591              suppressed = any(frame_filename.startswith(path) for path in self.suppress)
592  
593              text = Text.assemble(
594                  path_highlighter(Text(frame.filename, style="pygments.string")),
595                  (":", "pygments.text"),
596                  (str(frame.lineno), "pygments.number"),
597                  " in ",
598                  (frame.name, "pygments.function"),
599                  style="pygments.text",
600              )
601              if not frame.filename.startswith("<") and not first:
602                  yield ""
603              yield text
604              if frame.filename.startswith("<"):
605                  yield from render_locals(frame)
606                  continue
607              if not suppressed:
608                  try:
609                      code = read_code(frame.filename)
610                      lexer_name = self._guess_lexer(frame.filename, code)
611                      syntax = Syntax(
612                          code,
613                          lexer_name,
614                          theme=theme,
615                          line_numbers=True,
616                          line_range=(
617                              frame.lineno - self.extra_lines,
618                              frame.lineno + self.extra_lines,
619                          ),
620                          highlight_lines={frame.lineno},
621                          word_wrap=self.word_wrap,
622                          code_width=88,
623                          indent_guides=self.indent_guides,
624                          dedent=False,
625                      )
626                      yield ""
627                  except Exception as error:
628                      yield Text.assemble(
629                          (f"\n{error}", "traceback.error"),
630                      )
631                  else:
632                      yield (
633                          Columns(
634                              [
635                                  syntax,
636                                  *render_locals(frame),
637                              ],
638                              padding=1,
639                          )
640                          if frame.locals
641                          else syntax
642                      )
