20   def _script_names(
21       bin_dir: str, script_name: str, is_gui: bool
22   ) -> Generator[str, None, None]:
23       """Create the fully qualified name of the files created by 
24       {console,gui}_scripts for the given ``dist``. 
25       Returns the list of file names 
26       """
27       exe_name = os.path.join(bin_dir, script_name)
28       yield exe_name
29       if not WINDOWS:
30           return
31       yield f"{exe_name}.exe"
32       yield f"{exe_name}.exe.manifest"
33       if is_gui:
34           yield f"{exe_name}-script.pyw"
35       else:
36           yield f"{exe_name}-script.py"

39   def _unique(
40       fn: Callable[..., Generator[Any, None, None]]
41   ) -> Callable[..., Generator[Any, None, None]]:
42       @functools.wraps(fn)
43       def unique(*args: Any, **kw: Any) -> Generator[Any, None, None]:
44           seen: Set[Any] = set()
45           for item in fn(*args, **kw):
46               if item not in seen:
47                   seen.add(item)
48                   yield item
49   
50       return unique

54   def uninstallation_paths(dist: BaseDistribution) -> Generator[str, None, None]:
55       """ 
56       Yield all the uninstallation paths for dist based on RECORD-without-.py[co] 
57    
58       Yield paths to all the files in RECORD. For each .py file in RECORD, add 
59       the .pyc and .pyo in the same directory. 
60    
61       UninstallPathSet.add() takes care of the __pycache__ .py[co]. 
62    
63       If RECORD is not found, raises UninstallationError, 
64       with possible information from the INSTALLER file. 
65    
66       https://packaging.python.org/specifications/recording-installed-packages/ 
67       """
68       location = dist.location
69       assert location is not None, "not installed"
70   
71       entries = dist.iter_declared_entries()
72       if entries is None:
73           msg = "Cannot uninstall {dist}, RECORD file not found.".format(dist=dist)
74           installer = dist.installer
75           if not installer or installer == "pip":
76               dep = "{}=={}".format(dist.raw_name, dist.version)
77               msg += (
78                   " You might be able to recover from this via: "
79                   "'pip install --force-reinstall --no-deps {}'.".format(dep)
80               )
81           else:
82               msg += " Hint: The package was installed by {}.".format(installer)
83           raise UninstallationError(msg)
84   
85       for entry in entries:
86           path = os.path.join(location, entry)
87           yield path
88           if path.endswith(".py"):
89               dn, fn = os.path.split(path)
90               base = fn[:-3]
91               path = os.path.join(dn, base + ".pyc")
92               yield path
93               path = os.path.join(dn, base + ".pyo")
94               yield path

97   def compact(paths: Iterable[str]) -> Set[str]:
98       """Compact a path set to contain the minimal number of paths 
99       necessary to contain all paths in the set. If /a/path/ and 
100      /a/path/to/a/file.txt are both in the set, leave only the 
101      shorter path."""
102  
103      sep = os.path.sep
104      short_paths: Set[str] = set()
105      for path in sorted(paths, key=len):
106          should_skip = any(
107              path.startswith(shortpath.rstrip("*"))
108              and path[len(shortpath.rstrip("*").rstrip(sep))] == sep
109              for shortpath in short_paths
110          )
111          if not should_skip:
112              short_paths.add(path)
113      return short_paths

116  def compress_for_rename(paths: Iterable[str]) -> Set[str]:
117      """Returns a set containing the paths that need to be renamed. 
118   
119      This set may include directories when the original sequence of paths 
120      included every file on disk. 
121      """
122      case_map = {os.path.normcase(p): p for p in paths}
123      remaining = set(case_map)
124      unchecked = sorted({os.path.split(p)[0] for p in case_map.values()}, key=len)
125      wildcards: Set[str] = set()
126  
127      def norm_join(*a: str) -> str:
128          return os.path.normcase(os.path.join(*a))
129  
130      for root in unchecked:
131          if any(os.path.normcase(root).startswith(w) for w in wildcards):
132              # This directory has already been handled.
133              continue
134  
135          all_files: Set[str] = set()
136          all_subdirs: Set[str] = set()
137          for dirname, subdirs, files in os.walk(root):
138              all_subdirs.update(norm_join(root, dirname, d) for d in subdirs)
139              all_files.update(norm_join(root, dirname, f) for f in files)
140          # If all the files we found are in our remaining set of files to
141          # remove, then remove them from the latter set and add a wildcard
142          # for the directory.
143          if not (all_files - remaining):
144              remaining.difference_update(all_files)
145              wildcards.add(root + os.sep)
146  
147      return set(map(case_map.__getitem__, remaining)) | wildcards

150  def compress_for_output_listing(paths: Iterable[str]) -> Tuple[Set[str], Set[str]]:
151      """Returns a tuple of 2 sets of which paths to display to user 
152   
153      The first set contains paths that would be deleted. Files of a package 
154      are not added and the top-level directory of the package has a '*' added 
155      at the end - to signify that all it's contents are removed. 
156   
157      The second set contains files that would have been skipped in the above 
158      folders. 
159      """
160  
161      will_remove = set(paths)
162      will_skip = set()
163  
164      # Determine folders and files
165      folders = set()
166      files = set()
167      for path in will_remove:
168          if path.endswith(".pyc"):
169              continue
170          if path.endswith("__init__.py") or ".dist-info" in path:
171              folders.add(os.path.dirname(path))
172          files.add(path)
173  
174      # probably this one https://github.com/python/mypy/issues/390
175      _normcased_files = set(map(os.path.normcase, files))  # type: ignore
176  
177      folders = compact(folders)
178  
179      # This walks the tree using os.walk to not miss extra folders
180      # that might get added.
181      for folder in folders:
182          for dirpath, _, dirfiles in os.walk(folder):
183              for fname in dirfiles:
184                  if fname.endswith(".pyc"):
185                      continue
186  
187                  file_ = os.path.join(dirpath, fname)
188                  if (
189                      os.path.isfile(file_)
190                      and os.path.normcase(file_) not in _normcased_files
191                  ):
192                      # We are skipping this file. Add it to the set.
193                      will_skip.add(file_)
194  
195      will_remove = files | {os.path.join(folder, "*") for folder in folders}
196  
197      return will_remove, will_skip

200  class StashedUninstallPathSet:
201      """A set of file rename operations to stash files while 
202      tentatively uninstalling them."""
203  
204      def __init__(self) -> None:
205          # Mapping from source file root to [Adjacent]TempDirectory
206          # for files under that directory.
207          self._save_dirs: Dict[str, TempDirectory] = {}
208          # (old path, new path) tuples for each move that may need
209          # to be undone.
210          self._moves: List[Tuple[str, str]] = []
211  
212      def _get_directory_stash(self, path: str) -> str:
213          """Stashes a directory. 
214   
215          Directories are stashed adjacent to their original location if 
216          possible, or else moved/copied into the user's temp dir."""
217  
218          try:
219              save_dir: TempDirectory = AdjacentTempDirectory(path)
220          except OSError:
221              save_dir = TempDirectory(kind="uninstall")
222          self._save_dirs[os.path.normcase(path)] = save_dir
223  
224          return save_dir.path
225  
226      def _get_file_stash(self, path: str) -> str:
227          """Stashes a file. 
228   
229          If no root has been provided, one will be created for the directory 
230          in the user's temp directory."""
231          path = os.path.normcase(path)
232          head, old_head = os.path.dirname(path), None
233          save_dir = None
234  
235          while head != old_head:
236              try:
237                  save_dir = self._save_dirs[head]
238                  break
239              except KeyError:
240                  pass
241              head, old_head = os.path.dirname(head), head
242          else:
243              # Did not find any suitable root
244              head = os.path.dirname(path)
245              save_dir = TempDirectory(kind="uninstall")
246              self._save_dirs[head] = save_dir
247  
248          relpath = os.path.relpath(path, head)
249          if relpath and relpath != os.path.curdir:
250              return os.path.join(save_dir.path, relpath)
251          return save_dir.path
252  
253      def stash(self, path: str) -> str:
254          """Stashes the directory or file and returns its new location. 
255          Handle symlinks as files to avoid modifying the symlink targets. 
256          """
257          path_is_dir = os.path.isdir(path) and not os.path.islink(path)
258          if path_is_dir:
259              new_path = self._get_directory_stash(path)
260          else:
261              new_path = self._get_file_stash(path)
262  
263          self._moves.append((path, new_path))
264          if path_is_dir and os.path.isdir(new_path):
265              # If we're moving a directory, we need to
266              # remove the destination first or else it will be
267              # moved to inside the existing directory.
268              # We just created new_path ourselves, so it will
269              # be removable.
270              os.rmdir(new_path)
271          renames(path, new_path)
272          return new_path
273  
274      def commit(self) -> None:
275          """Commits the uninstall by removing stashed files."""
276          for _, save_dir in self._save_dirs.items():
277              save_dir.cleanup()
278          self._moves = []
279          self._save_dirs = {}
280  
281      def rollback(self) -> None:
282          """Undoes the uninstall by moving stashed files back."""
283          for p in self._moves:
284              logger.info("Moving to %s\n from %s", *p)
285  
286          for new_path, path in self._moves:
287              try:
288                  logger.debug("Replacing %s from %s", new_path, path)
289                  if os.path.isfile(new_path) or os.path.islink(new_path):
290                      os.unlink(new_path)
291                  elif os.path.isdir(new_path):
292                      rmtree(new_path)
293                  renames(path, new_path)
294              except OSError as ex:
295                  logger.error("Failed to restore %s", new_path)
296                  logger.debug("Exception: %s", ex)
297  
298          self.commit()
299  
300      @property
301      def can_rollback(self) -> bool:
302          return bool(self._moves)

305  class UninstallPathSet:
306      """A set of file paths to be removed in the uninstallation of a 
307      requirement."""
308  
309      def __init__(self, dist: BaseDistribution) -> None:
310          self._paths: Set[str] = set()
311          self._refuse: Set[str] = set()
312          self._pth: Dict[str, UninstallPthEntries] = {}
313          self._dist = dist
314          self._moved_paths = StashedUninstallPathSet()
315  
316      def _permitted(self, path: str) -> bool:
317          """ 
318          Return True if the given path is one we are permitted to 
319          remove/modify, False otherwise. 
320   
321          """
322          return is_local(path)
323  
324      def add(self, path: str) -> None:
325          head, tail = os.path.split(path)
326  
327          # we normalize the head to resolve parent directory symlinks, but not
328          # the tail, since we only want to uninstall symlinks, not their targets
329          path = os.path.join(normalize_path(head), os.path.normcase(tail))
330  
331          if not os.path.exists(path):
332              return
333          if self._permitted(path):
334              self._paths.add(path)
335          else:
336              self._refuse.add(path)
337  
338          # __pycache__ files can show up after 'installed-files.txt' is created,
339          # due to imports
340          if os.path.splitext(path)[1] == ".py":
341              self.add(cache_from_source(path))
342  
343      def add_pth(self, pth_file: str, entry: str) -> None:
344          pth_file = normalize_path(pth_file)
345          if self._permitted(pth_file):
346              if pth_file not in self._pth:
347                  self._pth[pth_file] = UninstallPthEntries(pth_file)
348              self._pth[pth_file].add(entry)
349          else:
350              self._refuse.add(pth_file)
351  
352      def remove(self, auto_confirm: bool = False, verbose: bool = False) -> None:
353          """Remove paths in ``self._paths`` with confirmation (unless 
354          ``auto_confirm`` is True)."""
355  
356          if not self._paths:
357              logger.info(
358                  "Can't uninstall '%s'. No files were found to uninstall.",
359                  self._dist.raw_name,
360              )
361              return
362  
363          dist_name_version = f"{self._dist.raw_name}-{self._dist.version}"
364          logger.info("Uninstalling %s:", dist_name_version)
365  
366          with indent_log():
367              if auto_confirm or self._allowed_to_proceed(verbose):
368                  moved = self._moved_paths
369  
370                  for_rename = compress_for_rename(self._paths)
371  
372                  for path in sorted(compact(for_rename)):
373                      moved.stash(path)
374                      logger.verbose("Removing file or directory %s", path)
375  
376                  for pth in self._pth.values():
377                      pth.remove()
378  
379                  logger.info("Successfully uninstalled %s", dist_name_version)
380  
381      def _allowed_to_proceed(self, verbose: bool) -> bool:
382          """Display which files would be deleted and prompt for confirmation"""
383  
384          def _display(msg: str, paths: Iterable[str]) -> None:
385              if not paths:
386                  return
387  
388              logger.info(msg)
389              with indent_log():
390                  for path in sorted(compact(paths)):
391                      logger.info(path)
392  
393          if not verbose:
394              will_remove, will_skip = compress_for_output_listing(self._paths)
395          else:
396              # In verbose mode, display all the files that are going to be
397              # deleted.
398              will_remove = set(self._paths)
399              will_skip = set()
400  
401          _display("Would remove:", will_remove)
402          _display("Would not remove (might be manually added):", will_skip)
403          _display("Would not remove (outside of prefix):", self._refuse)
404          if verbose:
405              _display("Will actually move:", compress_for_rename(self._paths))
406  
407          return ask("Proceed (Y/n)? ", ("y", "n", "")) != "n"
408  
409      def rollback(self) -> None:
410          """Rollback the changes previously made by remove()."""
411          if not self._moved_paths.can_rollback:
412              logger.error(
413                  "Can't roll back %s; was not uninstalled",
414                  self._dist.raw_name,
415              )
416              return
417          logger.info("Rolling back uninstall of %s", self._dist.raw_name)
418          self._moved_paths.rollback()
419          for pth in self._pth.values():
420              pth.rollback()
421  
422      def commit(self) -> None:
423          """Remove temporary save dir: rollback will no longer be possible."""
424          self._moved_paths.commit()

426      @classmethod
427      def from_dist(cls, dist: BaseDistribution) -> "UninstallPathSet":
428          dist_location = dist.location
429          info_location = dist.info_location
430          if dist_location is None:
431              logger.info(
432                  "Not uninstalling %s since it is not installed",
433                  dist.canonical_name,
434              )
435              return cls(dist)
436  
437          normalized_dist_location = normalize_path(dist_location)
438          if not dist.local:
439              logger.info(
440                  "Not uninstalling %s at %s, outside environment %s",
441                  dist.canonical_name,
442                  normalized_dist_location,
443                  sys.prefix,
444              )
445              return cls(dist)
446  
447          if normalized_dist_location in {
448              p
449              for p in {sysconfig.get_path("stdlib"), sysconfig.get_path("platstdlib")}
450              if p
451          }:
452              logger.info(
453                  "Not uninstalling %s at %s, as it is in the standard library.",
454                  dist.canonical_name,
455                  normalized_dist_location,
456              )
457              return cls(dist)
458  
459          paths_to_remove = cls(dist)
460          develop_egg_link = egg_link_path_from_location(dist.raw_name)
461  
462          # Distribution is installed with metadata in a "flat" .egg-info
463          # directory. This means it is not a modern .dist-info installation, an
464          # egg, or legacy editable.
465          setuptools_flat_installation = (
466              dist.installed_with_setuptools_egg_info
467              and info_location is not None
468              and os.path.exists(info_location)
469              # If dist is editable and the location points to a ``.egg-info``,
470              # we are in fact in the legacy editable case.
471              and not info_location.endswith(f"{dist.setuptools_filename}.egg-info")
472          )
473  
474          # Uninstall cases order do matter as in the case of 2 installs of the
475          # same package, pip needs to uninstall the currently detected version
476          if setuptools_flat_installation:
477              if info_location is not None:
478                  paths_to_remove.add(info_location)
479              installed_files = dist.iter_declared_entries()
480              if installed_files is not None:
481                  for installed_file in installed_files:
482                      paths_to_remove.add(os.path.join(dist_location, installed_file))
483              # FIXME: need a test for this elif block
484              # occurs with --single-version-externally-managed/--record outside
485              # of pip
486              elif dist.is_file("top_level.txt"):
487                  try:
488                      namespace_packages = dist.read_text("namespace_packages.txt")
489                  except FileNotFoundError:
490                      namespaces = []
491                  else:
492                      namespaces = namespace_packages.splitlines(keepends=False)
493                  for top_level_pkg in [
494                      p
495                      for p in dist.read_text("top_level.txt").splitlines()
496                      if p and p not in namespaces
497                  ]:
498                      path = os.path.join(dist_location, top_level_pkg)
499                      paths_to_remove.add(path)
500                      paths_to_remove.add(f"{path}.py")
501                      paths_to_remove.add(f"{path}.pyc")
502                      paths_to_remove.add(f"{path}.pyo")
503  
504          elif dist.installed_by_distutils:
505              raise UninstallationError(
506                  "Cannot uninstall {!r}. It is a distutils installed project "
507                  "and thus we cannot accurately determine which files belong "
508                  "to it which would lead to only a partial uninstall.".format(
509                      dist.raw_name,
510                  )
511              )
512  
513          elif dist.installed_as_egg:
514              # package installed by easy_install
515              # We cannot match on dist.egg_name because it can slightly vary
516              # i.e. setuptools-0.6c11-py2.6.egg vs setuptools-0.6rc11-py2.6.egg
517              paths_to_remove.add(dist_location)
518              easy_install_egg = os.path.split(dist_location)[1]
519              easy_install_pth = os.path.join(
520                  os.path.dirname(dist_location),
521                  "easy-install.pth",
522              )
523              paths_to_remove.add_pth(easy_install_pth, "./" + easy_install_egg)
524  
525          elif dist.installed_with_dist_info:
526              for path in uninstallation_paths(dist):
527                  paths_to_remove.add(path)
528  
529          elif develop_egg_link:
530              # PEP 660 modern editable is handled in the ``.dist-info`` case
531              # above, so this only covers the setuptools-style editable.
532              with open(develop_egg_link) as fh:
533                  link_pointer = os.path.normcase(fh.readline().strip())
534                  normalized_link_pointer = normalize_path(link_pointer)
535              assert os.path.samefile(
536                  normalized_link_pointer, normalized_dist_location
537              ), (
538                  f"Egg-link {link_pointer} does not match installed location of "
539                  f"{dist.raw_name} (at {dist_location})"
540              )
541              paths_to_remove.add(develop_egg_link)
542              easy_install_pth = os.path.join(
543                  os.path.dirname(develop_egg_link), "easy-install.pth"
544              )
545              paths_to_remove.add_pth(easy_install_pth, dist_location)
546  
547          else:
548              logger.debug(
549                  "Not sure how to uninstall: %s - Check: %s",
550                  dist,
551                  dist_location,
552              )
553  
554          if dist.in_usersite:
555              bin_dir = get_bin_user()
556          else:
557              bin_dir = get_bin_prefix()
558  
559          # find distutils scripts= scripts
560          try:
561              for script in dist.iter_distutils_script_names():
562                  paths_to_remove.add(os.path.join(bin_dir, script))
563                  if WINDOWS:
564                      paths_to_remove.add(os.path.join(bin_dir, f"{script}.bat"))
565          except (FileNotFoundError, NotADirectoryError):
566              pass
567  
568          # find console_scripts and gui_scripts
569          def iter_scripts_to_remove(
570              dist: BaseDistribution,
571              bin_dir: str,
572          ) -> Generator[str, None, None]:
573              for entry_point in dist.iter_entry_points():
574                  if entry_point.group == "console_scripts":
575                      yield from _script_names(bin_dir, entry_point.name, False)
576                  elif entry_point.group == "gui_scripts":
577                      yield from _script_names(bin_dir, entry_point.name, True)
578  
579          for s in iter_scripts_to_remove(dist, bin_dir):
580              paths_to_remove.add(s)
581  
582          return paths_to_remove

585  class UninstallPthEntries:
586      def __init__(self, pth_file: str) -> None:
587          self.file = pth_file
588          self.entries: Set[str] = set()
589          self._saved_lines: Optional[List[bytes]] = None
590  
591      def add(self, entry: str) -> None:
592          entry = os.path.normcase(entry)
593          # On Windows, os.path.normcase converts the entry to use
594          # backslashes.  This is correct for entries that describe absolute
595          # paths outside of site-packages, but all the others use forward
596          # slashes.
597          # os.path.splitdrive is used instead of os.path.isabs because isabs
598          # treats non-absolute paths with drive letter markings like c:foo\bar
599          # as absolute paths. It also does not recognize UNC paths if they don't
600          # have more than "\\sever\share". Valid examples: "\\server\share\" or
601          # "\\server\share\folder".
602          if WINDOWS and not os.path.splitdrive(entry)[0]:
603              entry = entry.replace("\\", "/")
604          self.entries.add(entry)
605  
606      def remove(self) -> None:
607          logger.verbose("Removing pth entries from %s:", self.file)
608  
609          # If the file doesn't exist, log a warning and return
610          if not os.path.isfile(self.file):
611              logger.warning("Cannot remove entries from nonexistent file %s", self.file)
612              return
613          with open(self.file, "rb") as fh:
614              # windows uses '\r\n' with py3k, but uses '\n' with py2.x
615              lines = fh.readlines()
616              self._saved_lines = lines
617          if any(b"\r\n" in line for line in lines):
618              endline = "\r\n"
619          else:
620              endline = "\n"
621          # handle missing trailing newline
622          if lines and not lines[-1].endswith(endline.encode("utf-8")):
623              lines[-1] = lines[-1] + endline.encode("utf-8")
624          for entry in self.entries:
625              try:
626                  logger.verbose("Removing entry: %s", entry)
627                  lines.remove((entry + endline).encode("utf-8"))
628              except ValueError:
629                  pass
630          with open(self.file, "wb") as fh:
631              fh.writelines(lines)

633      def rollback(self) -> bool:
