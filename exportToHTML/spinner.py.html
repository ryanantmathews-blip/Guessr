13   class Spinner:
14       def __init__(
15           self,
16           name: str,
17           text: "RenderableType" = "",
18           *,
19           style: Optional["StyleType"] = None,
20           speed: float = 1.0,
21       ) -> None:
22           """A spinner animation. 
23    
24           Args: 
25               name (str): Name of spinner (run python -m rich.spinner). 
26               text (RenderableType, optional): A renderable to display at the right of the spinner (str or Text typically). Defaults to "". 
27               style (StyleType, optional): Style for spinner animation. Defaults to None. 
28               speed (float, optional): Speed factor for animation. Defaults to 1.0. 
29    
30           Raises: 
31               KeyError: If name isn't one of the supported spinner animations. 
32           """
33           try:
34               spinner = SPINNERS[name]
35           except KeyError:
36               raise KeyError(f"no spinner called {name!r}")
37           self.text: "Union[RenderableType, Text]" = (
38               Text.from_markup(text) if isinstance(text, str) else text
39           )
40           self.frames = cast(List[str], spinner["frames"])[:]
41           self.interval = cast(float, spinner["interval"])
42           self.start_time: Optional[float] = None
43           self.style = style
44           self.speed = speed
45           self.frame_no_offset: float = 0.0
46           self._update_speed = 0.0
47   
48       def __rich_console__(
49           self, console: "Console", options: "ConsoleOptions"
50       ) -> "RenderResult":
51           yield self.render(console.get_time())
52   
53       def __rich_measure__(
54           self, console: "Console", options: "ConsoleOptions"
55       ) -> Measurement:
56           text = self.render(0)
57           return Measurement.get(console, options, text)
58   
59       def render(self, time: float) -> "RenderableType":
60           """Render the spinner for a given time. 
61    
62           Args: 
63               time (float): Time in seconds. 
64    
65           Returns: 
66               RenderableType: A renderable containing animation frame. 
67           """
68           if self.start_time is None:
69               self.start_time = time
70   
71           frame_no = ((time - self.start_time) * self.speed) / (
72               self.interval / 1000.0
73           ) + self.frame_no_offset
74           frame = Text(
75               self.frames[int(frame_no) % len(self.frames)], style=self.style or ""
76           )
77   
78           if self._update_speed:
79               self.frame_no_offset = frame_no
80               self.start_time = time
81               self.speed = self._update_speed
82               self._update_speed = 0.0
83   
84           if not self.text:
85               return frame
86           elif isinstance(self.text, (str, Text)):
87               return Text.assemble(frame, " ", self.text)
88           else:
89               table = Table.grid(padding=1)
90               table.add_row(frame, self.text)
91               return table
92   
93       def update(
94           self,
95           *,
96           text: "RenderableType" = "",
97           style: Optional["StyleType"] = None,
98           speed: Optional[float] = None,
99       ) -> None:
100          """Updates attributes of a spinner after it has been started. 
101   
102          Args: 
103              text (RenderableType, optional): A renderable to display at the right of the spinner (str or Text typically). Defaults to "". 
104              style (StyleType, optional): Style for spinner animation. Defaults to None. 
105              speed (float, optional): Speed factor for animation. Defaults to None. 
106          """
107          if text:
108              self.text = Text.from_markup(text) if isinstance(text, str) else text
109          if style:
110              self.style = style
111          if speed:
112              self._update_speed = speed
